64  subroutine re2_file_read(this, data)
65  class(re2_file_t) :: this
66  class(*), target, intent(inout) :: data
67  type(mesh_t), pointer :: msh
68  character(len=5) :: hdr_ver
69  character(len=54) :: hdr_str
70  integer :: nel, ndim, nelv, ierr
71  type(mpi_status) :: status
72  type(mpi_file) :: fh
73  integer (kind=MPI_OFFSET_KIND) :: mpi_offset
74  real(kind=sp) :: test
75  real(kind=dp) :: t2
76  integer :: ncurv, nbcs
77  type(linear_dist_t) :: dist
78  type(map_t) :: nm
79  type(map_file_t) :: map_file
80  character(len=1024) :: map_fname
81  logical :: read_map
82  integer :: re2_data_xy_size
83  integer :: re2_data_xyz_size
84  integer :: re2_data_cv_size
85  integer :: re2_data_bc_size
86  logical :: v2_format
87  character(len=LOG_SIZE) :: log_buf
88 
89  select type(data)
90  type is (mesh_t)
91  msh => data
92  class default
93  call neko_error('Invalid output data')
94  end select
95 
96  v2_format = .false.
97  open(unit=9,file=trim(this%fname), status='old', iostat=ierr)
98  call neko_log%message('Reading binary NEKTON file ' // this%fname)
99  read(9, '(a5,i9,i3,i9,a54)') hdr_ver, nel, ndim, nelv, hdr_str
100  if (hdr_ver .eq. '#v002' .or. hdr_ver .eq. '#v003') then
101  v2_format = .true.
102  call mpi_type_size(mpi_re2v2_data_xy, re2_data_xy_size, ierr)
103  call mpi_type_size(mpi_re2v2_data_xyz, re2_data_xyz_size, ierr)
104  call mpi_type_size(mpi_re2v2_data_cv, re2_data_cv_size, ierr)
105  call mpi_type_size(mpi_re2v2_data_bc, re2_data_bc_size, ierr)
106  else
107  call mpi_type_size(mpi_re2v1_data_xy, re2_data_xy_size, ierr)
108  call mpi_type_size(mpi_re2v1_data_xyz, re2_data_xyz_size, ierr)
109  call mpi_type_size(mpi_re2v1_data_cv, re2_data_cv_size, ierr)
110  call mpi_type_size(mpi_re2v1_data_bc, re2_data_bc_size, ierr)
111  end if
112 
113  write(log_buf,1) ndim, nelv
114 1 format('gdim = ', i1, ', nelements =', i7)
115  call neko_log%message(log_buf)
116  close(9)
117 
118  call filename_chsuffix(this%fname, map_fname,'map')
119 
120  inquire(file=map_fname, exist=read_map)
121  if (read_map) then
122  call map_init(nm, nelv, 2**ndim)
123  call map_file%init(map_fname)
124  call map_file%read(nm)
125  else
126  call neko_log%warning('No NEKTON map file found')
127  end if
128 
129  call mpi_file_open(neko_comm, trim(this%fname), &
130  mpi_mode_rdonly, mpi_info_null, fh, ierr)
131 
132  if (ierr .ne. 0) then
133  call neko_log%error("Can't open binary NEKTON file ")
134  end if
135  dist = linear_dist_t(nelv, pe_rank, pe_size, neko_comm)
136 
137  call mesh_init(msh, ndim, dist)
138 
139  ! Set offset (header)
140  mpi_offset = re2_hdr_size * mpi_character_size
141 
142  call mpi_file_read_at_all(fh, mpi_offset, test, 1, mpi_real, status, ierr)
143  mpi_offset = mpi_offset + mpi_real_size
144 
145  if (abs(re2_endian_test - test) .gt. 1e-4) then
146  call neko_error('Invalid endian of re2 file, byte swap not implemented yet')
147  end if
148 
149  call re2_file_read_points(msh, ndim, nel, dist, fh, &
150  mpi_offset, re2_data_xy_size, re2_data_xyz_size, v2_format)
151 
152 
153  ! Set offset to start of curved side data
154  mpi_offset = re2_hdr_size * mpi_character_size + mpi_real_size
155  if (ndim .eq. 2) then
156  mpi_offset = mpi_offset + dist%num_global() * re2_data_xy_size
157  else
158  mpi_offset = mpi_offset + dist%num_global() * re2_data_xyz_size
159  end if
160 
163  if (v2_format) then
164  call mpi_file_read_at_all(fh, mpi_offset, t2, 1, mpi_double_precision, status, ierr)
165  ncurv = t2
166  mpi_offset = mpi_offset + mpi_double_precision_size
167  call re2_file_read_curve(msh, ncurv, dist, fh, mpi_offset, v2_format)
168  mpi_offset = mpi_offset + ncurv * re2_data_cv_size
169  call mpi_file_read_at_all(fh, mpi_offset, t2, 1, mpi_double_precision, status, ierr)
170  nbcs = t2
171  mpi_offset = mpi_offset + mpi_double_precision_size
172 
173  call re2_file_read_bcs(msh, nbcs, dist, fh, mpi_offset, v2_format)
174  else
175  call mpi_file_read_at_all(fh, mpi_offset, ncurv, 1, mpi_integer, status, ierr)
176  mpi_offset = mpi_offset + mpi_integer_size
177  call re2_file_read_curve(msh, ncurv, dist, fh, mpi_offset, v2_format)
178  mpi_offset = mpi_offset + ncurv * re2_data_cv_size
179  call mpi_file_read_at_all(fh, mpi_offset, nbcs, 1, mpi_integer, status, ierr)
180  mpi_offset = mpi_offset + mpi_integer_size
181 
182  call re2_file_read_bcs(msh, nbcs, dist, fh, mpi_offset, v2_format)
183 
184  end if
185 
186  call mpi_file_close(fh, ierr)
187  call mesh_finalize(msh)
188 
189 
190  call neko_log%message('Done')
191 
192 
193  end subroutine re2_file_read

195  subroutine re2_file_write(this, data, t)
196  class(re2_file_t), intent(inout) :: this
197  class(*), target, intent(in) :: data
198  real(kind=rp), intent(in), optional :: t
199  type(re2v1_xy_t), allocatable :: re2_data_xy(:)
200  type(re2v1_xyz_t), allocatable :: re2_data_xyz(:)
201  type(mesh_t), pointer :: msh
202  character(len=5), parameter :: RE2_HDR_VER = '#v001'
203  character(len=54), parameter :: RE2_HDR_STR = 'RE2 exported by NEKO'
204  integer :: i, j, ierr, nelgv
205  type(mpi_status) :: status
206  type(mpi_file) :: fh
207  integer (kind=MPI_OFFSET_KIND) :: mpi_offset
208  integer :: element_offset
209  integer :: re2_data_xy_size
210  integer :: re2_data_xyz_size
211  character(len=LOG_SIZE) :: log_buf
212 
213  select type(data)
214  type is (mesh_t)
215  msh => data
216  class default
217  call neko_error('Invalid output data')
218  end select
219 
220  call mpi_type_size(mpi_re2v1_data_xy, re2_data_xy_size, ierr)
221  call mpi_type_size(mpi_re2v1_data_xyz, re2_data_xyz_size, ierr)
222  call mpi_reduce(msh%nelv, nelgv, 1, &
223  mpi_integer, mpi_sum, 0, neko_comm, ierr)
224  element_offset = 0
225  call mpi_exscan(msh%nelv, element_offset, 1, &
226  mpi_integer, mpi_sum, neko_comm, ierr)
227 
228  call neko_log%message('Writing data as a binary NEKTON file ' // this%fname)
229 
230  if (pe_rank .eq. 0) then
231  open(unit=9,file=trim(this%fname), status='new', iostat=ierr)
232  write(9, '(a5,i9,i3,i9,a54)') re2_hdr_ver, nelgv, msh%gdim,&
233  nelgv, re2_hdr_str
234  close(9)
235  end if
236 
237  call mpi_barrier(neko_comm, ierr)
238  call mpi_file_open(neko_comm, trim(this%fname), &
239  mpi_mode_wronly + mpi_mode_create, mpi_info_null, fh, ierr)
240  mpi_offset = re2_hdr_size * mpi_character_size
241 
242  call mpi_file_write_at(fh, mpi_offset, re2_endian_test, 1, &
243  mpi_real, status, ierr)
244  mpi_offset = mpi_offset + mpi_real_size
245 
246  if (msh%gdim .eq. 2) then
247  allocate(re2_data_xy(msh%nelv))
248  do i = 1, msh%nelv
249  re2_data_xy(i)%rgroup = 1.0 ! Not used
250  do j = 1, 4
251  re2_data_xy(i)%x(j) = real(msh%elements(i)%e%pts(j)%p%x(1))
252  re2_data_xy(i)%y(j) = real(msh%elements(i)%e%pts(j)%p%x(2))
253  end do
254  end do
255  mpi_offset = mpi_offset + element_offset * re2_data_xy_size
256  call mpi_file_write_at(fh, mpi_offset, &
257  re2_data_xy, msh%nelv, mpi_re2v1_data_xy, status, ierr)
258 
259  deallocate(re2_data_xy)
260 
261  else if (msh%gdim .eq. 3) then
262  allocate(re2_data_xyz(msh%nelv))
263  do i = 1, msh%nelv
264  re2_data_xyz(i)%rgroup = 1.0 ! Not used
265  do j = 1, 8
266  re2_data_xyz(i)%x(j) = real(msh%elements(i)%e%pts(j)%p%x(1))
267  re2_data_xyz(i)%y(j) = real(msh%elements(i)%e%pts(j)%p%x(2))
268  re2_data_xyz(i)%z(j) = real(msh%elements(i)%e%pts(j)%p%x(3))
269  end do
270  end do
271  mpi_offset = mpi_offset + element_offset * re2_data_xyz_size
272  call mpi_file_write_at(fh, mpi_offset, &
273  re2_data_xyz, msh%nelv, mpi_re2v1_data_xyz, status, ierr)
274 
275  deallocate(re2_data_xyz)
276  else
277  call neko_error("Invalid dimension of mesh")
278  end if
279 
280  call mpi_file_close(fh, ierr)
281  call neko_log%message('Done')
282 
284 
285  end subroutine re2_file_write

287  subroutine re2_file_read_points(msh, ndim, nel, dist, fh, &
288  mpi_offset, re2_data_xy_size, re2_data_xyz_size, v2_format)
289  type(mesh_t), intent(inout) :: msh
290  integer (kind=MPI_OFFSET_KIND) :: mpi_offset
291  integer, intent(inout) :: ndim
292  integer, intent(inout) :: nel
293  type(mpi_file), intent(inout) :: fh
294  integer, intent(in) :: re2_data_xy_size
295  integer, intent(in) :: re2_data_xyz_size
296  logical, intent(in) :: v2_format
297  type(linear_dist_t) :: dist
298  integer :: element_offset
299  type(re2v1_xy_t), allocatable :: re2v1_data_xy(:)
300  type(re2v1_xyz_t), allocatable :: re2v1_data_xyz(:)
301  type(re2v2_xy_t), allocatable :: re2v2_data_xy(:)
302  type(re2v2_xyz_t), allocatable :: re2v2_data_xyz(:)
303  type(mpi_status) :: status
304  type(htable_pt_t) :: htp
305  type(point_t) :: p(8)
306  integer :: pt_idx, nelv
307  integer :: i, j, ierr
308 
309 
310  nelv = dist%num_local()
311  element_offset = dist%start_idx()
312 
313  call htp%init(2**ndim * nel, ndim)
314  pt_idx = 0
315  if (ndim .eq. 2) then
316  mpi_offset = mpi_offset + element_offset * re2_data_xy_size
317  if (.not. v2_format) then
318  allocate(re2v1_data_xy(nelv))
319  call mpi_file_read_at_all(fh, mpi_offset, &
320  re2v1_data_xy, nelv, mpi_re2v1_data_xy, status, ierr)
321  do i = 1, nelv
322  do j = 1, 4
323  p(j) = point_t(real(re2v1_data_xy(i)%x(j),dp), &
324  real(re2v1_data_xy(i)%y(j),dp), 0.0d0)
325  call re2_file_add_point(htp, p(j), pt_idx)
326  end do
327  if(mod(i,nelv/10) .eq. 0) write(*,*) i, 'elements read'
328 
329  call mesh_add_element(msh, i, p(1), p(2), p(3), p(4))
330  end do
331  deallocate(re2v1_data_xy)
332  else
333  allocate(re2v2_data_xy(nelv))
334  call mpi_file_read_at_all(fh, mpi_offset, &
335  re2v2_data_xy, nelv, mpi_re2v2_data_xy, status, ierr)
336  do i = 1, nelv
337  do j = 1, 4
338  p(j) = point_t(re2v2_data_xy(i)%x(j), &
339  re2v2_data_xy(i)%y(j), 0.0d0)
340  call re2_file_add_point(htp, p(j), pt_idx)
341  end do
342  if(mod(i,nelv/10) .eq. 0) write(*,*) i, 'elements read'
343 
344  call mesh_add_element(msh, i, p(1), p(2), p(3), p(4))
345  end do
346  deallocate(re2v2_data_xy)
347  end if
348  else if (ndim .eq. 3) then
349  mpi_offset = mpi_offset + element_offset * re2_data_xyz_size
350  if (.not. v2_format) then
351  allocate(re2v1_data_xyz(nelv))
352  call mpi_file_read_at_all(fh, mpi_offset, &
353  re2v1_data_xyz, nelv, mpi_re2v1_data_xyz, status, ierr)
354  do i = 1, nelv
355  do j = 1, 8
356  p(j) = point_t(real(re2v1_data_xyz(i)%x(j),dp), &
357  real(re2v1_data_xyz(i)%y(j),dp),&
358  real(re2v1_data_xyz(i)%z(j),dp))
359  call re2_file_add_point(htp, p(j), pt_idx)
360  end do
361  if(mod(i,nelv/10) .eq. 0) write(*,*) i, 'elements read'
362  call mesh_add_element(msh, i, &
363  p(1), p(2), p(3), p(4), p(5), p(6), p(7), p(8))
364  end do
365  deallocate(re2v1_data_xyz)
366  else
367  allocate(re2v2_data_xyz(nelv))
368  call mpi_file_read_at_all(fh, mpi_offset, &
369  re2v2_data_xyz, nelv, mpi_re2v2_data_xyz, status, ierr)
370  do i = 1, nelv
371  do j = 1, 8
372  p(j) = point_t(re2v2_data_xyz(i)%x(j), &
373  re2v2_data_xyz(i)%y(j),&
374  re2v2_data_xyz(i)%z(j))
375  call re2_file_add_point(htp, p(j), pt_idx)
376  end do
377 
378  if(mod(i,nelv/10) .eq. 0) write(*,*) i, 'elements read'
379  call mesh_add_element(msh, i, &
380  p(1), p(2), p(3), p(4), p(5), p(6), p(7), p(8))
381  end do
382  deallocate(re2v2_data_xyz)
383  end if
384  end if
385 
386  call htp%free()
387  end subroutine re2_file_read_points

389  subroutine re2_file_read_curve(msh, ncurve, dist, fh, mpi_offset, v2_format)
390  type(mesh_t), intent(inout) :: msh
391  integer (kind=MPI_OFFSET_KIND) :: mpi_offset
392  integer, intent(inout) :: ncurve
393  type(linear_dist_t) :: dist
394  type(mpi_file), intent(inout) :: fh
395  logical, intent(in) :: v2_format
396  type(mpi_status) :: status
397  integer :: i, j, l, ierr, el_idx, id
398  type(re2v1_curve_t), allocatable :: re2v1_data_curve(:)
399  type(re2v2_curve_t), allocatable :: re2v2_data_curve(:)
400  real(kind=dp), allocatable :: curve_data(:,:,:)
401  integer, allocatable :: curve_type(:,:)
402  logical, allocatable :: curve_element(:)
403  character(len=1) :: chtemp
404  logical :: curve_skip = .false.
405 
406  allocate(curve_data(5,12,msh%nelv))
407  allocate(curve_element(msh%nelv))
408  allocate(curve_type(12,msh%nelv))
409  do i = 1, msh%nelv
410  curve_element(i) = .false.
411  do j = 1, 12
412  curve_type(j,i) = 0
413  do l = 1, 5
414  curve_data(l,j,i) = 0d0
415  end do
416  end do
417  end do
418 
419  if (.not. v2_format) then
420  allocate(re2v1_data_curve(ncurve))
421  call mpi_file_read_at_all(fh, mpi_offset, re2v1_data_curve, ncurve, &
422  mpi_re2v1_data_cv, status, ierr)
423  else
424  allocate(re2v2_data_curve(ncurve))
425  call mpi_file_read_at_all(fh, mpi_offset, re2v2_data_curve, ncurve, &
426  mpi_re2v2_data_cv, status, ierr)
427  end if
428  !This can probably be made nicer...
429  do i = 1, ncurve
430  if(v2_format) then
431  el_idx = re2v2_data_curve(i)%elem - dist%start_idx()
432  id = re2v2_data_curve(i)%zone
433  chtemp = re2v2_data_curve(i)%type
434  do j = 1, 5
435  curve_data(j,id, el_idx) = re2v2_data_curve(i)%point(j)
436  enddo
437  else
438  el_idx = re2v1_data_curve(i)%elem - dist%start_idx()
439  id = re2v1_data_curve(i)%zone
440  chtemp = re2v1_data_curve(i)%type
441  do j = 1, 5
442  curve_data(j,id, el_idx) = real(re2v1_data_curve(i)%point(j),dp)
443  enddo
444  end if
445 
446  curve_element(el_idx) = .true.
447  !This might need to be extended
448  select case(trim(chtemp))
449  case ('s')
450  curve_type(id,el_idx) = 1
451  call neko_log%warning('curve type s not supported, treating mesh as non-curved')
452  curve_skip = .true.
453  exit
454  case ('e')
455  curve_type(id,el_idx) = 2
456  call neko_log%warning('curve type e not supported, treating mesh as non-curved')
457  curve_skip = .true.
458  exit
459  case ('C')
460  curve_type(id,el_idx) = 3
461  case ('m')
462  curve_type(id,el_idx) = 4
463  case default
464  write(*,*) chtemp, 'curve type not supported yet, treating mesh as non-curved',id, el_idx
465 
466  curve_skip = .true.
467  end select
468  end do
469 
470  if( v2_format) then
471  deallocate(re2v2_data_curve)
472  else
473  deallocate(re2v1_data_curve)
474  end if
475  if (.not. curve_skip) then
476  do el_idx = 1, msh%nelv
477  if (curve_element(el_idx)) then
478  call mesh_mark_curve_element(msh, el_idx, curve_data(1,1,el_idx), curve_type(1,el_idx))
479  end if
480  end do
481  end if
482 
483  deallocate(curve_data)
484  deallocate(curve_element)
485  deallocate(curve_type)
486  end subroutine re2_file_read_curve

489  subroutine re2_file_read_bcs(msh, nbcs, dist, fh, mpi_offset, v2_format)
490  type(mesh_t), intent(inout) :: msh
491  integer (kind=MPI_OFFSET_KIND) :: mpi_offset
492  integer, intent(inout) :: nbcs
493  type(linear_dist_t) :: dist
494  type(mpi_file), intent(inout) :: fh
495  logical, intent(in) :: v2_format
496  type(mpi_status) :: status
497  integer :: pids(4)
498  integer :: sym_facet, label
499  integer :: p_el_idx, p_facet
500  integer :: i, j, ierr, el_idx
501  integer, parameter, dimension(6) :: facet_map = (/3, 2, 4, 1, 5, 6/)
502  logical :: periodic
503  type(re2v1_bc_t), allocatable :: re2v1_data_bc(:)
504  type(re2v2_bc_t), allocatable :: re2v2_data_bc(:)
505 
506  if (.not. v2_format) then
507  allocate(re2v1_data_bc(nbcs))
508  call mpi_file_read_at_all(fh, mpi_offset, re2v1_data_bc, nbcs, &
509  mpi_re2v1_data_bc, status, ierr)
510  else
511  allocate(re2v2_data_bc(nbcs))
512  call mpi_file_read_at_all(fh, mpi_offset, re2v2_data_bc, nbcs, &
513  mpi_re2v2_data_bc, status, ierr)
514  end if
515 
516  periodic = .false.
517 
519  if (v2_format) then ! V2 format
520  do i = 1, nbcs
521  el_idx = int(re2v2_data_bc(i)%elem) - dist%start_idx()
522  sym_facet = facet_map(int(re2v2_data_bc(i)%face))
