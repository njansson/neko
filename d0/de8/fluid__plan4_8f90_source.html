124  subroutine fluid_plan4_init(this, msh, lx, param)

139  associate(xh_lx => this%Xh%lx, xh_ly => this%Xh%ly, xh_lz => this%Xh%lz, &
140  dm_xh => this%dm_Xh, nelv => this%msh%nelv)

143  call field_init(this%u_res, dm_xh, "u_res")
144  call field_init(this%v_res, dm_xh, "v_res")
145  call field_init(this%w_res, dm_xh, "w_res")
146  call field_init(this%abx1, dm_xh, "abx1")
147  call field_init(this%aby1, dm_xh, "aby1")
148  call field_init(this%abz1, dm_xh, "abz1")
149 
150  call field_init(this%abx2, dm_xh, "abx2")
151  call field_init(this%aby2, dm_xh, "aby2")
152  call field_init(this%abz2, dm_xh, "abz2")
153 
154  call field_init(this%u_e, dm_xh, 'u_e')
155  call field_init(this%v_e, dm_xh, 'v_e')
156  call field_init(this%w_e, dm_xh, 'w_e')
157 
158  call field_init(this%wa1, dm_xh, 'wa1')
159  call field_init(this%wa2, dm_xh, 'wa2')
160  call field_init(this%wa3, dm_xh, 'wa3')

162  call field_init(this%ta1, dm_xh, 'ta1')
163  call field_init(this%ta2, dm_xh, 'ta2')
164  call field_init(this%ta3, dm_xh, 'ta3')
165 
166  call field_init(this%du, dm_xh, 'du')
167  call field_init(this%dv, dm_xh, 'dv')
168  call field_init(this%dw, dm_xh, 'dw')
169  call field_init(this%dp, dm_xh, 'dp')

189  call this%bc_vel_residual%mark_zone(msh%wall)

191  call this%bc_vel_residual%set_g(0.0_rp)
192  call bc_list_init(this%bclst_vel_residual)
193  call bc_list_add(this%bclst_vel_residual, this%bc_vel_residual)
194

197

199  this%flow_dir = param%vol_flow_dir

206  call field_init(this%p_vol, this%dm_Xh, 'p_vol')
207 
208  ! Add lagged term to checkpoint
209  call this%chkp%add_lag(this%ulag, this%vlag, this%wlag)
210  call advection_factory(this%adv, this%c_Xh, param%dealias, param%lxd)

272  subroutine fluid_plan4_step(this, t, tstep, ab_bdf)
273  class(fluid_plan4_t), intent(inout) :: this
274  real(kind=rp), intent(inout) :: t
275  type(abbdf_t), intent(inout) :: ab_bdf
276  integer, intent(inout) :: tstep
277  integer :: n, niter
278  type(ksp_monitor_t) :: ksp_results(4)
279  real(kind=rp), parameter :: one = 1.0
280  n = this%dm_Xh%n_dofs
281  niter = 1000
282 
283  associate(u => this%u, v => this%v, w => this%w, p => this%p, &
284  du => this%du, dv => this%dv, dw => this%dw, dp => this%dp, &
285  u_e => this%u_e, v_e => this%v_e, w_e => this%w_e, &
286  ta1 => this%ta1, ta2 => this%ta2, ta3 => this%ta3, &
287  u_res =>this%u_res, v_res => this%v_res, w_res => this%w_res, &
288  p_res => this%p_res, ax => this%Ax, f_xh => this%f_Xh, xh => this%Xh, &
289  c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
290  ulag => this%ulag, vlag => this%vlag, wlag => this%wlag, &
291  params => this%params, msh => this%msh)
292 
293  call fluid_plan4_sumab(u_e%x, u%x, ulag ,n, ab_bdf%ab, ab_bdf%nab)
294  call fluid_plan4_sumab(v_e%x, v%x, vlag ,n, ab_bdf%ab, ab_bdf%nab)
295  if (msh%gdim .eq. 3) then
296  call fluid_plan4_sumab(w_e%x, w%x, wlag,n, ab_bdf%ab, ab_bdf%nab)
297  end if
298 
299  call f_xh%eval()
300  call opcolv(f_xh%u, f_xh%v, f_xh%w, c_xh%B, msh%gdim, n)
301  call this%adv%apply(this%u, this%v, this%w, &
302  f_xh%u, f_xh%v, f_xh%w, &
303  xh, this%c_Xh, dm_xh%n_dofs)
304 
305  call makeabf(ta1, ta2, ta3,&
306  this%abx1, this%aby1, this%abz1,&
307  this%abx2, this%aby2, this%abz2, &
308  f_xh%u, f_xh%v, f_xh%w,&
309  params%rho, ab_bdf%ab, n, msh%gdim)
310  call makebdf(ta1, ta2, ta3,&
311  this%wa1, this%wa2, this%wa3,&
312  c_xh%h2, ulag, vlag, wlag, &
313  f_xh%u, f_xh%v, f_xh%w, u, v, w,&
314  c_xh%B, params%rho, params%dt, &
315  ab_bdf%bd, ab_bdf%nbd, n, msh%gdim)
316 
317  call ulag%update()
318  call vlag%update()
319  call wlag%update()
320 
321  ! mask Dirichlet boundaries (velocity)
322  call this%bc_apply_vel()
323 
324  ! compute pressure
325  call this%bc_apply_prs()
326  call fluid_plan4_pres_setup(c_xh%h1, c_xh%h2, params%rho, &
327  dm_xh%n_dofs, c_xh%ifh2)
328  call fluid_plan4_pres_residual(p, p_res%x, u, v, w, &
329  u_e, v_e, w_e, &
330  ta1, ta2, ta3, &
331  this%wa1, this%wa2, this%wa3, &
332  this%work1, this%work2, f_xh, &
333  c_xh, gs_xh, this%bc_prs_surface, &
334  ax, ab_bdf%bd(1), params%dt, &
335  params%Re, params%rho)
336 
337  !Sets tolerances
338  !call ctolspl (tolspl,respr)
339  call gs_op(gs_xh, p_res, gs_op_add)
340  call bc_list_apply_scalar(this%bclst_prs, p_res%x, p%dof%n_dofs)
341 
342  if( tstep .gt. 5) call this%proj%project_on(p_res%x, c_xh, n)
343  call this%pc_prs%update()
344  ksp_results(1) = this%ksp_prs%solve(ax, dp, p_res%x, n, c_xh, &
345  this%bclst_prs, gs_xh, niter)
346  if( tstep .gt. 5) call this%proj%project_back(dp%x, ax, c_xh, &
347  this%bclst_prs, gs_xh, n)
348  call add2(p%x, dp%x,n)
349  ! call ortho(this%p%x,n,this%Xh%lxyz*this%msh%glb_nelv)
350 
351  !We only need to update h2 once I think then use the flag to switch on/off
352  call fluid_plan4_vel_setup(c_xh%h1, c_xh%h2, &
353  params%Re, params%rho, ab_bdf%bd(1), &
354  params%dt, dm_xh%n_dofs, c_xh%ifh2)
355 
356  call fluid_plan4_vel_residual(ax, u, v, w, &
357  u_res, v_res, w_res, &
358  p, ta1, ta2, ta3, &
359  f_xh, c_xh, msh, xh, dm_xh%n_dofs)
360 
361  call gs_op(gs_xh, u_res, gs_op_add)
362  call gs_op(gs_xh, v_res, gs_op_add)
363  call gs_op(gs_xh, w_res, gs_op_add)
364 
365  call bc_list_apply_vector(this%bclst_vel_residual,&
366  u_res%x, v_res%x, w_res%x, dm_xh%n_dofs)
367  call this%pc_vel%update()
368 
369  ksp_results(2) = this%ksp_vel%solve(ax, du, u_res%x, n, &
370  c_xh, this%bclst_vel_residual, gs_xh, niter)
371  ksp_results(3) = this%ksp_vel%solve(ax, dv, v_res%x, n, &
372  c_xh, this%bclst_vel_residual, gs_xh, niter)
373  ksp_results(4) = this%ksp_vel%solve(ax, dw, w_res%x, n, &
374  c_xh, this%bclst_vel_residual, gs_xh, niter)
375 
376  call opadd2cm(u%x, v%x, w%x, du%x, dv%x, dw%x, one, n, msh%gdim)
377 
378  if (this%flow_dir .ne. 0) then
379  call plan4_vol_flow(this, ab_bdf, niter)
380  end if
381 
382  call fluid_step_info(tstep, t, params%dt, ksp_results)
383 
384  end associate
385  end subroutine fluid_plan4_step
386 
387  subroutine fluid_plan4_pres_setup(h1, h2, rho, n, ifh2)
388  integer, intent(in) :: n
389  real(kind=rp), intent(inout) :: h1(n)
390  real(kind=rp), intent(inout) :: h2(n)
391  real(kind=rp), intent(in) :: rho
392  real(kind=rp), parameter :: one = 1.0_rp
393  logical, intent(inout) :: ifh2
394  call cfill(h1, one /rho, n)
395  call rzero(h2, n)
396  ifh2 = .false.
397  end subroutine fluid_plan4_pres_setup
398 
399  subroutine fluid_plan4_vel_setup(h1, h2, Re, rho, bd, dt, n, ifh2)
400  integer, intent(in) :: n
401  real(kind=rp), intent(inout) :: h1(n)
402  real(kind=rp), intent(inout) :: h2(n)
403  real(kind=rp), intent(in) :: re
404  real(kind=rp), intent(in) :: rho
405  real(kind=rp), intent(in) :: bd
406  real(kind=rp), intent(in) :: dt
407  logical, intent(inout) :: ifh2
408  real(kind=rp), parameter :: one = 1.0_rp
409  real(kind=rp) :: dtbd
410  dtbd = rho * (bd / dt)
411  h1 = (one / re)
412  h2 = dtbd
413  ifh2 = .true.
414  end subroutine fluid_plan4_vel_setup
415 
416  subroutine fluid_plan4_vel_residual(Ax, u, v, w, u_res, v_res, w_res, &
417  p, ta1, ta2, ta3, f_Xh, c_Xh, msh, Xh, n)
418  class(ax_t), intent(in) :: ax
419  type(mesh_t), intent(inout) :: msh
420  type(space_t), intent(inout) :: Xh
421  type(field_t), intent(inout) :: p, u, v, w
422  type(field_t), intent(inout) :: u_res, v_res, w_res
423  type(field_t), intent(inout) :: ta1, ta2, ta3
424  type(source_t), intent(inout) :: f_Xh
425  type(coef_t), intent(inout) :: c_Xh
426  integer, intent(in) :: n
427 
428  call ax%compute(u_res%x, u%x, c_xh, msh, xh)
429  call ax%compute(v_res%x, v%x, c_xh, msh, xh)
430  if (msh%gdim .eq. 3) then
431  call ax%compute(w_res%x, w%x, c_xh, msh, xh)
432  end if
433 
434  call opchsign(u_res%x, v_res%x, w_res%x, msh%gdim, n)
435 
436  call opgrad(ta1%x, ta2%x, ta3%x, p%x, c_xh)
437 
438  call opadd2cm(u_res%x, v_res%x, w_res%x, &
439  ta1%x, ta2%x, ta3%x, -1.0_rp, n, msh%gdim)
440 
441  call opadd2cm(u_res%x, v_res%x, w_res%x, &
442  f_xh%u, f_xh%v, f_xh%w, 1.0_rp, n, msh%gdim)
443 
444  end subroutine fluid_plan4_vel_residual
445 
446  subroutine fluid_plan4_pres_residual(p, p_res, u, v, w, u_e, v_e, w_e, &
447  ta1, ta2, ta3, wa1, wa2, wa3, work1, work2, f_Xh, c_xh, gs_Xh, &
448  bc_prs_surface, Ax, bd, dt, Re, rho)
449  type(field_t), intent(inout) :: p, u, v, w
450  type(field_t), intent(inout) :: u_e, v_e, w_e
451  type(field_t), intent(inout) :: ta1, ta2, ta3
452  type(field_t), intent(inout) :: wa1, wa2, wa3
453  type(field_t), intent(inout) :: work1, work2
454  real(kind=rp), intent(inout) :: p_res(p%dof%n_dofs)
455  type(source_t), intent(inout) :: f_xh
456  type(coef_t), intent(inout) :: c_Xh
457  type(gs_t), intent(inout) :: gs_Xh
458  type(facet_normal_t), intent(inout) :: bc_prs_surface
459  class(ax_t), intent(inout) :: Ax
460  real(kind=rp), intent(inout) :: bd
461  real(kind=rp), intent(in) :: dt
462  real(kind=rp), intent(in) :: re
463  real(kind=rp), intent(in) :: rho
464  real(kind=rp) :: dtbd
465  integer :: n, gdim
466  integer :: i
467 
468  n = c_xh%dof%size()
469  gdim = c_xh%msh%gdim
470 
471  call curl(ta1, ta2, ta3, u_e, v_e, w_e, work1, work2, c_xh)
472  call curl(wa1, wa2, wa3, ta1, ta2, ta3, work1, work2, c_xh)
473  call opcolv(wa1%x, wa2%x, wa3%x, c_xh%B, gdim, n)
474 
475  work1 = (1.0_rp / re) / rho
476  call opcolv(wa1%x, wa2%x, wa3%x, work1%x, gdim, n)
477 
478  call ax%compute(p_res,p%x,c_xh,p%msh,p%Xh)
479  call chsign(p_res, n)
480 
481  do i = 1, n
482  ta1%x(i,1,1,1) = f_xh%u(i,1,1,1) / rho - wa1%x(i,1,1,1)
483  ta2%x(i,1,1,1) = f_xh%v(i,1,1,1) / rho - wa2%x(i,1,1,1)
484  ta3%x(i,1,1,1) = f_xh%w(i,1,1,1) / rho - wa3%x(i,1,1,1)
485  enddo
486 
487  !Need to consider cyclic bcs here...
488  call gs_op(gs_xh, ta1, gs_op_add)
489  call gs_op(gs_xh, ta2, gs_op_add)
490  call gs_op(gs_xh, ta3, gs_op_add)
491 
492  do i = 1, n

716  subroutine plan4_vol_flow(this, ab_bdf, niter)
717 ! Adust flow volume at end of time step to keep flow rate fixed by
718 ! adding an appropriate multiple of the linear solution to the Stokes
719 ! problem arising from a unit forcing in the X-direction. This assumes
720 ! that the flow rate in the X-direction is to be fixed (as opposed to Y-
721 ! or Z-) *and* that the periodic boundary conditions in the X-direction
722 ! occur at the extreme left and right ends of the mesh.
723 !
724 ! pff 6/28/98
725  type(fluid_plan4_t), intent(inout) :: this
726  type(abbdf_t), intent(inout) :: ab_bdf
727  integer, intent(in) :: niter
728  real(kind=rp) :: ifcomp, flow_rate, xsec
729  real(kind=rp) :: current_flow, delta_flow, base_flow, scale
730  integer :: n, ierr
731 
732  n = this%dm_Xh%n_dofs
733 
734 ! If either dt or the backwards difference coefficient change,
735 ! then recompute base flow solution corresponding to unit forcing:
736 
737  ifcomp = 0.0_rp
738 
739  if (this%params%dt .ne. this%dtlag .or. ab_bdf%bd(1) .ne. this%bdlag) then
740  ifcomp = 1.0_rp
741  end if
742 
743  this%dtlag = this%params%dt
744  this%bdlag = ab_bdf%bd(1)
745 
746  call mpi_allreduce(mpi_in_place, ifcomp, 1, &
747  mpi_real_precision, mpi_sum, neko_comm, ierr)
748  if (ifcomp .gt. 0d0) then
749  call plan4_compute_vol_flow(this, ab_bdf, niter)
750  end if
751 
752  if (this%flow_dir .eq. 1) then
753  current_flow=glsc2(this%u%x,this%c_Xh%B,n)/this%domain_length ! for X
754  else if (this%flow_dir .eq. 2) then
755  current_flow=glsc2(this%v%x,this%c_Xh%B,n)/this%domain_length ! for Y
756  else if (this%flow_dir .eq. 3) then
757  current_flow=glsc2(this%w%x,this%c_Xh%B,n)/this%domain_length ! for Z
758  end if
759 
760  if (this%avflow) then
761  xsec = this%c_Xh%volume / this%domain_length
762  flow_rate = this%flow_rate*xsec
763  endif
764 
765  delta_flow = flow_rate-current_flow
766 
767 ! Note, this scale factor corresponds to FFX, provided FFX has
768 ! not also been specified in userf. If ffx is also specified
769 ! in userf then the true FFX is given by ffx_userf + scale.
770 
771  scale = delta_flow/this%base_flow
772 
773  call add2s2(this%u%x,this%u_vol%x,scale,n)
774  call add2s2(this%v%x,this%v_vol%x,scale,n)
775  call add2s2(this%w%x,this%w_vol%x,scale,n)
776  call add2s2(this%p%x,this%p_vol%x,scale,n)
777  end subroutine plan4_vol_flow
