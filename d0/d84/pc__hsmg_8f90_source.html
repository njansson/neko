128  subroutine hsmg_init(this, msh, Xh, coef, dof, gs_h, bclst, crs_pctype)
129  class(hsmg_t), intent(inout), target :: this
130  type(mesh_t), intent(inout), target :: msh
131  type(space_t), intent(inout), target :: Xh
132  type(coef_t), intent(inout), target :: coef
133  type(dofmap_t), intent(inout), target :: dof
134  type(gs_t), intent(inout), target :: gs_h
135  type(bc_list_t), intent(inout), target :: bclst
136  character(len=*), optional :: crs_pctype
137  integer :: n, i
138 
139  call this%free()
140  if(xh%lx .lt. 5) then
141  call neko_error('Insufficient number of GLL points for hsmg precon. Minimum degree 4 and 5 GLL points required.')
142  end if
143  this%nlvls = 3
144  this%msh => msh
145  allocate(this%grids(this%nlvls))
146  allocate(this%w(dof%n_dofs))
147  allocate(this%r(dof%n_dofs))
148 
149 
150  ! Compute all elements as if they are deformed
151  call mesh_all_deformed(msh)
152 
153  n = dof%n_dofs
154  call field_init(this%e, dof,'work array')
155 
156  call space_init(this%Xh_crs, gll, 2, 2, 2)
157  this%dm_crs = dofmap_t(msh, this%Xh_crs)
158  call gs_init(this%gs_crs, this%dm_crs)
159  call field_init(this%e_crs, this%dm_crs,'work crs')
160  call coef_init(this%c_crs, this%gs_crs)
161 
162  call space_init(this%Xh_mg, gll, 4, 4, 4)
163  this%dm_mg = dofmap_t(msh, this%Xh_mg)
164  call gs_init(this%gs_mg, this%dm_mg)
165  call field_init(this%e_mg, this%dm_mg,'work midl')
166  call coef_init(this%c_mg, this%gs_mg)
167  ! Create backend specific Ax operator
168  call ax_helm_factory(this%ax)
169 
170 
171  ! Create a backend specific preconditioner
172  ! Note we can't call the pc factory since hsmg is a pc...
173  if (neko_bcknd_sx .eq. 1) then
174  allocate(sx_jacobi_t::this%pc_crs)
175  else if (neko_bcknd_xsmm .eq. 1) then
176  allocate(jacobi_t::this%pc_crs)
177  else if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
178  .or. (neko_bcknd_opencl .eq. 1)) then
179  allocate(device_jacobi_t::this%pc_crs)
180  else
181  allocate(jacobi_t::this%pc_crs)
182  end if
183 
184  ! Create a backend specific krylov solver
185  if (present(crs_pctype)) then
186  call krylov_solver_factory(this%crs_solver, &
187  this%dm_crs%n_dofs, trim(crs_pctype), m = this%pc_crs)
188  else
189  call krylov_solver_factory(this%crs_solver, &
190  this%dm_crs%n_dofs, 'cg', m = this%pc_crs)
191  end if
192 
193  call this%bc_crs%init(this%dm_crs)
194  call this%bc_mg%init(this%dm_mg)
195  call this%bc_reg%init(dof)
196  if (bclst%n .gt. 0) then
197  do i = 1, bclst%n
198  call this%bc_reg%mark_facets(bclst%bc(i)%bcp%marked_facet)
199  call this%bc_crs%mark_facets(bclst%bc(i)%bcp%marked_facet)
200  call this%bc_mg%mark_facets(bclst%bc(i)%bcp%marked_facet)
201  end do
202  end if
203  call this%bc_reg%finalize()
204  call this%bc_reg%set_g(real(0d0,rp))
205  call bc_list_init(this%bclst_reg)
206  call bc_list_add(this%bclst_reg, this%bc_reg)
207 
208  call this%bc_crs%finalize()
209  call this%bc_crs%set_g(real(0d0,rp))
210  call bc_list_init(this%bclst_crs)
211  call bc_list_add(this%bclst_crs, this%bc_crs)
212 
213 
214  call this%bc_mg%finalize()
215  call this%bc_mg%set_g(0.0_rp)
216  call bc_list_init(this%bclst_mg)
217  call bc_list_add(this%bclst_mg, this%bc_mg)
218 
219  call this%schwarz%init(xh, dof, gs_h, this%bclst_reg, msh)
220  call this%schwarz_mg%init(this%Xh_mg, this%dm_mg, this%gs_mg,&
221  this%bclst_mg, msh)
222 
223  call this%interp_fine_mid%init(xh,this%Xh_mg)
224  call this%interp_mid_crs%init(this%Xh_mg,this%Xh_crs)
225 
226  call hsmg_fill_grid(dof, gs_h, xh, coef, this%bclst_reg, this%schwarz, &
227  this%e, this%grids, 3)
228  call hsmg_fill_grid(this%dm_mg, this%gs_mg, this%Xh_mg, this%c_mg, &
229  this%bclst_mg, this%schwarz_mg, this%e_mg, &
230  this%grids, 2)
231  call hsmg_fill_grid(this%dm_crs, this%gs_crs, this%Xh_crs, &
232  this%c_crs, this%bclst_crs, this%schwarz_crs, &
233  this%e_crs, this%grids, 1)
234 
235  call hsmg_set_h(this)
236  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
237  .or. (neko_bcknd_opencl .eq. 1)) then
238  call device_map(this%w, this%w_d, n)
239  call device_map(this%r, this%r_d, n)
240  end if
241  select type(pc => this%pc_crs)
242  type is (jacobi_t)
243  call pc%init(this%c_crs, this%dm_crs, this%gs_crs)
244  type is (sx_jacobi_t)
245  call pc%init(this%c_crs, this%dm_crs, this%gs_crs)
246  type is (device_jacobi_t)
247  call pc%init(this%c_crs, this%dm_crs, this%gs_crs)
248  end select
249 
250 
251 
252  end subroutine hsmg_init

340  subroutine hsmg_solve(this, z, r, n)
341  integer, intent(inout) :: n
342  class(hsmg_t), intent(inout) :: this
343  real(kind=rp), dimension(n), intent(inout) :: z
344  real(kind=rp), dimension(n), intent(inout) :: r
345  type(c_ptr) :: z_d, r_d
346  type(ksp_monitor_t) :: crs_info
347 
348  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
349  .or. (neko_bcknd_opencl .eq. 1)) then
350  z_d = device_get_ptr(z)
351  r_d = device_get_ptr(r)
352  !We should not work with the input
353  call device_copy(this%r_d, r_d, n)
354 
355  !OVERLAPPING Schwarz exchange and solve
356  call this%grids(3)%schwarz%compute(z, this%r)
357  !! DOWNWARD Leg of V-cycle, we are pretty hardcoded here but w/e
358  call device_col2(this%r_d, this%grids(3)%coef%mult_d, &
359  this%grids(3)%dof%n_dofs)
360  !Restrict to middle level
361  call this%interp_fine_mid%map(this%w,this%r,this%msh%nelv,this%grids(2)%Xh)
362  call gs_op(this%grids(2)%gs_h, this%w, &
363  this%grids(2)%dof%n_dofs, gs_op_add)
364  !OVERLAPPING Schwarz exchange and solve
365  call this%grids(2)%schwarz%compute(this%grids(2)%e%x,this%w)
366  call device_col2(this%w_d, this%grids(2)%coef%mult_d, this%grids(2)%dof%n_dofs)
367  !restrict residual to crs
368  call this%interp_mid_crs%map(this%r,this%w,this%msh%nelv,this%grids(1)%Xh)
369  !Crs solve
370 
371  call gs_op(this%grids(1)%gs_h, this%r, &
372  this%grids(1)%dof%n_dofs, gs_op_add)
373  call bc_list_apply_scalar(this%grids(1)%bclst, this%r, &
374  this%grids(1)%dof%n_dofs)
375  crs_info = this%crs_solver%solve(this%Ax, this%grids(1)%e, this%r, &
376  this%grids(1)%dof%n_dofs, &
377  this%grids(1)%coef, &
378  this%grids(1)%bclst, &
379  this%grids(1)%gs_h, this%niter)
380  call bc_list_apply_scalar(this%grids(1)%bclst, this%grids(1)%e%x,&
381  this%grids(1)%dof%n_dofs)
382 
383 
384  call this%interp_mid_crs%map(this%w,this%grids(1)%e%x,this%msh%nelv,this%grids(2)%Xh)
385  call device_add2(this%grids(2)%e%x_d, this%w_d, this%grids(2)%dof%n_dofs)
386 
387  call this%interp_fine_mid%map(this%w,this%grids(2)%e%x,this%msh%nelv,this%grids(3)%Xh)
388  call device_add2(z_d, this%w_d, this%grids(3)%dof%n_dofs)
389  call gs_op(this%grids(3)%gs_h, z, &
390  this%grids(3)%dof%n_dofs, gs_op_add)
391  call device_col2(z_d, this%grids(3)%coef%mult_d, this%grids(3)%dof%n_dofs)
392  else
393  !We should not work with the input
394  call copy(this%r, r, n)
395 
396  !OVERLAPPING Schwarz exchange and solve
397  call this%grids(3)%schwarz%compute(z, this%r)
398  ! DOWNWARD Leg of V-cycle, we are pretty hardcoded here but w/e
399  call col2(this%r, this%grids(3)%coef%mult, &
400  this%grids(3)%dof%n_dofs)
401  !Restrict to middle level
402  call this%interp_fine_mid%map(this%w,this%r,this%msh%nelv,this%grids(2)%Xh)
403  call gs_op(this%grids(2)%gs_h, this%w, &
404  this%grids(2)%dof%n_dofs, gs_op_add)
405  !OVERLAPPING Schwarz exchange and solve
406  call this%grids(2)%schwarz%compute(this%grids(2)%e%x,this%w)
407  call col2(this%w, this%grids(2)%coef%mult, this%grids(2)%dof%n_dofs)
408  !restrict residual to crs
409  call this%interp_mid_crs%map(this%r,this%w,this%msh%nelv,this%grids(1)%Xh)
410  !Crs solve
411 
412  call gs_op(this%grids(1)%gs_h, this%r, &
413  this%grids(1)%dof%n_dofs, gs_op_add)
414  call bc_list_apply_scalar(this%grids(1)%bclst, this%r, &
415  this%grids(1)%dof%n_dofs)
416  crs_info = this%crs_solver%solve(this%Ax, this%grids(1)%e, this%r, &
417  this%grids(1)%dof%n_dofs, &
418  this%grids(1)%coef, &
419  this%grids(1)%bclst, &
420  this%grids(1)%gs_h, this%niter)
421  call bc_list_apply_scalar(this%grids(1)%bclst, this%grids(1)%e%x,&
422  this%grids(1)%dof%n_dofs)
423 
424 
425  call this%interp_mid_crs%map(this%w,this%grids(1)%e%x,this%msh%nelv,this%grids(2)%Xh)
426  call add2(this%grids(2)%e%x, this%w, this%grids(2)%dof%n_dofs)
427 
428  call this%interp_fine_mid%map(this%w,this%grids(2)%e%x,this%msh%nelv,this%grids(3)%Xh)
429  call add2(z, this%w, this%grids(3)%dof%n_dofs)
430  call gs_op(this%grids(3)%gs_h, z, &
431  this%grids(3)%dof%n_dofs, gs_op_add)
432  call col2(z, this%grids(3)%coef%mult, this%grids(3)%dof%n_dofs)
433 
434  end if
435  end subroutine hsmg_solve
436 end module hsmg
