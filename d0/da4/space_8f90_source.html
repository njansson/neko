191  end do
192  end do
193  end do
195  if (t .eq. gll) then
196  call dgll(s%dx, s%dxt, s%zg(1,1), s%lx, s%lx)

201  s%dz = 0d0
202  s%dzt = 0d0
203  end if

205  call setup_intp(s%dx,s%dxt,s%zg(1,1),s%zg(1,1),s%lx,s%lx,1)
206  call setup_intp(s%dy,s%dyt,s%zg(1,2),s%zg(1,2),s%ly,s%ly,1)
207  if (s%lz .gt. 1) then
208  call setup_intp(s%dz,s%dzt,s%zg(1,3),s%zg(1,3),s%lz,s%lz,1)
209  else
210  s%dz = 0d0
211  s%dzt = 0d0
212  end if
213  else
214  call neko_error("Invalid quadrature rule")
215  end if
216 
217  call space_compute_dist(s%dr_inv, s%zg(1,1), s%lx)
218  call space_compute_dist(s%ds_inv, s%zg(1,2), s%ly)
219  if (s%lz .gt. 1) then
220  call space_compute_dist(s%dt_inv, s%zg(1,3), s%lz)
221  else
222  s%dt_inv = 0d0
223  end if
224 
225  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
226  (neko_bcknd_opencl .eq. 1)) then
227  call device_map(s%dr_inv, s%dr_inv_d, s%lx)
228  call device_map(s%ds_inv, s%ds_inv_d, s%lx)
229  call device_map(s%dt_inv, s%dt_inv_d, s%lx)
230  call device_map(s%wx, s%wx_d, s%lx)
231  call device_map(s%wy, s%wy_d, s%lx)
232  call device_map(s%wz, s%wz_d, s%lx)
233  call device_map(s%dx, s%dx_d, s%lxy)
234  call device_map(s%dy, s%dy_d, s%lxy)
235  call device_map(s%dz, s%dz_d, s%lxy)
236  call device_map(s%dxt, s%dxt_d, s%lxy)
237  call device_map(s%dyt, s%dyt_d, s%lxy)
238  call device_map(s%dzt, s%dzt_d, s%lxy)
239  call device_map(s%w3, s%w3_d, s%lxyz)
240 
241  call device_memcpy(s%dr_inv, s%dr_inv_d, s%lx, host_to_device)
242  call device_memcpy(s%ds_inv, s%ds_inv_d, s%lx, host_to_device)
243  call device_memcpy(s%dt_inv, s%dt_inv_d, s%lx, host_to_device)
244  call device_memcpy(s%wx, s%wx_d, s%lx, host_to_device)
245  call device_memcpy(s%wy, s%wy_d, s%lx, host_to_device)
246  call device_memcpy(s%wz, s%wz_d, s%lx, host_to_device)
247  call device_memcpy(s%dx, s%dx_d, s%lxy, host_to_device)
248  call device_memcpy(s%dy, s%dy_d, s%lxy, host_to_device)
249  call device_memcpy(s%dz, s%dz_d, s%lxy, host_to_device)
250  call device_memcpy(s%dxt, s%dxt_d, s%lxy, host_to_device)
251  call device_memcpy(s%dyt, s%dyt_d, s%lxy, host_to_device)
252  call device_memcpy(s%dzt, s%dzt_d, s%lxy, host_to_device)
253  call device_memcpy(s%w3, s%w3_d, s%lxyz, host_to_device)
254 
255  ix = s%lx * 3
256  call device_map(s%zg, s%zg_d, ix)
257  call device_memcpy(s%zg, s%zg_d, ix, host_to_device)

418  subroutine space_compute_dist(dx, x, lx)
419  integer, intent(in) :: lx
420  real(kind=rp), intent(inout) :: dx(lx), x(lx)
421  integer :: i
422  dx(1) = x(2) - x(1)
423  do i = 2, lx - 1
