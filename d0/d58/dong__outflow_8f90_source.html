71  subroutine dong_outflow_set_vars(this, c_Xh, u, v, w, uinf, delta)
72  class(dong_outflow_t), intent(inout) :: this
73  type(coef_t), target, intent(in) :: c_Xh
74  type(field_t), target, intent(in) :: u, v, w
75  real(kind=rp), intent(in) :: uinf
76  real(kind=rp), optional, intent(in) :: delta
77  real(kind=rp), allocatable :: temp_x(:)
78  real(kind=rp), allocatable :: temp_y(:)
79  real(kind=rp), allocatable :: temp_z(:)
80  real(c_rp) :: dummy
81  integer :: i, m, k, facet, idx(4)
82  real(kind=rp) :: normal_xyz(3)
83 
84 
85  if (present(delta)) then
86  this%delta = delta
87  else
88  this%delta = 0.01
89  end if
90  this%uinf = uinf
91  this%u => u
92  this%v => v
93  this%c_Xh=> c_xh
94  this%w => w
95  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1)) then
96  call device_alloc(this%normal_x_d,c_sizeof(dummy)*this%msk(0))
97  call device_alloc(this%normal_y_d,c_sizeof(dummy)*this%msk(0))
98  call device_alloc(this%normal_z_d,c_sizeof(dummy)*this%msk(0))
99  m = this%msk(0)
100  allocate(temp_x(m))
101  allocate(temp_y(m))
102  allocate(temp_z(m))
103  do i = 1, m
104  k = this%msk(i)
105  facet = this%facet(i)
106  idx = nonlinear_index(k,this%Xh%lx, this%Xh%lx,this%Xh%lx)
107  normal_xyz = coef_get_normal(this%c_Xh, &
108  idx(1), idx(2), idx(3), idx(4),facet)
109  temp_x(i) = normal_xyz(1)
110  temp_y(i) = normal_xyz(2)
111  temp_z(i) = normal_xyz(3)
112  end do
113  call device_memcpy(temp_x, this%normal_x_d, m, host_to_device)
114  call device_memcpy(temp_y, this%normal_y_d, m, host_to_device)
115  call device_memcpy(temp_z, this%normal_z_d, m, host_to_device)
116  deallocate( temp_x, temp_y, temp_z)
117  end if
118 
119  end subroutine dong_outflow_set_vars

123  subroutine dong_outflow_apply_scalar(this, x, n)
124  class(dong_outflow_t), intent(inout) :: this
125  integer, intent(in) :: n
126  real(kind=rp), intent(inout), dimension(n) :: x
127  integer :: i, m, k, facet, idx(4)
128  real(kind=rp) :: vn, s0, ux, uy, uz, normal_xyz(3)
129 
130  m = this%msk(0)
131  do i = 1, m
132  k = this%msk(i)
133  facet = this%facet(i)
134  ux = this%u%x(k,1,1,1)
135  uy = this%v%x(k,1,1,1)
136  uz = this%w%x(k,1,1,1)
137  idx = nonlinear_index(k,this%Xh%lx, this%Xh%lx,this%Xh%lx)
138  normal_xyz = coef_get_normal(this%c_Xh, &
139  idx(1), idx(2), idx(3), idx(4),facet)
140  vn = ux*normal_xyz(1) + uy*normal_xyz(2) + uz*normal_xyz(3)
141  s0 = 0.5_rp*(1.0_rp - tanh(vn / (this%uinf * this%delta)))
142 
143  x(k)=-0.5*(ux*ux+uy*uy+uz*uz)*s0
144  end do
145 end subroutine dong_outflow_apply_scalar
