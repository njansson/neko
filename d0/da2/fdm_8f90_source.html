105  subroutine fdm_init(this, Xh, dm, gs_h)
106  class(fdm_t), intent(inout) :: this
107  type(space_t), target, intent(inout) :: Xh
108  type(dofmap_t), target, intent(inout) :: dm
109  type(gs_t), target, intent(inout) :: gs_h
110  !We only really use ah, bh
111  real(kind=rp), dimension((Xh%lx)**2) :: ah, bh, ch, dh, zh
112  real(kind=rp), dimension((Xh%lx)**2) :: dph, jph, bgl, zglhat, dgl, jgl, wh
113  integer :: nl, n, nelv
114 
115  n = xh%lx -1 !Polynomnial degree
116  nl = xh%lx + 2 !Schwarz!
117  nelv = dm%msh%nelv
118  call fdm_free(this)
119  allocate(this%s(nl*nl,2,dm%msh%gdim, dm%msh%nelv))
120  allocate(this%d(nl**3,dm%msh%nelv))
121  allocate(this%swplen(xh%lx, xh%lx, xh%lx,dm%msh%nelv))
122  allocate(this%len_lr(nelv), this%len_ls(nelv), this%len_lt(nelv))
123  allocate(this%len_mr(nelv), this%len_ms(nelv), this%len_mt(nelv))
124  allocate(this%len_rr(nelv), this%len_rs(nelv), this%len_rt(nelv))
125 
126  ! Zeroing here enables easier debugging since then
127  ! MPI messages in GS are deterministic
128  call rzero(this%swplen, xh%lxyz * dm%msh%nelv)
129 
130  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
131  .or. (neko_bcknd_opencl .eq. 1)) then
132  call device_map(this%s, this%s_d,nl*nl*2*dm%msh%gdim*dm%msh%nelv)
133  call device_map(this%d, this%d_d,nl**dm%msh%gdim*dm%msh%nelv)
134  call device_map(this%swplen,this%swplen_d, xh%lxyz*dm%msh%nelv)
135  end if
136 
137  call semhat(ah, bh, ch, dh, zh, dph, jph, bgl, zglhat, dgl, jgl, n, wh)
138  this%Xh => xh
139  this%dof => dm
140  this%gs_h => gs_h
141  this%msh => dm%msh
142 
143  call swap_lengths(this, dm%x, dm%y, dm%z, dm%msh%nelv, dm%msh%gdim)
144 
145  call fdm_setup_fast(this, ah, bh, nl, n)
146 
147  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
148  .or. (neko_bcknd_opencl .eq. 1)) then
149  call device_memcpy(this%s, this%s_d,nl*nl*2*dm%msh%gdim*dm%msh%nelv, host_to_device)
150  call device_memcpy(this%d, this%d_d,nl**dm%msh%gdim*dm%msh%nelv, host_to_device)
151  call device_memcpy(this%swplen, this%swplen_d,xh%lxyz*dm%msh%nelv, host_to_device)
152  end if
153  end subroutine fdm_init

155  subroutine swap_lengths(this, x, y, z, nelv, gdim)
156  type(fdm_t), intent(inout) :: this
157  integer, intent(in) :: gdim, nelv
158  real(kind=rp), dimension(this%Xh%lxyz,nelv) , intent(in):: x, y, z
159  integer :: j, k, e, n2, nz0, nzn, nx, lx1, n
160 
161  associate(l => this%swplen, xh =>this%Xh, &
162  llr => this%len_lr, lls => this%len_ls, llt => this%len_lt, &
163  lmr => this%len_mr, lms => this%len_ms, lmt => this%len_mt, &
164  lrr => this%len_rr, lrs => this%len_rs, lrt => this%len_rt)
165  lx1 = this%Xh%lx
166  n2 = lx1 - 1
167  nz0 = 1
168  nzn = 1
169  nx = lx1 - 2
170  if (gdim .eq. 3) then
171  nz0 = 0
172  nzn = n2
173  end if
174  call plane_space(lmr, lms, lmt, 0, n2, xh%wx, x, y, z,&
175  nx, n2, nz0, nzn, nelv, gdim)
176  n = n2 + 1
177  if (gdim .eq. 3) then
178  do e = 1,nelv
179  do j = 2,n2
180  do k = 2,n2
181  l(1,k,j,e) = lmr(e)
182  l(n,k,j,e) = lmr(e)
183  l(k,1,j,e) = lms(e)
184  l(k,n,j,e) = lms(e)
185  l(k,j,1,e) = lmt(e)
186  l(k,j,n,e) = lmt(e)
187  end do
188  end do
189  end do
190  if (neko_bcknd_cuda .eq. 1 .or. neko_bcknd_hip .eq. 1) then
191  call device_memcpy(l, this%swplen_d, this%dof%n_dofs,host_to_device)
192  call gs_op(this%gs_h, l, this%dof%n_dofs, gs_op_add)
193  call device_memcpy(l, this%swplen_d, this%dof%n_dofs,device_to_host)
194  else
195  call gs_op(this%gs_h, l, this%dof%n_dofs, gs_op_add)
196  end if
197 
198  do e = 1,nelv
199  llr(e) = l(1,2,2,e) - lmr(e)
200  lrr(e) = l(n,2,2,e) - lmr(e)
201  lls(e) = l(2,1,2,e) - lms(e)
202  lrs(e) = l(2,n,2,e) - lms(e)
203  llt(e) = l(2,2,1,e) - lmt(e)
204  lrt(e) = l(2,2,n,e) - lmt(e)
205  end do
206  else
207  do e = 1,nelv
208  do j = 2,n2
209  l(1,j,1,e) = lmr(e)
210  l(n,j,1,e) = lmr(e)
211  l(j,1,1,e) = lms(e)
212  l(j,n,1,e) = lms(e)
213  end do
214  end do
215 
216  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
217  .or. (neko_bcknd_opencl .eq. 1)) then
218  call device_memcpy(l, this%swplen_d, this%dof%n_dofs,host_to_device)
219  call gs_op(this%gs_h, l, this%dof%n_dofs, gs_op_add)
220  call device_memcpy(l, this%swplen_d, this%dof%n_dofs,device_to_host)
221  else
222  call gs_op(this%gs_h, l, this%dof%n_dofs, gs_op_add)
223  end if
224 
225  do e = 1,nelv
226  llr(e) = l(1,2,1,e) - lmr(e)
227  lrr(e) = l(n,2,1,e) - lmr(e)
228  lls(e) = l(2,1,1,e) - lms(e)
229  lrs(e) = l(2,n,1,e) - lms(e)
230  end do
231  end if
232  end associate
233  end subroutine swap_lengths

238  subroutine plane_space(lr, ls, lt, i1, i2, w, x, y, z, &
239  nx, nxn, nz0, nzn, nelv, gdim)
240  integer, intent(in) :: nxn, nzn, i1, i2, nelv, gdim, nx, nz0
241  real(kind=rp), intent(inout) :: lr(nelv), ls(nelv), lt(nelv)
242  real(kind=rp), intent(inout) :: w(nx)
243  real(kind=rp), intent(in) :: x(0:nxn,0:nxn,nz0:nzn,nelv)
244  real(kind=rp), intent(in) :: y(0:nxn,0:nxn,nz0:nzn,nelv)
245  real(kind=rp), intent(in) :: z(0:nxn,0:nxn,nz0:nzn,nelv)
246  real(kind=rp) :: lr2, ls2, lt2, weight, wsum
247  integer :: ny, nz, j1, k1, j2, k2, i, j, k, ie
248  ny = nx
249  nz = nx
250  j1 = i1
251  k1 = i1
252  j2 = i2
253  k2 = i2
254  ! Now, for each element, compute lr,ls,lt between specified planes
255  do ie = 1,nelv
256  if (gdim .eq. 3) then
257  lr2 = 0d0
258  wsum = 0d0
259  do k = 1,nz
260  do j = 1,ny
261  weight = w(j)*w(k)
262  lr2 = lr2 + weight /&
263  ( (x(i2,j,k,ie)-x(i1,j,k,ie))**2&
264  + (y(i2,j,k,ie)-y(i1,j,k,ie))**2&
265  + (z(i2,j,k,ie)-z(i1,j,k,ie))**2 )
266  wsum = wsum + weight
267  end do
268  end do
269  lr2 = lr2/wsum
270  lr(ie) = 1d0/sqrt(lr2)
271  ls2 = 0d0
272  wsum = 0d0
273  do k = 1,nz
274  do i = 1,nx
275  weight = w(i)*w(k)
276  ls2 = ls2 + weight / &
277  ( (x(i,j2,k,ie)-x(i,j1,k,ie))**2 &
278  + (y(i,j2,k,ie)-y(i,j1,k,ie))**2 &
279  + (z(i,j2,k,ie)-z(i,j1,k,ie))**2 )
280  wsum = wsum + weight
281  end do
282  end do
283  ls2 = ls2/wsum
284  ls(ie) = 1d0/sqrt(ls2)
285  lt2 = 0d0
286  wsum = 0d0
287  do j=1,ny
288  do i=1,nx
289  weight = w(i)*w(j)
290  lt2 = lt2 + weight / &
291  ( (x(i,j,k2,ie)-x(i,j,k1,ie))**2 &
292  + (y(i,j,k2,ie)-y(i,j,k1,ie))**2 &
293  + (z(i,j,k2,ie)-z(i,j,k1,ie))**2 )
294  wsum = wsum + weight
295  end do
296  end do
297  lt2 = lt2/wsum
298  lt(ie) = 1d0/sqrt(lt2)
299  else ! 2D
300  lr2 = 0d0
301  wsum = 0d0
302  do j=1,ny
303  weight = w(j)
304  lr2 = lr2 + weight / &
305  ( (x(i2,j,1,ie)-x(i1,j,1,ie))**2 &
306  + (y(i2,j,1,ie)-y(i1,j,1,ie))**2 )
307  wsum = wsum + weight
308  enddo
309  lr2 = lr2/wsum
310  lr(ie) = 1d0/sqrt(lr2)
311  ls2 = 0d0
312  wsum = 0d0
313  do i=1,nx
314  weight = w(i)
315  ls2 = ls2 + weight / &
316  ( (x(i,j2,1,ie)-x(i,j1,1,ie))**2 &
317  + (y(i,j2,1,ie)-y(i,j1,1,ie))**2 )
318  wsum = wsum + weight
319  enddo
320  ls2 = ls2/wsum
321  ls(ie) = 1d0/sqrt(ls2)
322  endif
323  enddo
324  ie = 1014
325  end subroutine plane_space

328  subroutine fdm_setup_fast(this, ah, bh, nl, n)
329  integer, intent(in) :: nl, n
330  type(fdm_t), intent(inout) :: this
331  real(kind=rp), intent(inout) :: ah(n+1,n+1), bh(n+1)
332  real(kind=rp), dimension(2*this%Xh%lx + 4) :: lr, ls, lt
333  integer :: i, j, k
334  integer :: ie, il, nr, ns, nt
335  integer :: lbr, rbr, lbs, rbs, lbt, rbt
336  real(kind=rp) :: eps, diag
337 
338  associate(s => this%s, d => this%d, &
339  llr => this%len_lr, lls => this%len_ls, llt => this%len_lt, &
340  lmr => this%len_mr, lms => this%len_ms, lmt => this%len_mt, &
341  lrr => this%len_rr, lrs => this%len_rs, lrt => this%len_rt)
342  do ie=1,this%dof%msh%nelv
343  lbr = this%dof%msh%facet_type(1, ie)
344  rbr = this%dof%msh%facet_type(2, ie)
345  lbs = this%dof%msh%facet_type(3, ie)
346  rbs = this%dof%msh%facet_type(4, ie)
347  lbt = this%dof%msh%facet_type(5, ie)
348  rbt = this%dof%msh%facet_type(6, ie)
349 
350  nr = nl
351  ns = nl
352  nt = nl
353  call fdm_setup_fast1d(s(1,1,1,ie), lr, nr, lbr, rbr, &
354  llr(ie), lmr(ie), lrr(ie), ah, bh, n)
355  call fdm_setup_fast1d(s(1,1,2,ie), ls, ns, lbs, rbs, &
356  lls(ie), lms(ie), lrs(ie), ah, bh, n)
357  if(this%dof%msh%gdim .eq. 3) then
358  call fdm_setup_fast1d(s(1,1,3,ie), lt, nt, lbt, rbt, &
359  llt(ie), lmt(ie), lrt(ie), ah, bh, n)
360  end if
361 
362  il = 1
363  if(.not. this%dof%msh%gdim .eq. 3) then
364  eps = 1d-5 * (vlmax(lr(2), nr-2) + vlmax(ls(2), ns-2))
365  do j = 1, ns
366  do i = 1, nr
367  diag = lr(i) + ls(j)
368  if (diag .gt. eps) then
369  d(il,ie) = 1.0_rp / diag
370  else
371  d(il,ie) = 0.0_rp
372  endif
373  il = il + 1
374  end do
375  end do
376  else
377  eps = 1d-5 * (vlmax(lr(2), nr-2) + &
378  vlmax(ls(2),ns-2) + vlmax(lt(2), nt-2))
379  do k = 1, nt
380  do j = 1, ns
381  do i = 1, nr
382  diag = lr(i) + ls(j) + lt(k)
383  if (diag .gt. eps) then
384  d(il,ie) = 1.0_rp / diag
385  else
386  d(il,ie) = 0.0_rp
387  endif
388  il = il + 1
389  end do
390  end do
391  end do
392  endif
393  end do
394  end associate
395 
396  end subroutine fdm_setup_fast
397 
398  subroutine fdm_setup_fast1d(s, lam, nl, lbc, rbc, ll, lm, lr, ah, bh, n)
399  integer, intent(in) :: nl, lbc, rbc, n
400  real(kind=rp), intent(inout) :: s(nl, nl, 2), lam(nl), ll, lm, lr
401  real(kind=rp), intent(inout) :: ah(0:n, 0:n), bh(0:n)

405  lx1 = n + 1

416  call trsp(s(1,1,2), nl, s, nl)

418  end subroutine fdm_setup_fast1d

472  subroutine fdm_setup_fast1d_a(a, lbc, rbc, ll, lm, lr, ah, n)
473  integer, intent(in) ::lbc, rbc, n
474  real(kind=rp), intent(inout) :: a(0:n+2,0:n+2), ll, lm, lr
475  real(kind=rp), intent(inout) :: ah(0:n,0:n)
476  real(kind=rp) :: fac
477  integer :: i, j, i0, i1
478 
479  i0 = 0
480  if(lbc .eq. 1) i0 = 1
481  i1 = n
482  if(rbc .eq. 1) i1 = n - 1
483 
484  call rzero(a, (n+3) * (n+3))
485 
486  fac = 2.0_rp / lm
487  a(1,1) = 1.0_rp
488  a(n+1,n+1) = 1.0-rp
489 
490  do j = i0, i1
491  do i = i0, i1
492  a(i+1,j+1) = fac * ah(i,j)
493  enddo
494  enddo
495

497  fac = 2.0_rp / ll
