52  subroutine krylov_solver_factory(ksp, n, solver, abstol, M)
53  class(ksp_t), allocatable, target, intent(inout) :: ksp
54  integer, intent(in), value :: n
55  character(len=*) :: solver
56  real(kind=rp), optional :: abstol
57  class(pc_t), optional, intent(inout), target :: m
58 
59  if (allocated(ksp)) then
60  call krylov_solver_destroy(ksp)
61  deallocate(ksp)
62  end if
63 
64  if (trim(solver) .eq. 'cg') then
65  if (neko_bcknd_sx .eq. 1) then
66  allocate(sx_cg_t::ksp)
67  else if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
68  (neko_bcknd_opencl .eq. 1)) then
69  allocate(cg_device_t::ksp)
70  else
71  allocate(cg_t::ksp)
72  end if
73  else if (trim(solver) .eq. 'pipecg') then
74  if (neko_bcknd_sx .eq. 1) then
75  allocate(sx_pipecg_t::ksp)
76  else if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
77  (neko_bcknd_opencl .eq. 1)) then
78  allocate(pipecg_device_t::ksp)
79  else
80  allocate(pipecg_t::ksp)
81  end if
82  else if (trim(solver) .eq. 'cacg') then
83  allocate(cacg_t::ksp)
84  else if (trim(solver) .eq. 'gmres') then
85  if (neko_bcknd_sx .eq. 1) then
86  allocate(sx_gmres_t::ksp)
87  else if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
88  (neko_bcknd_opencl .eq. 1)) then
89  allocate(gmres_device_t::ksp)
90  else
91  allocate(gmres_t::ksp)
92  end if
93  else if (trim(solver) .eq. 'bicgstab') then
94  allocate(bicgstab_t::ksp)
95  else
96  call neko_error('Unknown Krylov solver')
97  end if
98 
99  if (present(abstol) .and. present(m)) then
100  select type(kp => ksp)
101  type is(cg_t)
102  call kp%init(n, m = m, abs_tol = abstol)
103  type is(sx_cg_t)
104  call kp%init(n, m = m, abs_tol = abstol)
105  type is(cg_device_t)
106  call kp%init(n, m = m, abs_tol = abstol)
107  type is(pipecg_t)
108  call kp%init(n, m = m, abs_tol = abstol)
109  type is(sx_pipecg_t)
110  call kp%init(n, m = m, abs_tol = abstol)
111  type is(pipecg_device_t)
112  call kp%init(n, m = m, abs_tol = abstol)
113  type is(cacg_t)
114  call kp%init(n, m = m, abs_tol = abstol)
115  type is(gmres_t)
116  call kp%init(n, m = m, abs_tol = abstol)
117  type is(sx_gmres_t)
118  call kp%init(n, m = m, abs_tol = abstol)
119  type is(gmres_device_t)
120  call kp%init(n, m = m, abs_tol = abstol)
121  type is(bicgstab_t)
122  call kp%init(n, m = m, abs_tol = abstol)
123  end select
124  else if (present(abstol)) then
125  select type(kp => ksp)
126  type is(cg_t)
127  call kp%init(n, abs_tol = abstol)
128  type is(sx_cg_t)
129  call kp%init(n, abs_tol = abstol)
130  type is(cg_device_t)
131  call kp%init(n, abs_tol = abstol)
132  type is(pipecg_t)
133  call kp%init(n, abs_tol = abstol)
134  type is(sx_pipecg_t)
135  call kp%init(n, abs_tol = abstol)
136  type is (pipecg_device_t)
137  call kp%init(n, abs_tol = abstol)
138  type is(cacg_t)
139  call kp%init(n, abs_tol = abstol)
140  type is(gmres_t)
141  call kp%init(n, abs_tol = abstol)
142  type is(sx_gmres_t)
143  call kp%init(n, abs_tol = abstol)
144  type is(gmres_device_t)
145  call kp%init(n, abs_tol = abstol)
146  type is(bicgstab_t)
147  call kp%init(n, abs_tol = abstol)
148  end select
149  else if (present(m)) then
150  select type(kp => ksp)
151  type is(cg_t)
152  call kp%init(n, m = m)
153  type is(sx_cg_t)
154  call kp%init(n, m = m)
155  type is(cg_device_t)
156  call kp%init(n, m = m)
157  type is(pipecg_t)
158  call kp%init(n, m = m)
159  type is(sx_pipecg_t)
160  call kp%init(n, m = m)
161  type is (pipecg_device_t)
162  call kp%init(n, m = m)
163  type is(cacg_t)
164  call kp%init(n, m = m)
165  type is(gmres_t)
166  call kp%init(n, m = m)
167  type is(sx_gmres_t)
168  call kp%init(n, m = m)
169  type is(gmres_device_t)
170  call kp%init(n, m = m)
171  type is(bicgstab_t)
172  call kp%init(n, m = m)
173  end select
174  else
175  select type(kp => ksp)
176  type is(cg_t)
177  call kp%init(n)
178  type is(sx_cg_t)
179  call kp%init(n)
180  type is(cg_device_t)
181  call kp%init(n)
182  type is(pipecg_t)
183  call kp%init(n)
184  type is(sx_pipecg_t)
185  call kp%init(n)
186  type is (pipecg_device_t)
187  call kp%init(n)
188  type is(cacg_t)
189  call kp%init(n)
190  type is(gmres_t)
191  call kp%init(n)
192  type is(sx_gmres_t)
193  call kp%init(n)
194  type is(gmres_device_t)
195  call kp%init(n)
196  type is(bicgstab_t)
197  call kp%init(n)
198  end select
199  end if
200 
201  end subroutine krylov_solver_factory
