78  subroutine case_init(C, case_file)
79  type(case_t), target, intent(inout) :: C
80  character(len=*), intent(in) :: case_file
81 
82  ! Namelist for case description
83  character(len=NEKO_FNAME_LEN) :: mesh_file = ''
84  character(len=80) :: fluid_scheme = ''
85  character(len=80) :: source_term = ''
86  character(len=80) :: initial_condition = ''
87  integer :: lx = 0
88  type(param_io_t) :: params
89  namelist /neko_case/ mesh_file, fluid_scheme, lx, &
90  source_term, initial_condition
91 
92  integer :: ierr
93  type(file_t) :: msh_file, bdry_file, part_file
94  type(mesh_fld_t) :: msh_part
95  integer, parameter :: nbytes = neko_fname_len + 240 + 8
96  character buffer(nbytes)
97  integer :: pack_index
98  type(mesh_fld_t) :: parts
99 
100  call neko_log%section('Case')
101  call neko_log%message('Reading case file ' // trim(case_file))
102 
103  !
104  ! Read case description
105  !
106 
107  if (pe_rank .eq. 0) then
108  open(10, file=trim(case_file))
109  read(10, nml=neko_case)
110  read(10, *) params
111  close(10)
112 
113  pack_index = 1
114  call mpi_pack(mesh_file, neko_fname_len, mpi_character, &
115  buffer, nbytes, pack_index, neko_comm, ierr)
116  call mpi_pack(fluid_scheme, 80, mpi_character, &
117  buffer, nbytes, pack_index, neko_comm, ierr)
118  call mpi_pack(source_term, 80, mpi_character, &
119  buffer, nbytes, pack_index, neko_comm, ierr)
120  call mpi_pack(initial_condition, 80, mpi_character, &
121  buffer, nbytes, pack_index, neko_comm, ierr)
122  call mpi_pack(lx, 1, mpi_integer, &
123  buffer, nbytes, pack_index, neko_comm, ierr)
124  call mpi_bcast(buffer, nbytes, mpi_packed, 0, neko_comm, ierr)
125  call mpi_bcast(params%p, 1, mpi_neko_params, 0, neko_comm, ierr)
126  else
127  call mpi_bcast(buffer, nbytes, mpi_packed, 0, neko_comm, ierr)
128  pack_index = 1
129 
130  call mpi_unpack(buffer, nbytes, pack_index, &
131  mesh_file, neko_fname_len, mpi_character, neko_comm, ierr)
132  call mpi_unpack(buffer, nbytes, pack_index, &
133  fluid_scheme, 80, mpi_character, neko_comm, ierr)
134  call mpi_unpack(buffer, nbytes, pack_index, &
135  source_term, 80, mpi_character, neko_comm, ierr)
136  call mpi_unpack(buffer, nbytes, pack_index, &
137  initial_condition, 80, mpi_character, neko_comm, ierr)
138  call mpi_unpack(buffer, nbytes, pack_index, &
139  lx, 1, mpi_integer, neko_comm, ierr)
140  call mpi_bcast(params%p, 1, mpi_neko_params, 0, neko_comm, ierr)
141  end if
142 
143  msh_file = file_t(trim(mesh_file))
144  call msh_file%read(c%msh)
145  c%params = params%p
146 
147  !
148  ! Load Balancing
149  !
150  if (pe_size .gt. 1 .and. c%params%loadb) then
151  call neko_log%section('Load Balancing')
152  call parmetis_partmeshkway(c%msh, parts)
153  call redist_mesh(c%msh, parts)
154  call neko_log%end_section()
155  end if
156 
157 
158  !
159  ! Setup user defined functions
160  !
161  call c%usr%init()
162  call c%usr%usr_msh_setup(c%msh)
163 
164  !
165  ! Setup fluid scheme
166  !
167  call fluid_scheme_factory(c%fluid, trim(fluid_scheme))
168  call c%fluid%init(c%msh, lx, c%params)
169 
170  !
171  ! Setup user defined conditions
172  !
173  if (trim(c%params%fluid_inflow) .eq. 'user') then
174  call c%fluid%set_usr_inflow(c%usr%fluid_usr_if)
175  end if
176 
177  !
178  ! Setup source term
179  !
180  if (trim(source_term) .eq. 'user') then
181  call c%fluid%set_source(trim(source_term), usr_f=c%usr%fluid_usr_f)
182  else if (trim(source_term) .eq. 'user_vector') then
183  call c%fluid%set_source(trim(source_term), usr_f_vec=c%usr%fluid_usr_f_vector)
184  else
185  call c%fluid%set_source(trim(source_term))
186  end if
187 
188  !
189  ! Setup initial conditions
190  !
191  if (len_trim(initial_condition) .gt. 0) then
192  if (trim(initial_condition) .ne. 'user') then
193  call set_flow_ic(c%fluid%u, c%fluid%v, c%fluid%w, c%fluid%p, &
194  c%fluid%c_Xh, c%fluid%gs_Xh, initial_condition, c%params)
195  else
196  call set_flow_ic(c%fluid%u, c%fluid%v, c%fluid%w, c%fluid%p, &
197  c%fluid%c_Xh, c%fluid%gs_Xh, c%usr%fluid_usr_ic, c%params)
198  end if
199  end if
200 
201  ! Add initial conditions to BDF scheme (if present)
202  select type(f => c%fluid)
203  type is(fluid_pnpn_t)
204  call f%ulag%set(f%u)
205  call f%vlag%set(f%v)
206  call f%wlag%set(f%w)
207  type is(fluid_plan4_t)
208  call f%ulag%set(f%u)
209  call f%vlag%set(f%v)
210  call f%wlag%set(f%w)
211  type is(device_fluid_plan4_t)
212  call f%ulag%set(f%u)
213  call f%vlag%set(f%v)
214  call f%wlag%set(f%w)
215  end select
216 
217  !
218  ! Validate that the case is properly setup for time-stepping
219  !
220  call c%fluid%validate
221 
222  !
223  ! Set order of timestepper
224  !
225  call c%ab_bdf%set_time_order(c%params%time_order)
226 
227  !
228  ! Save boundary markings for fluid (if requested)
229  !
230  if (c%params%output_bdry) then
231  bdry_file = file_t(trim(c%params%output_dir)//'bdry.fld')
232  call bdry_file%write(c%fluid%bdry)
233  end if
234 
235  !
236  ! Save mesh partitions (if requested)
237  !
238  if (c%params%output_part) then
239  call mesh_field_init(msh_part, c%msh, 'MPI_Rank')
240  msh_part%data = pe_rank
241  part_file = file_t(trim(c%params%output_dir)//'partitions.vtk')
242  call part_file%write(msh_part)
243  call mesh_field_free(msh_part)
244  end if
245 
246  !
247  ! Setup sampler
248  !
249  call c%s%init(c%params%nsamples, c%params%T_end)
250  c%f_out = fluid_output_t(c%fluid, path=c%params%output_dir)
251  call c%s%add(c%f_out)
252 
253  !
254  ! Save checkpoints (if requested)
255  !
256  if (c%params%output_chkp) then
257  c%f_chkp = chkp_output_t(c%fluid%chkp, path=c%params%output_dir)
258  call c%s%add(c%f_chkp)
259  end if
260 
261  !
262  ! Setup statistics
263  !
264  call c%q%init(c%params%stats_begin)
265  if (c%params%stats_mean_flow) then
266  call c%q%add(c%fluid%mean%u)
267  call c%q%add(c%fluid%mean%v)
268  call c%q%add(c%fluid%mean%w)
269  call c%q%add(c%fluid%mean%p)
270 
271  if (c%params%output_mean_flow) then
272  c%f_mf = mean_flow_output_t(c%fluid%mean, c%params%stats_begin, &
273  path=c%params%output_dir)
274  call c%s%add(c%f_mf)
275  end if
276  end if
277 
278  if (c%params%stats_mean_sqr_flow) then
279  call c%q%add(c%fluid%mean_sqr%uu)
280  call c%q%add(c%fluid%mean_sqr%vv)
281  call c%q%add(c%fluid%mean_sqr%ww)
282  call c%q%add(c%fluid%mean_sqr%pp)
283 
284  if (c%params%output_mean_sqr_flow) then
285  c%f_msqrf = mean_sqr_flow_output_t(c%fluid%mean_sqr, &
286  c%params%stats_begin, &
287  path=c%params%output_dir)
288  call c%s%add(c%f_msqrf)
289  end if
290  end if
291 
292  !
293  ! Setup joblimit
294  !
295  call jobctrl_set_time_limit(c%params%jlimit)
296 
297  call neko_log%end_section()
298 
299  end subroutine case_init

306  call c%fluid%free()
307  deallocate(c%fluid)
308  end if
309 
310  call mesh_free(c%msh)
311 
312  call c%s%free()
313 
314  call c%q%free()
315 
316  end subroutine case_free
