280  function gmres_device_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
281  class(gmres_device_t), intent(inout) :: this
282  class(ax_t), intent(inout) :: ax
283  type(field_t), intent(inout) :: x
284  integer, intent(inout) :: n
285  real(kind=rp), dimension(n), intent(inout) :: f
286  type(coef_t), intent(inout) :: coef
287  type(bc_list_t), intent(inout) :: blst
288  type(gs_t), intent(inout) :: gs_h
289  type(ksp_monitor_t) :: ksp_results
290  integer, optional, intent(in) :: niter
291  integer :: iter
292  integer :: i, j, k
293  real(kind=rp) :: rnorm, alpha, temp, lr, alpha2, norm_fac
294  logical :: conv
295  type(c_ptr) :: f_d, x_d
296 
297  f_d = device_get_ptr(f)
298  x_d = x%x_d
299 
300  conv = .false.
301  iter = 0
302 
303  associate(w => this%w, c => this%c, r => this%r, z => this%z, h => this%h, &
304  v => this%v, s => this%s, gam => this%gam, &
305  v_d => this%v_d, w_d => this%w_d, r_d => this%r_d)
306 
307  norm_fac = 1.0_rp / sqrt(coef%volume)
308  call rzero(gam, this%m_restart + 1)
309  call rone(s, this%m_restart)
310  call rone(c, this%m_restart)
311  call rzero(h, this%m_restart * this%m_restart)
312  call device_rzero(x%x_d, n)
313  call device_rzero(this%gam_d, this%m_restart + 1)
314  call device_rone(this%s_d, this%m_restart)
315  call device_rone(this%c_d, this%m_restart)
316 
317  call rzero(this%h, this%m_restart**2)
318  do j = 1, this%m_restart
319  call device_rzero(this%h_d(j), this%m_restart)
320  end do
321  do while (.not. conv .and. iter .lt. niter)
322 
323  if(iter.eq.0) then
324  call device_copy(r_d, f_d, n)
325  else
326  call device_copy(r_d, f_d, n)
327  call ax%compute(w, x%x, coef, x%msh, x%Xh)
328  call gs_op(gs_h, w, n, gs_op_add)
329  call bc_list_apply(blst, w, n)
330  call device_sub2(r_d, w_d, n)
331  end if
332 
333  gam(1) = sqrt(device_glsc3(r_d, r_d, coef%mult_d, n))
334  if(iter.eq.0) then
335  ksp_results%res_start = gam(1) * norm_fac
336  end if
337 
338  if (gam(1) .eq. 0) return
339 
340  rnorm = 0.0_rp
341  temp = 1.0_rp / gam(1)
342  call device_cmult2(v_d(1), r_d, temp, n)
343  do j = 1, this%m_restart
344  iter = iter+1
345 
346  call this%M%solve(z(1,j), v(1,j), n)
347 
348  call ax%compute(w, z(1,j), coef, x%msh, x%Xh)
349  call gs_op(gs_h, w, n, gs_op_add)
350  call bc_list_apply(blst, w, n)
351  call device_glsc3_many(h(1,j), w_d, this%v_d_d,coef%mult_d, j, n)
352 
353  call device_memcpy_r1(h(:,j), this%h_d(j), j, host_to_device)
354 
355  alpha2 = device_gmres_part2(w_d, this%v_d_d, this%h_d(j), coef%mult_d, j, n)
356 
357  alpha = sqrt(alpha2)
358  do i=1,j-1
359  temp = h(i,j)
360  h(i ,j) = c(i)*temp + s(i) * h(i+1,j)
361  h(i+1,j) = -s(i)*temp + c(i) * h(i+1,j)
362  end do
363 
364  rnorm = 0.0_rp
365  if(alpha .eq. 0.0_rp) then
366  conv = .true.
367  exit
368  end if
369 
370  lr = sqrt(h(j,j) * h(j,j) + alpha**2)
371  temp = 1.0_rp / lr
372  c(j) = h(j,j) * temp
373  s(j) = alpha * temp
374  h(j,j) = lr
375  call device_memcpy_r1(h(:,j), this%h_d(j), j, host_to_device)
376  gam(j+1) = -s(j) * gam(j)
377  gam(j) = c(j) * gam(j)
378 
379  rnorm = abs(gam(j+1)) * norm_fac
380  if (rnorm .lt. this%abs_tol) then
381  conv = .true.
382  exit
383  end if
384 
385  if (iter + 1 .gt. niter) exit
386 
387  if( j .lt. this%m_restart) then
388  temp = 1.0_rp / alpha
389  call device_cmult2(v_d(j+1), w_d, temp, n)
390  end if
391 
392  end do
393 
394  j = min(j, this%m_restart)
395  do k = j, 1, -1
396  temp = gam(k)
397  do i = j, k+1, -1
398  temp = temp - h(k,i) * c(i)
399  end do
400  c(k) = temp / h(k,k)
401  end do
402  call device_memcpy(c, this%c_d, j, host_to_device)
403  call device_add2s2_many(x_d, this%z_d_d,this%c_d, j, n)
404  end do
405 
406  end associate
407 
408  ksp_results%res_final = rnorm
409  ksp_results%iter = iter
410 
411  end function gmres_device_solve
