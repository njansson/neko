48 void nonlinear_index(const int idx, const int lx, int *index) {
49  const int idx2 = idx -1;
50  index[3] = idx2/(lx * lx * lx) ;
51  index[2] = (idx2 - (lx*lx*lx)*index[3])/(lx * lx);
52  index[1] = (idx2 - (lx*lx*lx)*index[3] - (lx*lx) * index[2]) / lx;
53  index[0] = (idx2 - (lx*lx*lx)*index[3] - (lx*lx) * index[2]) - lx*index[1];
54  index[0]++;
55  index[1]++;
56  index[2]++;
57  index[3]++;
58 }
