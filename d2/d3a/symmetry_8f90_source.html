67  subroutine symmetry_init_msk(this, c)
68  class(symmetry_t), intent(inout) :: this
69  type(coef_t), intent(in) :: c
70  integer :: i, m, j, l
71  type(tuple_i4_t), pointer :: bfp(:)
72  real(kind=rp) :: sx,sy,sz
73  real(kind=rp), parameter :: tol = 1d-3
74  type(tuple_i4_t) :: bc_facet
75  integer :: facet, el
76 
77  call symmetry_free(this)
78 
79  call this%bc_x%init(c%dof)
80  call this%bc_y%init(c%dof)
81  call this%bc_z%init(c%dof)
82 
83  associate(nx => c%nx, ny => c%ny, nz => c%nz)
84  bfp => this%marked_facet%array()
85  do i = 1, this%marked_facet%size()
86  bc_facet = bfp(i)
87  facet = bc_facet%x(1)
88  el = bc_facet%x(2)
89  sx = 0d0
90  sy = 0d0
91  sz = 0d0
92  select case (facet)
93  case(1,2)
94  do l = 2, c%Xh%lx - 1
95  do j = 2, c%Xh%lx -1
96  sx = sx + abs(abs(nx(l, j, facet, el)) - 1d0)
97  sy = sy + abs(abs(ny(l, j, facet, el)) - 1d0)
98  sz = sz + abs(abs(nz(l, j, facet, el)) - 1d0)
99  end do
100  end do
101  case(3,4)
102  do l = 2, c%Xh%lx - 1
103  do j = 2, c%Xh%lx - 1
104  sx = sx + abs(abs(nx(l, j, facet, el)) - 1d0)
105  sy = sy + abs(abs(ny(l, j, facet, el)) - 1d0)
106  sz = sz + abs(abs(nz(l, j, facet, el)) - 1d0)
107  end do
108  end do
109  case(5,6)
110  do l = 2, c%Xh%lx - 1
111  do j = 2, c%Xh%lx - 1
112  sx = sx + abs(abs(nx(l, j, facet, el)) - 1d0)
113  sy = sy + abs(abs(ny(l, j, facet, el)) - 1d0)
114  sz = sz + abs(abs(nz(l, j, facet, el)) - 1d0)
115  end do
116  end do
117  end select
118  sx = sx / (c%Xh%lx - 2)**2
119  sy = sy / (c%Xh%lx - 2)**2
120  sz = sz / (c%Xh%lx - 2)**2
121 
122  if (sx .lt. tol) then
123  call this%bc_x%mark_facet(facet, el)
124  end if
125 
126  if (sy .lt. tol) then
127  call this%bc_y%mark_facet(facet, el)
128  end if
129 
130  if (sz .lt. tol) then
131  call this%bc_z%mark_facet(facet, el)
132  end if
133  end do
134  end associate
135  call this%bc_x%finalize()
136  call this%bc_x%set_g(0.0_rp)
137  call this%bc_y%finalize()
138  call this%bc_y%set_g(0.0_rp)
139  call this%bc_z%finalize()
140  call this%bc_z%set_g(0.0_rp)
141 
142  end subroutine symmetry_init_msk
