183  subroutine gs_device_mpi_buf_init(this, pe_order, dof_stack, mark_dupes)
184  class(gs_device_mpi_buf_t), intent(inout) :: this
185  integer, allocatable, intent(inout) :: pe_order(:)
186  type(stack_i4_t), allocatable, intent(inout) :: dof_stack(:)
187  logical, intent(in) :: mark_dupes
188  integer, allocatable :: dofs(:)
189  integer :: i, j, total
190  integer(c_size_t) :: sz
191  type(htable_i4_t) :: doftable
192  integer :: dupe, marked, k
193  real(c_rp) :: rp_dummy
194  integer(c_int32_t) :: i4_dummy
195 
196  call device_mpi_init_reqs(size(pe_order), this%reqs)
197 
198  allocate(this%ndofs(size(pe_order)))
199  allocate(this%offset(size(pe_order)))
200 
201  total = 0
202  do i = 1, size(pe_order)
203  this%ndofs(i) = dof_stack(pe_order(i))%size()
204  this%offset(i) = total
205  total = total + this%ndofs(i)
206  end do
207 
208  this%total = total
209 
210  sz = c_sizeof(rp_dummy) * total
211  call device_alloc(this%buf_d, sz)
212 
213  sz = c_sizeof(i4_dummy) * total
214  call device_alloc(this%dof_d, sz)
215 
216  if (mark_dupes) call doftable%init(2*total)
217  allocate(dofs(total))
218 
219  ! Copy from dof_stack into dofs, optionally marking duplicates with doftable
220  marked = 0
221  do i = 1, size(pe_order)
222  ! %array() breaks on cray
223  select type (arr => dof_stack(pe_order(i))%data)
224  type is (integer)
225  do j = 1, this%ndofs(i)
226  k = this%offset(i) + j
227  if (mark_dupes) then
228  if (doftable%get(arr(j), dupe) .eq. 0) then
229  if (dofs(dupe) .gt. 0) then
230  dofs(dupe) = -dofs(dupe)
231  marked = marked + 1
232  end if
233  dofs(k) = -arr(j)
234  marked = marked + 1
235  else
236  call doftable%set(arr(j), k)
237  dofs(k) = arr(j)
238  end if
239  else
240  dofs(k) = arr(j)
241  end if
242  end do
243  end select
244  end do
245 
246  call device_memcpy(dofs, this%dof_d, total, host_to_device)
247 
248  deallocate(dofs)
249  call doftable%free()
250 
251  end subroutine gs_device_mpi_buf_init

293  class(gs_device_mpi_t), intent(inout) :: this
294  integer, intent(in) :: n
295  real(kind=rp), dimension(n), intent(inout) :: u

297  type(c_ptr) :: u_d
