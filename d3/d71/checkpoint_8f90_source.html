70  subroutine chkp_init(this, u, v, w, p)
71  class(chkp_t), intent(inout) :: this
72  type(field_t), intent(in), target :: u
73  type(field_t), intent(in), target :: v
74  type(field_t), intent(in), target :: w
75  type(field_t), intent(in), target :: p
76 
77  ! Check that all velocity components are defined on the same
78  ! function space
79  if ( u%Xh .ne. v%Xh .or. &
80  u%Xh .ne. w%Xh ) then
81  call neko_error('Different function spaces for each velocity component')
82  end if
83 
84  ! Check that both velocity and pressure is defined on the same mesh
85  if ( u%msh%nelv .ne. p%msh%nelv ) then
86  call neko_error('Velocity and pressure defined on different meshes')
87  end if
88

114  subroutine chkp_sync_host(this)
115  class(chkp_t), intent(inout) :: this
116 
117  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
118  (neko_bcknd_opencl .eq. 1)) then
119  associate(u=>this%u, v=>this%v, w=>this%w, &
120  ulag=>this%ulag, vlag=>this%vlag, wlag=>this%wlag)
121 
122  if (associated(this%u) .and. associated(this%v) .and. &
123  associated(this%w)) then
124  call device_memcpy(u%x, u%x_d, u%dof%size(), device_to_host)
