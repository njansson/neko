79  function dofmap_init(msh, Xh) result(this)
80  type(mesh_t), target, intent(inout) :: msh
81  type(space_t), target, intent(inout) :: xh
82  type(dofmap_t) :: this
83 
84  if ((msh%gdim .eq. 3 .and. xh%lz .eq. 1) .or. &
85  (msh%gdim .eq. 2 .and. xh%lz .gt. 1)) then
86  call neko_error("Invalid dimension of function space for the given mesh")
87  end if
88 
89 
90  call dofmap_free(this)
91 
92  this%msh => msh
93  this%Xh => xh
94 
95  this%n_dofs = xh%lx* xh%ly * xh%lz * msh%nelv
96 
97  !
98  ! Assign a unique id for all dofs
99  !
100 
101  allocate(this%dof(xh%lx, xh%ly, xh%lz, msh%nelv))
102  allocate(this%shared_dof(xh%lx, xh%ly, xh%lz, msh%nelv))
103 
104  this%dof = 0
105  this%shared_dof = .false.
106 
108  if (msh%gdim .eq. 3) then
109  call dofmap_number_points(this)
110  call dofmap_number_edges(this)
111  call dofmap_number_faces(this)
112  else
113  call dofmap_number_points(this)
114  call dofmap_number_edges(this)
115  end if
116 
117  !
118  ! Generate x,y,z-coordinates for all dofs
119  !
120 
121  allocate(this%x(xh%lx, xh%ly, xh%lz, msh%nelv))
122  allocate(this%y(xh%lx, xh%ly, xh%lz, msh%nelv))
123  allocate(this%z(xh%lx, xh%ly, xh%lz, msh%nelv))
124 
125  this%x = 0d0
126  this%y = 0d0
127  this%z = 0d0
129 
130  call dofmap_generate_xyz(this)
131 
132  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
133  (neko_bcknd_opencl .eq. 1)) then
134  call device_map(this%x, this%x_d, this%n_dofs)
135  call device_map(this%y, this%y_d, this%n_dofs)
136  call device_map(this%z, this%z_d, this%n_dofs)
137 
138  call device_memcpy(this%x, this%x_d, this%n_dofs, host_to_device)
139  call device_memcpy(this%y, this%y_d, this%n_dofs, host_to_device)
140  call device_memcpy(this%z, this%z_d, this%n_dofs, host_to_device)
141  end if
142 
143  end function dofmap_init

276  subroutine dofmap_number_edges(this)
277  type(dofmap_t), target :: this
278  type(mesh_t), pointer :: msh
279  type(space_t), pointer :: Xh
280  integer :: i,j,k
281  integer :: global_id
282  type(tuple_i4_t) :: edge
283  integer(kind=i8) :: num_dofs_edges(3) ! #dofs for each dir (r, s, t)
284  integer(kind=i8) :: edge_id, edge_offset
285  logical :: shared_dof
286 
287  msh => this%msh
288  xh => this%Xh
289 
290  ! Number of dofs on an edge excluding end-points
291  num_dofs_edges(1) = int(xh%lx - 2, i8)
292  num_dofs_edges(2) = int(xh%ly - 2, i8)
293  num_dofs_edges(3) = int(xh%lz - 2, i8)
294  edge_offset = int(msh%glb_mpts, i8) + int(1, 4)
295 
296  do i = 1, msh%nelv
297 
298  select type(ep=>msh%elements(i)%e)
299  type is (hex_t)
300  !
301  ! Number edges in r-direction
302  !
303  call ep%edge_id(edge, 1)
304  shared_dof = mesh_is_shared(msh, edge)
305  global_id = mesh_get_global(msh, edge)
306  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(1)
307  !Reverse order of tranversal if edge is reversed
308  do j = 2, xh%lx - 1
309  k = j
310  if(int(edge%x(1), i8) .ne. this%dof(1,1,1,i)) k = xh%lx+1-j
311  this%dof(k, 1, 1, i) = edge_id
312  this%shared_dof(k, 1, 1, i) = shared_dof
313  edge_id = edge_id + 1
314  end do
315 
316  call ep%edge_id(edge, 3)
317  shared_dof = mesh_is_shared(msh, edge)
318  global_id = mesh_get_global(msh, edge)
319  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(1)
320  do j = 2, xh%lx - 1
321  k = j
322  if(int(edge%x(1), i8) .ne. this%dof(1,1,xh%lz,i)) k = xh%lx+1-j
323  this%dof(k, 1, xh%lz, i) = edge_id
324  this%shared_dof(k, 1, xh%lz, i) = shared_dof
325  edge_id = edge_id + 1
326  end do
327 
328  call ep%edge_id(edge, 2)
329  shared_dof = mesh_is_shared(msh, edge)
330  global_id = mesh_get_global(msh, edge)
331  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(1)
332  do j = 2, xh%lx - 1
333  k = j
334  if(int(edge%x(1), i8) .ne. this%dof(1,xh%ly,1,i)) k = xh%lx+1-j
335  this%dof(k, xh%ly, 1, i) = edge_id
336  this%shared_dof(k, xh%ly, 1, i) = shared_dof
337  edge_id = edge_id + 1
338  end do
339 
340  call ep%edge_id(edge, 4)
341  shared_dof = mesh_is_shared(msh, edge)
342  global_id = mesh_get_global(msh, edge)
343  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(1)
344  do j = 2, xh%lx - 1
345  k = j
346  if(int(edge%x(1), i8) .ne. this%dof(1,xh%ly,xh%lz,i)) k = xh%lx+1-j
347  this%dof(k, xh%ly, xh%lz, i) = edge_id
348  this%shared_dof(k, xh%ly, xh%lz, i) = shared_dof
349  edge_id = edge_id + 1
350  end do
351 
352 
353  !
354  ! Number edges in s-direction
355  !

357  shared_dof = mesh_is_shared(msh, edge)
358  global_id = mesh_get_global(msh, edge)
359  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(2)
360  do j = 2, xh%ly - 1
361  k = j
362  if(int(edge%x(1), i8) .ne. this%dof(1,1,1,i)) k = xh%ly+1-j

364  this%shared_dof(1, k, 1, i) = shared_dof

370  global_id = mesh_get_global(msh, edge)
371  edge_id = edge_offset + int((global_id - 1), i8) * num_dofs_edges(2)
372  do j = 2, xh%ly - 1
373  k = j
374  if(int(edge%x(1), i8) .ne. this%dof(1,1,xh%lz,i)) k = xh%ly+1-j
375  this%dof(1, k, xh%lz, i) = edge_id

634  function dofmap_facetidx(face_order, face, facet_id, k1, j1, lk1, lj1) result(facet_idx)
635  type(tuple4_i4_t) :: face_order, face
636  integer(kind=i8) :: facet_idx, facet_id
637  integer :: k1, j1, lk1, lj1
638  integer :: k,j,lk,lj
639 
640  k = k1 - 2
641  j = j1 - 2
642  lk = lk1 - 2
643  lj = lj1 - 2
644 
645  ! Given the indexes k,j for a GLL point on the inner part of the
646  ! face, we assign a unique number to it that depends on the
647  ! corner with the lowest id and its neighbour with the lowest
648  ! id. The id is assigned in this way to be consistent regardless
649  ! of how the faces are rotated or mirrored.
650  !
651  ! 4 -------- 3
652  ! | | k
653  ! |----->| ^
654  ! |----->| |
655  ! |----->| |
656  ! 1 -------- 2 0--->j
657 
658 
659  if(face_order%x(1) .eq. face%x(1)) then
660  if(face_order%x(2) .lt. face_order%x(4)) then
661  facet_idx = facet_id + j + k*lj
662  else
663  facet_idx = facet_id + j*lk + k
664  end if
665  else if(face_order%x(2) .eq. face%x(1)) then
666  if(face_order%x(3) .lt. face_order%x(1)) then
667  facet_idx = facet_id + lk*(lj-1-j) + k
668  else
669  facet_idx = facet_id + (lj-1-j) + k*lj
670  end if
671  else if(face_order%x(3) .eq. face%x(1)) then
672  if(face_order%x(4) .lt. face_order%x(2)) then
673  facet_idx = facet_id + (lj-1-j) + lj*(lk-1-k)
674  else
675  facet_idx = facet_id + lk*(lj-1-j) + (lk-1-k)
676  end if
677  else if(face_order%x(4) .eq. face%x(1)) then
678  if(face_order%x(1) .lt. face_order%x(3)) then
679  facet_idx = facet_id + lk*j + (lk-1-k)
680  else
681  facet_idx = facet_id + j + lj*(lk-1-k)
682  end if
683  end if
684 
685  end function dofmap_facetidx

890  subroutine gh_face_extend_3d(x, zg, n, gh_type, e, v)
891  integer, intent(in) :: n
892  real(kind=rp), intent(inout) :: x(n, n, n)
893  real(kind=rp), intent(in) :: zg(n)

897  real(kind=rp) :: si, sj, sk, hi, hj, hk
898 
899 !
900 ! Build vertex interpolant
901 !
902  ntot = n**3
903  call rzero(v, ntot)

905  do jj = 1, n, n-1
906  do ii = 1, n, n-1
907  do k = 1, n

916  end do
917  end do
918  end do
919  end do

922  call copy(x, v, ntot)
923  return
924  end if

931 !

933  do jj = 1, n, n-1

938  hk = 0.5*((n-kk)*(1-zg(k))+(kk-1)*(1+zg(k)))/(n-1)

940  end do

942  end do
943  end do
944  end do

950  do k = 1, n
951  do j = 1, n
952  do i = 1, n

954  hk = 0.5*((n-kk)*(1-zg(k))+(kk-1)*(1+zg(k)))/(n-1)
955  e(i,j,k) = e(i,j,k) + hi*hk*(x(ii,j,kk)-v(ii,j,kk))
956  end do
957  end do
958  end do
959  end do
960  end do

963 !

965  do ii = 1, n, n-1
966  do k = 1, n
967  do j = 1, n
968  do i = 1, n

974  end do
975  end do
976  end do

979

981  call copy(x, e, ntot)
982  return
983  end if
984 !
985 ! Extend faces
986 !
987  call rzero(v, ntot)
988 !
989 ! x-edges
990 !
991  do ii = 1, n, n-1
992  do k = 1, n
993  do j = 1, n
994  do i = 1, n
995  hi = 0.5*((n-ii)*(1-zg(i))+(ii-1)*(1+zg(i)))/(n-1)
996  v(i,j,k) = v(i,j,k) + hi*(x(ii,j,k)-e(ii,j,k))
997  end do
998  end do
999  end do
1000  end do
1001 !
1002 ! y-edges
1003 !
1004  do jj = 1, n, n-1
1005  do k = 1, n
1006  do j = 1, n
1007  do i = 1, n
1008  hj = 0.5*((n-jj)*(1-zg(j))+(jj-1)*(1+zg(j)))/(n-1)
1009  v(i,j,k) = v(i,j,k) + hj*(x(i,jj,k)-e(i,jj,k))
1010  end do
1011  end do
1012  end do
1013  end do
1014 !
1015 ! z-edges
1016 !
1017  do kk= 1 , n, n-1
