54  subroutine redist_mesh(msh, parts)
55  type(mesh_t), intent(inout), target :: msh
56  type(mesh_fld_t), intent(in) :: parts
57  type(stack_nh_t), allocatable :: new_mesh_dist(:)
58  type(stack_nz_t), allocatable :: new_zone_dist(:)
59  type(stack_nc_t), allocatable :: new_curve_dist(:)
60  type(nmsh_hex_t) :: el
61  type(nmsh_hex_t), pointer :: np(:)
62  type(nmsh_hex_t), allocatable :: recv_buf_msh(:)
63  type(nmsh_zone_t), pointer :: zp(:)
64  type(nmsh_zone_t), allocatable :: recv_buf_zone(:)
65  type(nmsh_curve_el_t), pointer :: cp(:)
66  type(nmsh_curve_el_t), allocatable :: recv_buf_curve(:)
67  class(element_t), pointer :: ep
68  integer, allocatable :: recv_buf_idx(:), send_buf_idx(:)
69  type(mpi_status) :: status
70  integer :: i, j, k, ierr, max_recv_idx
71  integer :: src, dst, recv_size, gdim, tmp, new_el_idx, new_pel_idx
72  integer :: max_recv(3)
73  type(point_t) :: p(8)
74  type(htable_i4_t) :: el_map, glb_map
75  type(stack_i4_t) :: pe_lst
76 
77 
78  !
79  ! Reset possible periodic ids
80  !
81  call mesh_reset_periodic_ids(msh)
82 
83  !
84  ! Extract new zone distributions
85  !
86 
87  allocate(new_zone_dist(0:pe_size - 1))
88  do i = 0, pe_size - 1
89  call new_zone_dist(i)%init()
90  end do
91 
92  call redist_zone(msh, msh%wall, 1, parts, new_zone_dist)
93  call redist_zone(msh, msh%inlet, 2, parts, new_zone_dist)
94  call redist_zone(msh, msh%outlet, 3, parts, new_zone_dist)
95  call redist_zone(msh, msh%sympln, 4, parts, new_zone_dist)
96  call redist_zone(msh, msh%periodic, 5, parts, new_zone_dist)
97 
98  !
99  ! Extract new curve info. distributions
100  !
101  allocate(new_curve_dist(0:pe_size - 1))
102  do i = 0, pe_size - 1
103  call new_curve_dist(i)%init()
104  end do
105 
106  call redist_curve(msh, msh%curve, parts, new_curve_dist)
107 
108  !
109  ! Extract new mesh distribution
110  !
111 
112  allocate(new_mesh_dist(0:(pe_size - 1)))
113  do i = 0, pe_size - 1
114  call new_mesh_dist(i)%init()
115  end do
116 
117  do i = 1, msh%nelv
118  ep => msh%elements(i)%e
119  el%el_idx = ep%id()
120  do j = 1, 8
121  el%v(j)%v_idx = ep%pts(j)%p%id()
122  el%v(j)%v_xyz = ep%pts(j)%p%x
123  end do
124  call new_mesh_dist(parts%data(i))%push(el)
125  end do
126 
127 
128  gdim = msh%gdim
129  call mesh_free(msh)
130 
131  max_recv = 0
132  do i = 0, pe_size - 1
133  max_recv(1) = max(max_recv(1), new_mesh_dist(i)%size())
134  max_recv(2) = max(max_recv(2), new_zone_dist(i)%size())
135  max_recv(3) = max(max_recv(3), new_curve_dist(i)%size())
136  end do
137 
138  call mpi_allreduce(mpi_in_place, max_recv, 3, mpi_integer, &
139  mpi_max, neko_comm, ierr)
140  allocate(recv_buf_msh(max_recv(1)))
141  allocate(recv_buf_zone(max_recv(2)))
142  allocate(recv_buf_curve(max_recv(3)))
143 
144  do i = 1, pe_size - 1
145  src = modulo(pe_rank - i + pe_size, pe_size)
146  dst = modulo(pe_rank + i, pe_size)
147 
148  ! We should use the %array() procedure, which works great for
149  ! GNU, Intel and NEC, but it breaks horribly on Cray when using
150  ! certain data types
151  select type (nmd_array => new_mesh_dist(dst)%data)
152  type is (nmsh_hex_t)
153  call mpi_sendrecv(nmd_array, &
154  new_mesh_dist(dst)%size(), mpi_nmsh_hex, dst, 0, recv_buf_msh, &
155  max_recv(1), mpi_nmsh_hex, src, 0, neko_comm, status, ierr)
156  end select
157  call mpi_get_count(status, mpi_nmsh_hex, recv_size, ierr)
158 
159  do j = 1, recv_size
160  call new_mesh_dist(pe_rank)%push(recv_buf_msh(j))
161  end do
162 
163  ! We should use the %array() procedure, which works great for
164  ! GNU, Intel and NEC, but it breaks horribly on Cray when using
165  ! certain data types
166  select type (nzd_array => new_zone_dist(dst)%data)
167  type is (nmsh_zone_t)
168  call mpi_sendrecv(nzd_array, &
169  new_zone_dist(dst)%size(), mpi_nmsh_zone, dst, 1, recv_buf_zone,&
170  max_recv(2), mpi_nmsh_zone, src, 1, neko_comm, status, ierr)
171  end select
172  call mpi_get_count(status, mpi_nmsh_zone, recv_size, ierr)
173 
174  do j = 1, recv_size
175  call new_zone_dist(pe_rank)%push(recv_buf_zone(j))
176  end do
177 
178  call mpi_sendrecv(new_curve_dist(dst)%array(), &
179  new_curve_dist(dst)%size(), mpi_nmsh_curve, dst, 2, recv_buf_curve,&
180  max_recv(3), mpi_nmsh_curve, src, 2, neko_comm, status, ierr)
181  call mpi_get_count(status, mpi_nmsh_curve, recv_size, ierr)
182 
183  do j = 1, recv_size
184  call new_curve_dist(pe_rank)%push(recv_buf_curve(j))
185  end do
186  end do
187 
188  deallocate(recv_buf_msh)
189  deallocate(recv_buf_zone)
190  deallocate(recv_buf_curve)
191 
192  do i = 0, pe_size - 1
193  if (i .ne. pe_rank) then
194  call new_mesh_dist(i)%free
195  call new_zone_dist(i)%free
196  call new_curve_dist(i)%free
197  end if
198  end do
199 
200  !
201  ! Create a new mesh based on the given distribution
202  !
203  call mesh_init(msh, gdim, new_mesh_dist(pe_rank)%size())
204 
205  call el_map%init(new_mesh_dist(pe_rank)%size())
206  call glb_map%init(new_mesh_dist(pe_rank)%size())
207 
208  ! We should use the %array() procedure, which works great for
209  ! GNU, Intel and NEC, but it breaks horribly on Cray when using
210  ! certain data types
211  select type (np => new_mesh_dist(pe_rank)%data)
212  type is (nmsh_hex_t)
213  do i = 1, new_mesh_dist(pe_rank)%size()
214  do j = 1, 8
215  p(j) = point_t(np(i)%v(j)%v_xyz, np(i)%v(j)%v_idx)
216  end do
217  call mesh_add_element(msh, i, &
218  p(1), p(2), p(3), p(4), p(5), p(6), p(7), p(8))
219 
220  if (el_map%get(np(i)%el_idx, tmp) .gt. 0) then
221  ! Old glb to new local
222  tmp = i
223  call el_map%set(np(i)%el_idx, tmp)
224 
225  ! Old glb to new glb
226  tmp = msh%elements(i)%e%id()
227  call glb_map%set(np(i)%el_idx, tmp)
228  else
229  call neko_error('Global element id already defined')
230  end if
231  end do
232  end select
233  call new_mesh_dist(pe_rank)%free()
234 
235 
236  !
237  ! Figure out new mesh distribution (necessary for periodic zones)
238  !
239  call pe_lst%init()
240 
241  ! We should use the %array() procedure, which works great for
242  ! GNU, Intel and NEC, but it breaks horribly on Cray when using
243  ! certain data types
244  select type(zp => new_zone_dist(pe_rank)%data)
245  type is (nmsh_zone_t)
246  do i = 1, new_zone_dist(pe_rank)%size()
247  if (zp(i)%type .eq. 5) then
248  call pe_lst%push(zp(i)%p_e)
249  end if
250  end do
251  end select
252 
253  max_recv_idx = 2 * pe_lst%size()
254  call mpi_allreduce(mpi_in_place, max_recv_idx, 1, mpi_integer, &
255  mpi_max, neko_comm, ierr)
256  allocate(recv_buf_idx(max_recv_idx))
257  allocate(send_buf_idx(max_recv_idx))
258 
259  do i = 1, pe_size - 1
260  src = modulo(pe_rank - i + pe_size, pe_size)
261  dst = modulo(pe_rank + i, pe_size)
262 
263  ! We should use the %array() procedure, which works great for
264  ! GNU, Intel and NEC, but it breaks horribly on Cray when using
265  ! certain data types
266  select type (pe_lst_array => pe_lst%data)
267  type is (integer)
268  call mpi_sendrecv(pe_lst_array, &
269  pe_lst%size(), mpi_integer, dst, 0, recv_buf_idx, &
270  max_recv_idx, mpi_integer, src, 0, neko_comm, status, ierr)
271  end select
272  call mpi_get_count(status, mpi_integer, recv_size, ierr)
273 
274  k = 0
275  do j = 1, recv_size
276  if (glb_map%get(recv_buf_idx(j), tmp) .eq. 0) then
277  k = k + 1
278  send_buf_idx(k) = recv_buf_idx(j)
279  k = k + 1
280  send_buf_idx(k) = tmp
281  end if
282  end do
283 
284  call mpi_sendrecv(send_buf_idx, k, mpi_integer, src, 1, &
285  recv_buf_idx, max_recv_idx, mpi_integer, dst, 1, &
286  neko_comm, status, ierr)
287  call mpi_get_count(status, mpi_integer, recv_size, ierr)
288 
289  do j = 1, recv_size, 2
290  call glb_map%set(recv_buf_idx(j), recv_buf_idx(j+1))
291  end do
292  end do
293  deallocate(recv_buf_idx)
294  deallocate(send_buf_idx)
295  call pe_lst%free()
296 
297  !
298  ! Add zone data for new mesh distribution
299  !
300  zp => new_zone_dist(pe_rank)%array()
301  do i = 1, new_zone_dist(pe_rank)%size()
302  if (el_map%get(zp(i)%e, new_el_idx) .gt. 0) then
303  call neko_error('Missing element after redistribution')
304  end if
305  select case(zp(i)%type)
306  case(1)
307  call mesh_mark_wall_facet(msh, zp(i)%f, new_el_idx)
308  case(2)
309  call mesh_mark_inlet_facet(msh, zp(i)%f, new_el_idx)
310  case(3)
311  call mesh_mark_outlet_facet(msh, zp(i)%f, new_el_idx)
312  case(4)
313  call mesh_mark_sympln_facet(msh, zp(i)%f, new_el_idx)
314  case(5)
315  if (glb_map%get(zp(i)%p_e, new_pel_idx) .gt. 0) then
316  call neko_error('Missing periodic element after redistribution')
317  end if
318 
319  call mesh_mark_periodic_facet(msh, zp(i)%f, new_el_idx, &
320  zp(i)%p_f, new_pel_idx, zp(i)%glb_pt_ids)
321  end select
322  end do
323  do i = 1, new_zone_dist(pe_rank)%size()
324  if (el_map%get(zp(i)%e, new_el_idx) .gt. 0) then
325  call neko_error('Missing element after redistribution')
326  end if
327  select case(zp(i)%type)
328  case(5)
329  if (glb_map%get(zp(i)%p_e, new_pel_idx) .gt. 0) then
330  call neko_error('Missing periodic element after redistribution')
331  end if
332 
333  call mesh_apply_periodic_facet(msh, zp(i)%f, new_el_idx, &
334  zp(i)%p_f, new_pel_idx, zp(i)%glb_pt_ids)
335  end select
336  end do
337 
338  call new_zone_dist(pe_rank)%free()
339 
340  !
341  ! Add curve element information for new mesh distribution
342  !
343  cp => new_curve_dist(pe_rank)%array()
344  do i = 1, new_curve_dist(pe_rank)%size()
345  if (el_map%get(cp(i)%e, new_el_idx) .gt. 0) then
346  call neko_error('Missing element after redistribution')
347  end if
348  call mesh_mark_curve_element(msh, new_el_idx, cp(i)%curve_data, cp(i)%type)
349  end do
350  call new_curve_dist(pe_rank)%free()
351 
352  call mesh_finalize(msh)
353 
354  end subroutine redist_mesh

357  subroutine redist_zone(msh, z, type, parts, new_dist)
358  type(mesh_t), intent(inout) :: msh
359  class(zone_t), intent(in) :: z
360  integer, intent(in) :: type
361  type(mesh_fld_t), intent(in) :: parts
362  type(stack_nz_t), intent(inout), allocatable :: new_dist(:)
363  type(nmsh_zone_t) :: nmsh_zone
364  integer :: i, j, zone_el
365 
366  select type(zp => z)
367  type is (zone_periodic_t)
368  do i = 1, zp%size
369  zone_el = zp%facet_el(i)%x(2)
370  nmsh_zone%e = zp%facet_el(i)%x(2) + msh%offset_el
371  nmsh_zone%f = zp%facet_el(i)%x(1)
372  nmsh_zone%p_e = zp%p_facet_el(i)%x(2)
373  nmsh_zone%p_f = zp%p_facet_el(i)%x(1)
374  nmsh_zone%glb_pt_ids = zp%p_ids(i)%x
375  nmsh_zone%type = type
376  call new_dist(parts%data(zone_el))%push(nmsh_zone)
377  end do
378  type is (zone_t)
379  do i = 1, zp%size
380  zone_el = zp%facet_el(i)%x(2)
381  nmsh_zone%e = zp%facet_el(i)%x(2) + msh%offset_el
382  nmsh_zone%f = zp%facet_el(i)%x(1)
383  nmsh_zone%type = type
384  call new_dist(parts%data(zone_el))%push(nmsh_zone)
385  end do
386  end select
387  end subroutine redist_zone
