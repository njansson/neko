159  function gmres_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
160  class(gmres_t), intent(inout) :: this
161  class(ax_t), intent(inout) :: ax
162  type(field_t), intent(inout) :: x
163  integer, intent(inout) :: n
164  real(kind=rp), dimension(n), intent(inout) :: f
165  type(coef_t), intent(inout) :: coef
166  type(bc_list_t), intent(inout) :: blst
167  type(gs_t), intent(inout) :: gs_h
168  type(ksp_monitor_t) :: ksp_results
169  integer, optional, intent(in) :: niter
170  integer :: iter
171  integer :: i, j, k, l, ierr
172  real(kind=rp) :: w_plus(neko_blk_size), x_plus(neko_blk_size)
173  real(kind=rp) :: rnorm, alpha, temp, lr, alpha2, norm_fac
174  logical :: conv
175 
176  conv = .false.
177  iter = 0
178 
179  associate(w => this%w, c => this%c, r => this%r, z => this%z, h => this%h, &
180  v => this%v, s => this%s, gam => this%gam, wk1 =>this%wk1)
181 
182  norm_fac = 1.0_rp / sqrt(coef%volume)
183  call rzero(x%x, n)
184  call rzero(gam, this%lgmres + 1)
185  call rone(s, this%lgmres)
186  call rone(c, this%lgmres)
187  call rzero(h, this%lgmres * this%lgmres)
188  do while (.not. conv .and. iter .lt. niter)
189 
190  if(iter.eq.0) then
191  call copy(r, f, n)
192  else
193  call copy(r, f, n)
194  call ax%compute(w, x%x, coef, x%msh, x%Xh)
195  call gs_op(gs_h, w, n, gs_op_add)
196  call bc_list_apply(blst, w, n)
197  call sub2(r, w, n)
198  end if
199 
200  gam(1) = sqrt(glsc3(r, r, coef%mult, n))
201  if(iter.eq.0) then
202  ksp_results%res_start = gam(1) * norm_fac
203  end if
204 
205  if (gam(1) .eq. 0) return
206 
207  rnorm = 0.0_rp
208  temp = 1.0_rp / gam(1)
209  call cmult2(v(1,1), r, temp, n)
210  do j = 1, this%lgmres
211  iter = iter+1
212 
213  call this%M%solve(z(1,j), v(1,j), n)
214 
215  call ax%compute(w, z(1,j), coef, x%msh, x%Xh)
216  call gs_op(gs_h, w, n, gs_op_add)
217  call bc_list_apply(blst, w, n)
218 
219  do l = 1, j
220  h(l,j) = 0.0_rp
221  enddo
222 
223  do i = 0, n, neko_blk_size
224  if (i + neko_blk_size .le. n) then
225  do l = 1, j
226  do k = 1, neko_blk_size
227  h(l,j) = h(l,j) + &
228  w(i+k) * v(i+k,l) * coef%mult(i+k,1,1,1)
229  end do
230  end do
231  else
232  do k = 1, n-i
233  do l = 1, j
234  h(l,j) = h(l,j) + &
235  w(i+k) * v(i+k,l) * coef%mult(i+k,1,1,1)
236  end do
237  end do
238  end if
239  end do
240 
241  call mpi_allreduce(h(1,j), wk1, j, &
242  mpi_real_precision, mpi_sum, neko_comm, ierr)
243  call copy(h(1,j), wk1, j)
244 
245  alpha2 = 0.0_rp
246  do i = 0,n,neko_blk_size
247  if (i + neko_blk_size .le. n) then
248  do k = 1, neko_blk_size
249  w_plus(k) = 0.0_rp
250  end do
251  do l = 1,j
252  do k = 1, neko_blk_size
253  w_plus(k) = w_plus(k) - h(l,j) * v(i+k,l)
254  end do
255  end do
256  do k = 1, neko_blk_size
257  w(i+k) = w(i+k) + w_plus(k)
258  alpha2 = alpha2 + w(i+k)**2 * coef%mult(i+k,1,1,1)
259  end do
260  else
261  do k = 1, n-i
262  w_plus(1) = 0.0_rp
263  do l = 1, j
264  w_plus(1) = w_plus(1) - h(l,j) * v(i+k,l)
265  end do
266  w(i+k) = w(i+k) + w_plus(1)
267  alpha2 = alpha2 + (w(i+k)**2) * coef%mult(i+k,1,1,1)
268  end do
269  end if
270  end do
271 
272  call mpi_allreduce(alpha2, temp, 1, &
273  mpi_real_precision, mpi_sum, neko_comm, ierr)
274  alpha2 = temp
275  alpha = sqrt(alpha2)
276  do i=1,j-1
277  temp = h(i,j)
278  h(i ,j) = c(i)*temp + s(i) * h(i+1,j)
279  h(i+1,j) = -s(i)*temp + c(i) * h(i+1,j)
280  end do
281 
282  rnorm = 0.0_rp
283  if(alpha .eq. 0.0_rp) then
284  conv = .true.
285  exit
286  end if
287 
288  lr = sqrt(h(j,j) * h(j,j) + alpha**2)
289  temp = 1.0_rp / lr
290  c(j) = h(j,j) * temp
291  s(j) = alpha * temp
292  h(j,j) = lr
293  gam(j+1) = -s(j) * gam(j)
294  gam(j) = c(j) * gam(j)
295 
296  rnorm = abs(gam(j+1)) * norm_fac
297  if (rnorm .lt. this%abs_tol) then
298  conv = .true.
299  exit
300  end if
301 
302  if (iter + 1 .gt. niter) exit
303 
304  if( j .lt. this%lgmres) then
305  temp = 1.0_rp / alpha
306  call cmult2(v(1,j+1), w, temp, n)
307  end if
308 
309  end do
310 
311  j = min(j, this%lgmres)
312  do k = j, 1, -1
313  temp = gam(k)
314  do i = j, k+1, -1
315  temp = temp - h(k,i) * c(i)
316  end do
317  c(k) = temp / h(k,k)
318  end do
319 
320  do i = 0, n, neko_blk_size
321  if (i + neko_blk_size .le. n) then
322  do k = 1, neko_blk_size
323  x_plus(k) = 0.0_rp
324  end do
325  do l = 1,j
326  do k = 1, neko_blk_size
327  x_plus(k) = x_plus(k) + c(l) * z(i+k,l)
328  end do
329  end do
330  do k = 1, neko_blk_size
331  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + x_plus(k)
332  end do
333  else
334  do k = 1, n-i
335  x_plus(1) = 0.0_rp
336  do l = 1, j
337  x_plus(1) = x_plus(1) + c(l) * z(i+k,l)
338  end do
339  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + x_plus(1)
340  end do
341  end if
342  end do
343  end do
344 
345  end associate
346 
347  ksp_results%res_final = rnorm
348  ksp_results%iter = iter
349 
350  end function gmres_solve
351
