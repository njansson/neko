83  subroutine tensr3(v, nv, u, nu, A, Bt, Ct, w)
84  integer :: nv
85  integer :: nu
86  real(kind=rp), intent(inout) :: v(nv, nv, nv)
87  real(kind=rp), intent(inout) :: u(nu, nu, nu)
88  real(kind=rp), intent(inout) :: w(nu*nu*nv)
89  real(kind=rp), intent(inout) :: a(nv, nu)
90  real(kind=rp), intent(inout) :: bt(nu, nv)
91  real(kind=rp), intent(inout) :: ct(nu, nv)
92  integer :: j, k, l, nunu, nvnv, nunv
93 
94  nunu = nu**2
95  nvnv = nv**2
96  nunv = nu*nv
97 
99 
100  call mxm(a, nv, u, nu, v, nunu)
101  k = 1
102  l = 1
103  do j = 1, nu
104  call mxm(v(k, 1, 1), nv, bt, nu, w(l), nv)
105  k = k + nunv
106  l = l + nvnv
107  end do
108  call mxm(w, nvnv, ct, nu, v, nv)
109 
110  end subroutine tensr3

218  subroutine addtnsr(s, h1, h2, h3, nx, ny, nz)
219 
220  !Map and add to S a tensor product form of the three functions H1,H2,H3.
221  !This is a single element routine used for deforming geometry.
222  integer, intent(in) :: nx, ny, nz
223  real(kind=rp), intent(in) :: h1(nx), h2(ny), h3(nz)
224  real(kind=rp), intent(inout) :: s(nx, ny, nz)
225  real(kind=rp) :: hh
226  integer :: ix, iy, iz
227 
228  do iz = 1,nz
229  do iy = 1,ny
230  hh = h2(iy)*h3(iz)
231  do ix = 1,nx
232  s(ix,iy,iz) = s(ix,iy,iz)+hh*h1(ix)
233  end do
