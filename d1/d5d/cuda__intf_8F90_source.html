119  subroutine cuda_device_name(name)
120  character(len=*), intent(inout) :: name
121  character(kind=c_char, len=8192), target :: prop
122  integer :: end_pos
123 
124  !
125  ! Yes this is an ugly hack!
126  ! Since we're only interested in the device name (first 256 bytes)
127  ! we pass down a large enough chunk of memory to the cuda runtime
128  ! and extract what we need later on
129  !
130  ! This will of course break if sizeof(cudaDeviceProp) > 8192
131  !
132 
133  if (cudagetdeviceproperties(c_loc(prop), 0) .ne. cudasuccess) then
134  call neko_error('Failed to query device')
135  end if
136 
137  end_pos = scan(prop(1:256), c_null_char)
138  if(end_pos .ge. 2) then
139  name(1:end_pos-1) = prop(1:end_pos-1)
140  endif
141  end subroutine cuda_device_name
