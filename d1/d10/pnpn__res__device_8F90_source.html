214  subroutine pnpn_prs_res_device_compute(p, p_res, u, v, w, u_e, v_e, w_e, &
215  ta1, ta2, ta3, wa1, wa2, wa3, work1, work2, f_Xh, c_Xh, gs_Xh, &
216  bc_prs_surface, bc_sym_surface, Ax, bd, dt, Re, rho)
217  type(field_t), intent(inout) :: p, u, v, w
218  type(field_t), intent(inout) :: u_e, v_e, w_e
219  type(field_t), intent(inout) :: ta1, ta2, ta3
220  type(field_t), intent(inout) :: wa1, wa2, wa3
221  type(field_t), intent(inout) :: work1, work2
222  type(field_t), intent(inout) :: p_res
223  type(source_t), intent(inout) :: f_Xh
224  type(coef_t), intent(inout) :: c_Xh
225  type(gs_t), intent(inout) :: gs_Xh
226  type(facet_normal_t), intent(inout) :: bc_prs_surface
227  type(facet_normal_t), intent(inout) :: bc_sym_surface
228  class(ax_t), intent(inout) :: Ax
229  real(kind=rp), intent(inout) :: bd
230  real(kind=rp), intent(in) :: dt
231  real(kind=rp), intent(in) :: re
232  real(kind=rp), intent(in) :: rho
233  real(kind=rp) :: dtbd
234  integer :: n, gdim
235 
236  n = u%dof%size()
237  gdim = c_xh%msh%gdim
238 
239  call curl(ta1, ta2, ta3, u_e, v_e, w_e, work1, work2, c_xh)
240  call curl(wa1, wa2, wa3, ta1, ta2, ta3, work1, work2, c_xh)
241 
242 #ifdef HAVE_HIP
243  call pnpn_prs_res_part1_hip(ta1%x_d, ta2%x_d, ta3%x_d, &
244  wa1%x_d, wa2%x_d, wa3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, &
245  c_xh%B_d, c_xh%h1_d, re, rho, n)
246 
247 #elif HAVE_CUDA
248  call pnpn_prs_res_part1_cuda(ta1%x_d, ta2%x_d, ta3%x_d, &
249  wa1%x_d, wa2%x_d, wa3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, &
250  c_xh%B_d, c_xh%h1_d, re, rho, n)
251 #elif HAVE_OPENCL
252  call pnpn_prs_res_part1_opencl(ta1%x_d, ta2%x_d, ta3%x_d, &
253  wa1%x_d, wa2%x_d, wa3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, &
254  c_xh%B_d, c_xh%h1_d, re, rho, n)
255 #endif
256  c_xh%ifh2 = .false.
257 
258  call gs_op(gs_xh, ta1, gs_op_add)
259  call gs_op(gs_xh, ta2, gs_op_add)
260  call gs_op(gs_xh, ta3, gs_op_add)
261 
262  call device_opcolv(ta1%x_d, ta2%x_d, ta3%x_d, c_xh%Binv_d, gdim, n)
263 
264  call cdtp(wa1%x, ta1%x, c_xh%drdx, c_xh%dsdx, c_xh%dtdx, c_xh)
265  call cdtp(wa2%x, ta2%x, c_xh%drdy, c_xh%dsdy, c_xh%dtdy, c_xh)
266  call cdtp(wa3%x, ta3%x, c_xh%drdz, c_xh%dsdz, c_xh%dtdz, c_xh)
267 
268  call ax%compute(p_res%x,p%x,c_xh,p%msh,p%Xh)
269 
270 #ifdef HAVE_HIP
271  call pnpn_prs_res_part2_hip(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, n);
272 #elif HAVE_CUDA
273  call pnpn_prs_res_part2_cuda(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, n);
274 #elif HAVE_OPENCL
275  call pnpn_prs_res_part2_opencl(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, n);
276 #endif
277 
278  !
279  ! Surface velocity terms
280  call device_rzero(wa1%x_d, n)
281  call device_rzero(wa2%x_d, n)
282  call device_rzero(wa3%x_d, n)
283  dtbd = 1.0_rp
284 
285  call bc_sym_surface%apply_surfvec_dev(wa1%x_d, wa2%x_d, wa3%x_d, ta1%x_d, ta2%x_d, ta3%x_d)
286 
287 #ifdef HAVE_HIP
288  call pnpn_prs_res_part3_hip(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, dtbd, n);
289 #elif HAVE_CUDA
290  call pnpn_prs_res_part3_cuda(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, dtbd, n);
291 #elif HAVE_OPENCL
292  call pnpn_prs_res_part3_opencl(p_res%x_d, wa1%x_d, wa2%x_d, wa3%x_d, dtbd, n);
293 #endif
294  !
295  dtbd = bd / dt
296 
297  call device_rzero(ta1%x_d, n)
298  call device_rzero(ta2%x_d, n)
299  call device_rzero(ta3%x_d, n)
300 
301  call bc_prs_surface%apply_surfvec_dev(ta1%x_d, ta2%x_d, ta3%x_d, &
302  u%x_d, v%x_d, w%x_d)
303 
304 #ifdef HAVE_HIP
305  call pnpn_prs_res_part3_hip(p_res%x_d, ta1%x_d, ta2%x_d, ta3%x_d, dtbd, n);
306 #elif HAVE_CUDA
307  call pnpn_prs_res_part3_cuda(p_res%x_d, ta1%x_d, ta2%x_d, ta3%x_d, dtbd, n);
308 #elif HAVE_OPENCL
309  call pnpn_prs_res_part3_opencl(p_res%x_d, ta1%x_d, ta2%x_d, ta3%x_d, dtbd, n);
310 #endif
311 
312  end subroutine pnpn_prs_res_device_compute

314  subroutine pnpn_vel_res_device_compute(Ax, u, v, w, u_res, v_res, w_res, &
315  p, ta1, ta2, ta3, f_Xh, c_Xh, msh, Xh, Re, rho, bd, dt, n)
316  class(ax_t), intent(in) :: Ax
317  type(mesh_t), intent(inout) :: msh
318  type(space_t), intent(inout) :: Xh
319  type(field_t), intent(inout) :: p, u, v, w
320  type(field_t), intent(inout) :: u_res, v_res, w_res
321  type(field_t), intent(inout) :: ta1, ta2, ta3
322  type(source_t), intent(inout) :: f_Xh
323  type(coef_t), intent(inout) :: c_Xh
324  real(kind=rp), intent(in) :: re
325  real(kind=rp), intent(in) :: rho
326  real(kind=rp), intent(in) :: bd
327  real(kind=rp), intent(in) :: dt
328  integer, intent(in) :: n
329 
330  call device_cfill(c_xh%h1_d, (1.0_rp / re), n)

333 
334  call ax%compute(u_res%x, u%x, c_xh, msh, xh)
335  call ax%compute(v_res%x, v%x, c_xh, msh, xh)
336  call ax%compute(w_res%x, w%x, c_xh, msh, xh)
337 
338  call opgrad(ta1%x, ta2%x, ta3%x, p%x, c_xh)
339 
340 #ifdef HAVE_HIP
341  call pnpn_vel_res_update_hip(u_res%x_d, v_res%x_d, w_res%x_d, &
342  ta1%x_d, ta2%x_d, ta3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, n)
343 #elif HAVE_CUDA
344  call pnpn_vel_res_update_cuda(u_res%x_d, v_res%x_d, w_res%x_d, &
345  ta1%x_d, ta2%x_d, ta3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, n)
346 #elif HAVE_OPENCL
347  call pnpn_vel_res_update_opencl(u_res%x_d, v_res%x_d, w_res%x_d, &
348  ta1%x_d, ta2%x_d, ta3%x_d, f_xh%u_d, f_xh%v_d, f_xh%w_d, n)
349 #endif
350 
351  end subroutine pnpn_vel_res_device_compute
