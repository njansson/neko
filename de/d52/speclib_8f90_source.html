77 !
78 ! ABBRIVIATIONS:
79 !
80 ! M - Set of mesh points
81 ! Z - Set of collocation/quadrature points
82 ! W - Set of quadrature weights
83 ! H - Lagrangian interpolant
84 ! D - Derivative operator
85 ! I - Interpolation operator
86 ! GL - Gauss Legendre
87 ! GLL - Gauss-Lobatto Legendre
88 ! GJ - Gauss Jacobi
89 ! GLJ - Gauss-Lobatto Jacobi
90 !
91 !
92 ! MAIN ROUTINES:
93 !
94 ! Points and weights:
95 !
96 ! ZWGL Compute Gauss Legendre points and weights
97 ! ZWGLL Compute Gauss-Lobatto Legendre points and weights
98 ! ZWGJ Compute Gauss Jacobi points and weights (general)
99 ! ZWGLJ Compute Gauss-Lobatto Jacobi points and weights (general)
100 !
101 ! Lagrangian interpolants:
102 !
103 ! HGL Compute Gauss Legendre Lagrangian interpolant
104 ! HGLL Compute Gauss-Lobatto Legendre Lagrangian interpolant
105 ! HGJ Compute Gauss Jacobi Lagrangian interpolant (general)
106 ! HGLJ Compute Gauss-Lobatto Jacobi Lagrangian interpolant (general)
107 !
108 ! Derivative operators:
109 !
110 ! DGLL Compute Gauss-Lobatto Legendre derivative matrix
111 ! DGLLGL Compute derivative matrix for a staggered mesh (GLL->GL)
112 ! DGJ Compute Gauss Jacobi derivative matrix (general)
113 ! DGLJ Compute Gauss-Lobatto Jacobi derivative matrix (general)
114 ! DGLJGJ Compute derivative matrix for a staggered mesh (GLJ->GJ) (general)
115 !
116 ! Interpolation operators:
117 !
118 ! IGLM Compute interpolation operator GL -> M
119 ! IGLLM Compute interpolation operator GLL -> M
120 ! IGJM Compute interpolation operator GJ -> M (general)
121 ! IGLJM Compute interpolation operator GLJ -> M (general)
122 !
123 ! Other:
124 !
125 ! PNLEG Compute Legendre polynomial of degree N
126 ! PNDLEG Compute derivative of Legendre polynomial of degree N
127 !
128 ! Comments:
129 !
130 ! Note that many of the above routines exist in both single and
131 ! double precision. If the name of the single precision routine is
132 ! SUB, the double precision version is called SUBD. In most cases
133 ! all the "low-level" arithmetic is done in double precision, even
134 ! for the single precsion versions.
135 !
136 ! Useful references:
137 !
138 ! [1] Gabor Szego: Orthogonal Polynomials, American Mathematical Society,
139 ! Providence, Rhode Island, 1939.
140 ! [2] Abramowitz & Stegun: Handbook of Mathematical Functions,
141 ! Dover, New York, 1972.
142 ! [3] Canuto, Hussaini, Quarteroni & Zang: Spectral Methods in Fluid
143 ! Dynamics, Springer-Verlag, 1988.

189  SUBROUTINE zwgj (Z,W,NP,ALPHA,BETA)
190 !--------------------------------------------------------------------
191 !
192 ! Generate NP GAUSS JACOBI points (Z) and weights (W)
193 ! associated with Jacobi polynomial P(N)(alpha>-1,beta>-1).
194 ! The polynomial degree N=NP-1.
195 ! Single precision version.
196 !
197 !--------------------------------------------------------------------
198  parameter(nmax=84)
199  parameter(nzd = nmax)
200  REAL(KIND=rp) zd(nzd),wd(nzd),alphad,betad
201  REAL(KIND=rp) z(1),w(1),alpha,beta
202 
203  npmax = nzd
204  IF (np.GT.npmax) THEN
205  WRITE (6,*) 'Too large polynomial degree in ZWGJ'
206  WRITE (6,*) 'Maximum polynomial degree is',nmax
207  WRITE (6,*) 'Here NP=',np
208  call neko_error
209  ENDIF
210  alphad = alpha
211  betad = beta
212  CALL zwgjd (zd,wd,np,alphad,betad)
213  DO 100 i=1,np
214  z(i) = zd(i)
215  w(i) = wd(i)
216  100 CONTINUE
217  RETURN
218  END SUBROUTINE

220  SUBROUTINE zwgjd (Z,W,NP,ALPHA,BETA)
221 !--------------------------------------------------------------------
222 !
223 ! Generate NP GAUSS JACOBI points (Z) and weights (W)
224 ! associated with Jacobi polynomial P(N)(alpha>-1,beta>-1).
225 ! The polynomial degree N=NP-1.
226 ! Double precision version.
227 !
228 !--------------------------------------------------------------------
229  IMPLICIT REAL(KIND=RP) (a-h,o-z)
230  REAL(KIND=rp) z(1),w(1),alpha,beta
231 
232  n = np-1
233  dn = ((n))
234  one = 1.
235  two = 2.
236  apb = alpha+beta
237 
238  IF (np.LE.0) THEN
239  WRITE (6,*) 'ZWGJD: Minimum number of Gauss points is 1',np
240  call neko_error
241  ENDIF
242  IF ((alpha.LE.-one).OR.(beta.LE.-one)) THEN
243  WRITE (6,*) 'ZWGJD: Alpha and Beta must be greater than -1'
244  call neko_error
245  ENDIF
246 
247  IF (np.EQ.1) THEN
248  z(1) = (beta-alpha)/(apb+two)
249  w(1) = gammaf(alpha+one)*gammaf(beta+one)/gammaf(apb+two) &
250  * two**(apb+one)
251  RETURN
252  ENDIF
253 
254  CALL jacg (z,np,alpha,beta)
255 
256  np1 = n+1
257  np2 = n+2
258  dnp1 = ((np1))
259  dnp2 = ((np2))
260  fac1 = dnp1+alpha+beta+one
261  fac2 = fac1+dnp1
262  fac3 = fac2+one
263  fnorm = pnormj(np1,alpha,beta)
264  rcoef = (fnorm*fac2*fac3)/(two*fac1*dnp2)
265  DO 100 i=1,np
266  CALL jacobf (p,pd,pm1,pdm1,pm2,pdm2,np2,alpha,beta,z(i))
267  w(i) = -rcoef/(p*pdm1)
268  100 CONTINUE
269  RETURN
270  END SUBROUTINE

303  SUBROUTINE zwgljd (Z,W,NP,ALPHA,BETA)
304 !--------------------------------------------------------------------
305 !
306 ! Generate NP GAUSS LOBATTO JACOBI points (Z) and weights (W)
307 ! associated with Jacobi polynomial P(N)(alpha>-1,beta>-1).
308 ! The polynomial degree N=NP-1.
309 ! Double precision version.
310 !
311 !--------------------------------------------------------------------
312  IMPLICIT REAL(KIND=RP) (a-h,o-z)
313  REAL(KIND=rp) z(np),w(np),alpha,beta
314 
315  n = np-1
316  nm1 = n-1
317  one = 1.
318  two = 2.
319 
320  IF (np.LE.1) THEN
321  WRITE (6,*) 'ZWGLJD: Minimum number of Gauss-Lobatto points is 2'
322  WRITE (6,*) 'ZWGLJD: alpha,beta:',alpha,beta,np
323  call neko_error
324  ENDIF
325  IF ((alpha.LE.-one).OR.(beta.LE.-one)) THEN
326  WRITE (6,*) 'ZWGLJD: Alpha and Beta must be greater than -1'
327  call neko_error
328  ENDIF
329 
330  IF (nm1.GT.0) THEN
331  alpg = alpha+one
332  betg = beta+one
333  CALL zwgjd (z(2),w(2),nm1,alpg,betg)
334  ENDIF
335  z(1) = -one
336  z(np) = one
337  DO 100 i=2,np-1
338  w(i) = w(i)/(one-z(i)**2)
339  100 CONTINUE
340  CALL jacobf (p,pd,pm1,pdm1,pm2,pdm2,n,alpha,beta,z(1))
341  w(1) = endw1(n,alpha,beta)/(two*pd)
342  CALL jacobf (p,pd,pm1,pdm1,pm2,pdm2,n,alpha,beta,z(np))
343  w(np) = endw2(n,alpha,beta)/(two*pd)
344 
345 ! RETURN
346  END SUBROUTINE

348  REAL(KIND=rp) FUNCTION endw1 (N,ALPHA,BETA)
349  IMPLICIT REAL(KIND=RP) (a-h,o-z)
350  REAL(kind=rp) alpha,beta
351  zero = 0.
352  one = 1.
353  two = 2.
354  three = 3.
355  four = 4.
356  apb = alpha+beta
357  IF (n.EQ.0) THEN
358  endw1 = zero
359  RETURN
360  ENDIF
361  f1 = gammaf(alpha+two)*gammaf(beta+one)/gammaf(apb+three)
362  f1 = f1*(apb+two)*two**(apb+two)/two
363  IF (n.EQ.1) THEN
364  endw1 = f1
365  RETURN
366  ENDIF
367  fint1 = gammaf(alpha+two)*gammaf(beta+one)/gammaf(apb+three)
368  fint1 = fint1*two**(apb+two)
369  fint2 = gammaf(alpha+two)*gammaf(beta+two)/gammaf(apb+four)
370  fint2 = fint2*two**(apb+three)
371  f2 = (-two*(beta+two)*fint1 + (apb+four)*fint2) &
372  * (apb+three)/four
373  IF (n.EQ.2) THEN
374  endw1 = f2
375  RETURN
376  ENDIF
377  DO 100 i=3,n
378  di = ((i-1))
379  abn = alpha+beta+di
380  abnn = abn+di
381  a1 = -(two*(di+alpha)*(di+beta))/(abn*abnn*(abnn+one))
382  a2 = (two*(alpha-beta))/(abnn*(abnn+two))
383  a3 = (two*(abn+one))/((abnn+two)*(abnn+one))
384  f3 = -(a2*f2+a1*f1)/a3
385  f1 = f2
386  f2 = f3
387  100 CONTINUE
388  endw1 = f3
389  RETURN
390  END FUNCTION
391 
392  REAL(kind=rp) FUNCTION endw2 (N,ALPHA,BETA)
393  IMPLICIT REAL(KIND=RP) (a-h,o-z)
394  REAL(kind=rp) alpha,beta
395  zero = 0.
396  one = 1.
397  two = 2.
398  three = 3.
399  four = 4.
400  apb = alpha+beta
401  IF (n.EQ.0) THEN
402  endw2 = zero
403  RETURN
404  ENDIF
405  f1 = gammaf(alpha+one)*gammaf(beta+two)/gammaf(apb+three)
406  f1 = f1*(apb+two)*two**(apb+two)/two
407  IF (n.EQ.1) THEN
408  endw2 = f1
409  RETURN
410  ENDIF
411  fint1 = gammaf(alpha+one)*gammaf(beta+two)/gammaf(apb+three)
412  fint1 = fint1*two**(apb+two)
413  fint2 = gammaf(alpha+two)*gammaf(beta+two)/gammaf(apb+four)
414  fint2 = fint2*two**(apb+three)
415  f2 = (two*(alpha+two)*fint1 - (apb+four)*fint2) &
416  * (apb+three)/four
417  IF (n.EQ.2) THEN
418  endw2 = f2
419  RETURN
420  ENDIF
421  DO 100 i=3,n
422  di = ((i-1))
423  abn = alpha+beta+di
424  abnn = abn+di
425  a1 = -(two*(di+alpha)*(di+beta))/(abn*abnn*(abnn+one))
426  a2 = (two*(alpha-beta))/(abnn*(abnn+two))
427  a3 = (two*(abn+one))/((abnn+two)*(abnn+one))
428  f3 = -(a2*f2+a1*f1)/a3
429  f1 = f2
430  f2 = f3
431  100 CONTINUE
432  endw2 = f3
433  RETURN
434  END FUNCTION

436  REAL(kind=rp) FUNCTION gammaf (X)
437  IMPLICIT REAL(KIND=RP) (a-h,o-z)
438  REAL(kind=rp) x
439  zero = 0.0
440  half = 0.5
441  one = 1.0
442  two = 2.0
443  four = 4.0
444  pi = four*atan(one)
445  gammaf = one
446  IF (x.EQ.-half) gammaf = -two*sqrt(pi)
447  IF (x.EQ. half) gammaf = sqrt(pi)
448  IF (x.EQ. one ) gammaf = one
449  IF (x.EQ. two ) gammaf = one
450  IF (x.EQ. 1.5 ) gammaf = sqrt(pi)/2.
451  IF (x.EQ. 2.5) gammaf = 1.5*sqrt(pi)/2.
452  IF (x.EQ. 3.5) gammaf = 0.5*(2.5*(1.5*sqrt(pi)))
453  IF (x.EQ. 3. ) gammaf = 2.
454  IF (x.EQ. 4. ) gammaf = 6.
455  IF (x.EQ. 5. ) gammaf = 24.
456  IF (x.EQ. 6. ) gammaf = 120.
457  RETURN
458  END FUNCTION

460  REAL(kind=rp) FUNCTION pnormj (N,ALPHA,BETA)
461  IMPLICIT REAL(KIND=RP) (a-h,o-z)
462  REAL(kind=rp) alpha,beta
463  one = 1.
464  two = 2.
465  dn = ((n))
466  const = alpha+beta+one
467  IF (n.LE.1) THEN
468  prod = gammaf(dn+alpha)*gammaf(dn+beta)
469  prod = prod/(gammaf(dn)*gammaf(dn+alpha+beta))
470  pnormj = prod * two**const/(two*dn+const)
471  RETURN
472  ENDIF
473  prod = gammaf(alpha+one)*gammaf(beta+one)
474  prod = prod/(two*(one+const)*gammaf(const+one))
475  prod = prod*(one+alpha)*(two+alpha)
476  prod = prod*(one+beta)*(two+beta)
477  DO 100 i=3,n
478  dindx = ((i))
479  frac = (dindx+alpha)*(dindx+beta)/(dindx*(dindx+alpha+beta))
480  prod = prod*frac
481  100 CONTINUE
482  pnormj = prod * two**const/(two*dn+const)
483  RETURN
484  END FUNCTION

486  SUBROUTINE jacg (XJAC,NP,ALPHA,BETA)
487 !--------------------------------------------------------------------
488 !
489 ! Compute NP Gauss points XJAC, which are the zeros of the
490 ! Jacobi polynomial J(NP) with parameters ALPHA and BETA.
491 ! ALPHA and BETA determines the specific type of Gauss points.
492 ! Examples:
493 ! ALPHA = BETA = 0.0 -> Legendre points
494 ! ALPHA = BETA = -0.5 -> Chebyshev points
495 !
496 !--------------------------------------------------------------------
497  IMPLICIT REAL(KIND=RP) (a-h,o-z)
498  REAL(KIND=rp) xjac(1)
499  DATA kstop /10/
500  DATA eps/1.0e-12/
501  n = np-1
502  one = 1.
503  dth = 4.*atan(one)/(2.*((n))+2.)
504  DO 40 j=1,np
505  IF (j.EQ.1) THEN
506  x = cos((2.*(((j))-1.)+1.)*dth)
507  ELSE
508  x1 = cos((2.*(((j))-1.)+1.)*dth)
509  x2 = xlast
510  x = (x1+x2)/2.
511  ENDIF
512  DO 30 k=1,kstop
513  CALL jacobf (p,pd,pm1,pdm1,pm2,pdm2,np,alpha,beta,x)
514  recsum = 0.
515  jm = j-1
516  DO 29 i=1,jm
517  recsum = recsum+1./(x-xjac(np-i+1))
518  29 CONTINUE
519  delx = -p/(pd-recsum*p)
520  x = x+delx
521  IF (abs(delx) .LT. eps) GOTO 31
522  30 CONTINUE
523  31 CONTINUE
524  xjac(np-j+1) = x
525  xlast = x
526  40 CONTINUE
527  DO 200 i=1,np
528  xmin = 2.
529  DO 100 j=i,np
530  IF (xjac(j).LT.xmin) THEN
531  xmin = xjac(j)
532  jmin = j
533  ENDIF
534  100 CONTINUE
535  IF (jmin.NE.i) THEN
536  swap = xjac(i)
537  xjac(i) = xjac(jmin)
538  xjac(jmin) = swap
539  ENDIF
540  200 CONTINUE
541  RETURN
542  END SUBROUTINE

544  SUBROUTINE jacobf (POLY,PDER,POLYM1,PDERM1,POLYM2,PDERM2,N,ALP,BET,X)
545 !--------------------------------------------------------------------
546 !
547 ! Computes the Jacobi polynomial (POLY) and its derivative (PDER)
548 ! of degree N at X.
549 !
550 !--------------------------------------------------------------------
551  IMPLICIT REAL(KIND=RP) (a-h,o-z)
552  apb = alp+bet
553  poly = 1.
554  pder = 0.
555  IF (n .EQ. 0) RETURN
556  polyl = poly
557  pderl = pder
558  poly = (alp-bet+(apb+2.)*x)/2.
559  pder = (apb+2.)/2.
560  IF (n .EQ. 1) RETURN
561  DO 20 k=2,n
562  dk = ((k))
563  a1 = 2.*dk*(dk+apb)*(2.*dk+apb-2.)
564  a2 = (2.*dk+apb-1.)*(alp**2-bet**2)
565  b3 = (2.*dk+apb-2.)
566  a3 = b3*(b3+1.)*(b3+2.)
567  a4 = 2.*(dk+alp-1.)*(dk+bet-1.)*(2.*dk+apb)
568  polyn = ((a2+a3*x)*poly-a4*polyl)/a1
569  pdern = ((a2+a3*x)*pder-a4*pderl+a3*poly)/a1
570  psave = polyl
571  pdsave = pderl
572  polyl = poly
573  poly = polyn
574  pderl = pder
575  pder = pdern
576  20 CONTINUE
577  polym1 = polyl
578  pderm1 = pderl
579  polym2 = psave
580  pderm2 = pdsave
581  RETURN
582  END SUBROUTINE

584  REAL(KIND=rp) FUNCTION hgj (II,Z,ZGJ,NP,ALPHA,BETA)
585 !---------------------------------------------------------------------
586 !
587 ! Compute the value of the Lagrangian interpolant HGJ through
588 ! the NP Gauss Jacobi points ZGJ at the point Z.
589 ! Single precision version.
590 !
591 !---------------------------------------------------------------------
592  parameter(nmax=84)
593  parameter(nzd = nmax)

600  WRITE (6,*) 'Here NP=',np

602  ENDIF
603  zd = z
604  DO 100 i=1,np
605  zgjd(i) = zgj(i)
606  100 CONTINUE

608  betad = beta
609  hgj = hgjd(ii,zd,zgjd,np,alphad,betad)
610  RETURN

613  REAL(kind=rp) FUNCTION hgjd (II,Z,ZGJ,NP,ALPHA,BETA)
614 !---------------------------------------------------------------------
615 !
616 ! Compute the value of the Lagrangian interpolant HGJD through
617 ! the NZ Gauss-Lobatto Jacobi points ZGJ at the point Z.
618 ! Double precision version.
619 !

946  REAL(kind=rp) FUNCTION pnleg (Z,N)
947 !---------------------------------------------------------------------
948 !
949 ! Compute the value of the Nth order Legendre polynomial at Z.
950 ! (Simpler than JACOBF)
951 ! Based on the recursion formula for the Legendre polynomials.
952 !
953 !---------------------------------------------------------------------
954 !
955 ! This next statement is to overcome the underflow bug in the i860.
956 ! It can be removed at a later date. 11 Aug 1990 pff.
957 !
958  IMPLICIT REAL(KIND=RP) (a-h,o-z)
959  REAL(kind=rp) z, p1, p2, p3
960  IF(abs(z) .LT. 1.0e-25) z = 0.0
961 
962 
963  p1 = 1.
964  IF (n.EQ.0) THEN
965  pnleg = p1
966  RETURN
967  ENDIF
968  p2 = z
969  p3 = p2
970  DO 10 k = 1, n-1
971  fk = (k)
972  p3 = ((2.*fk+1.)*z*p2 - fk*p1)/(fk+1.)
973  p1 = p2
974  p2 = p3
975  10 CONTINUE
976  pnleg = p3
977  if (n.eq.0) pnleg = 1.
978  RETURN
979  END FUNCTION

981  REAL(kind=rp) FUNCTION pndleg (Z,N)
982 !----------------------------------------------------------------------
983 !
984 ! Compute the derivative of the Nth order Legendre polynomial at Z.
985 ! (Simpler than JACOBF)
986 ! Based on the recursion formula for the Legendre polynomials.
987 !
988 !----------------------------------------------------------------------
989  IMPLICIT REAL(KIND=RP) (a-h,o-z)
990  REAL(kind=rp) p1, p2, p1d, p2d, p3d, z
991  p1 = 1.
992  p2 = z
993  p1d = 0.
994  p2d = 1.
995  p3d = 1.
996  DO 10 k = 1, n-1
997  fk = (k)
998  p3 = ((2.*fk+1.)*z*p2 - fk*p1)/(fk+1.)
999  p3d = ((2.*fk+1.)*p2 + (2.*fk+1.)*z*p2d - fk*p1d)/(fk+1.)
1000  p1 = p2
1001  p2 = p3
1002  p1d = p2d
1003  p2d = p3d
1004  10 CONTINUE
1005  pndleg = p3d
1006  IF (n.eq.0) pndleg = 0.
1007  RETURN
1008  END FUNCTION
