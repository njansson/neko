122  subroutine abbdf_set_bd(this, dtbd)
123  class(abbdf_t), intent(inout) :: this
124  real(kind=rp), intent(inout), dimension(10) :: dtbd
125  real(kind=rp), dimension(10,10) :: bdmat
126  real(kind=rp), dimension(10) :: bdrhs
127  real(kind=rp), dimension(10) :: bd_old
128  real(kind=rp) :: bdf
129  integer, parameter :: ldim = 10
130  integer, dimension(10) :: ir, ic
131  integer :: ibd, nsys, i
132 
133  associate(nbd => this%nbd, bd => this%bd, bd_d => this%bd_d)
134  bd_old = bd
135  nbd = nbd + 1
136  nbd = min(nbd, this%time_order)
137  call rzero(bd, 10)
138  if (nbd .eq. 1) then
139  bd(1) = 1.0_rp
140  bdf = 1.0_rp
141  else if (nbd .ge. 2) then
142  nsys = nbd + 1
143  call bdsys(bdmat, bdrhs, dtbd, nbd, ldim)
144  call lu(bdmat, nsys, ldim, ir, ic)
145  call solve(bdrhs, bdmat, 1, nsys, ldim, ir, ic)
146  do i = 1, nbd
147  bd(i) = bdrhs(i)
148  end do
149  bdf = bdrhs(nbd + 1)
150  endif
151 
152  !Normalize
153  do ibd = nbd, 1, -1
154  bd(ibd + 1) = bd(ibd)
155  end do
156  bd(1) = 1.0_rp
157  do ibd= 1, nbd + 1
158  bd(ibd) = bd(ibd)/bdf
159  end do
160 
161  if (c_associated(bd_d)) then
162  if (maxval(abs(bd - bd_old)) .gt. 1e-10_rp) then
163  call device_memcpy(bd, bd_d, 10, host_to_device)
164  end if
165  end if
166  end associate
167 
168  end subroutine abbdf_set_bd

180  subroutine abbdf_set_abbd(this, dtlag)
181  class(abbdf_t), intent(inout) :: this
182  real(kind=rp), intent(inout), dimension(10) :: dtlag
183  real(kind=rp) :: dt0, dt1, dt2, dts, dta, dtb, dtc, dtd, dte
184  real(kind=rp), dimension(10) :: ab_old
185  associate(nab => this%nab, nbd => this%nbd, ab => this%ab, ab_d => this%ab_d)
186  ab_old = ab
187  nab = nab + 1
188  nab = min(nab, this%time_order)
189 
190  dt0 = dtlag(1)
191  dt1 = dtlag(2)
192  dt2 = dtlag(3)
193  call rzero(ab, 10)
194 
195  if (nab .eq. 1) then
196  ab(1) = 1.0_rp
197  else if (nab .eq. 2) then
198  dta = dt0 / dt1
199  if (nbd .eq. 1) then
200  ab(2) = -0.5_rp * dta
201  ab(1) = 1.0_rp - ab(2)
202  else if (nbd .eq. 2) then
203  ab(2) = -dta
204  ab(1) = 1.0_rp - ab(2)
205  endif
206  else if (nab .eq. 3) then
207  dts = dt1 + dt2
208  dta = dt0 / dt1
209  dtb = dt1 / dt2
210  dtc = dt0 / dt2
211  dtd = dts / dt1
212  dte = dt0 / dts
213  if (nbd .eq. 1) then
214  ab(3) = dte*( 0.5d0*dtb + dtc/3d0 )
215  ab(2) = -0.5_rp * dta - ab(3) * dtd
216  ab(1) = 1.0_rp - ab(2) - ab(3)
217  elseif (nbd .eq. 2) then
218  ab(3) = 2.0_rp / 3.0_rp * dtc * (1.0_rp / dtd + dte)
219  ab(2) = -dta - ab(3) * dtd
220  ab(1) = 1.0_rp - ab(2) - ab(3)
221  elseif (nbd .eq. 3) then
222  ab(3) = dte * (dtb + dtc)
223  ab(2) = -dta * (1.0_rp + dtb + dtc)
224  ab(1) = 1.0_rp - ab(2) - ab(3)
225  endif
226  endif
227 
228  if (c_associated(ab_d)) then
229  if (maxval(abs(ab - ab_old)) .gt. 1e-10_rp) then
230  call device_memcpy(ab, ab_d, 10, host_to_device)
231  end if
232  end if
233  end associate
234 
235  end subroutine abbdf_set_abbd

243  subroutine bdsys (a, b, dt, nbd, ldim)
244  integer :: ldim, j, n, k, i, nsys, nbd
245  real(kind=rp) :: a(ldim,9),b(9),dt(9)
246  real(kind=rp) :: sumdt
247 
248  CALL rzero (a, ldim**2)
249  n = nbd + 1
250  DO j = 1, nbd
251  a(1,j) = 1.0_rp
252  end DO
253  a(1,nbd+1) = 0.0_rp
254  b(1) = 1.0_rp
255  DO j = 1, nbd
256  sumdt = 0.0_rp
257  DO k = 1, j
258  sumdt = sumdt + dt(k)
259  end DO
260  a(2,j) = sumdt
261  end DO
262  a(2,nbd+1) = -dt(1)
263  b(2) = 0.0_rp
264  DO i = 3, nbd + 1
265  DO j = 1, nbd
266  sumdt = 0.0_rp
267  DO k = 1, j
268  sumdt = sumdt + dt(k)
269  end DO
270  a(i,j) = sumdt**(i-1)
271  end DO
272  a(i,nbd+1) = 0.0_rp
273  b(i) = 0.0_rp
274  end DO
275 
276  end subroutine bdsys

307  ic(k) = icm

310  b = a(k,j)

313  END DO
314 300 IF(m .EQ. k) GOTO 500

329  END DO
