372  subroutine mesh_generate_flags(m)
373  type(mesh_t), intent(inout) :: m
374  real(kind=dp) :: u(3), v(3), w(3), temp
375  integer :: e
376 
377  do e = 1, m%nelv
378  if (m%gdim .eq. 2) then
379  m%dfrmd_el(e) = .false.
380  u = m%elements(e)%e%pts(2)%p%x - m%elements(e)%e%pts(1)%p%x
381  v = m%elements(e)%e%pts(4)%p%x - m%elements(e)%e%pts(1)%p%x
382  temp = u(1)*v(1) + u(2)*v(2)
383  if(.not. abscmp(temp, 0d0)) m%dfrmd_el(e) = .true.
384  else
385  m%dfrmd_el(e) = .false.
386  u = m%elements(e)%e%pts(2)%p%x - m%elements(e)%e%pts(1)%p%x
387  v = m%elements(e)%e%pts(4)%p%x - m%elements(e)%e%pts(1)%p%x
388  w = m%elements(e)%e%pts(5)%p%x - m%elements(e)%e%pts(1)%p%x
389  temp = u(1)*v(1) + u(2)*v(2) + u(3)*v(3)
390  if(.not. abscmp(temp, 0d0)) m%dfrmd_el(e) = .true.
391  temp = u(1)*w(1) + u(2)*w(2) + u(3)*w(3)
392  if(.not. abscmp(temp, 0d0)) m%dfrmd_el(e) = .true.
393  u = m%elements(e)%e%pts(8)%p%x - m%elements(e)%e%pts(7)%p%x
394  v = m%elements(e)%e%pts(6)%p%x - m%elements(e)%e%pts(7)%p%x
395  w = m%elements(e)%e%pts(3)%p%x - m%elements(e)%e%pts(7)%p%x
396  temp = u(1)*v(1) + u(2)*v(2) + u(3)*v(3)
397  if(.not. abscmp(temp, 0d0)) m%dfrmd_el(e) = .true.
398  temp = u(1)*w(1) + u(2)*w(2) + u(3)*w(3)
399  if(.not. abscmp(temp, 0d0)) m%dfrmd_el(e) = .true.
400  end if
401  end do
402  end subroutine mesh_generate_flags

411  subroutine mesh_generate_conn(m)
412  type(mesh_t), target, intent(inout) :: m
413  type(tuple_i4_t) :: edge
414  type(tuple4_i4_t) :: face
415  type(tuple_i4_t) :: facet_data
416 
417  integer :: i, j, k, ierr, el_glb_idx, n_sides, n_nodes
418 
419  if (m%lconn) return
420 
421  if (m%gdim .eq. 2) then
422  n_sides = 4
423  n_nodes = 2
424  else
425  n_sides = 6
426  n_nodes = 4
427  end if
428 
429  ! Compute global number of unique points
430  call mpi_allreduce(m%max_pts_id, m%glb_mpts, 1, &
431  mpi_integer, mpi_max, neko_comm, ierr)
432 
433  !
434  ! Find all (local) boundaries
435  !
436 
440  select type (fmp => m%facet_map)
441  type is(htable_i4t2_t)
442  do k = 1, 2
443  do i = 1, m%nelv
444  el_glb_idx = i + m%offset_el
445  do j = 1, n_sides
446  call m%elements(i)%e%facet_id(edge, j)
447 
448  ! Assume that all facets are on the exterior
449  facet_data%x = (/ 0, 0/)
450 
451  !check it this face has shown up earlier
452  if (fmp%get(edge, facet_data) .eq. 0) then
453  !if element is already recognized on face
454  if (facet_data%x(1) .eq. el_glb_idx ) then
455  m%facet_neigh(j, i) = facet_data%x(2)
456  else if( facet_data%x(2) .eq. el_glb_idx) then
457  m%facet_neigh(j, i) = facet_data%x(1)
458  !if this is the second element, arrange so low id is first
459  else if(facet_data%x(1) .gt. el_glb_idx) then
460  facet_data%x(2) = facet_data%x(1)
461  facet_data%x(1) = el_glb_idx
462  m%facet_neigh(j, i) = facet_data%x(2)
463  call fmp%set(edge, facet_data)
464  else if(facet_data%x(1) .lt. el_glb_idx) then
465  facet_data%x(2) = el_glb_idx
466  m%facet_neigh(j, i) = facet_data%x(1)
467  call fmp%set(edge, facet_data)
468  end if
469  else
470  facet_data%x(1) = el_glb_idx
471  m%facet_neigh(j, i) = facet_data%x(2)
472  call fmp%set(edge, facet_data)
473  end if
474  end do
475  end do
476  end do
477  type is(htable_i4t4_t)
478 
479  do k = 1, 2
480  do i = 1, m%nelv
481  el_glb_idx = i + m%offset_el
482  do j = 1, n_sides
483  call m%elements(i)%e%facet_id(face, j)
484 
485  facet_data%x = (/ 0, 0/)
486 
487  !check it this face has shown up earlier
488  if (fmp%get(face, facet_data) .eq. 0) then
489  !if element is already recognized on face
490  if (facet_data%x(1) .eq. el_glb_idx ) then
491  m%facet_neigh(j, i) = facet_data%x(2)
492  else if( facet_data%x(2) .eq. el_glb_idx) then
493  m%facet_neigh(j, i) = facet_data%x(1)
494  !if this is the second element, arrange so low id is first
495  else if(facet_data%x(1) .gt. el_glb_idx) then
496  facet_data%x(2) = facet_data%x(1)
497  facet_data%x(1) = el_glb_idx
498  m%facet_neigh(j, i) = facet_data%x(2)
499  call fmp%set(face, facet_data)
500  else if(facet_data%x(1) .lt. el_glb_idx) then
501  facet_data%x(2) = el_glb_idx
502  m%facet_neigh(j, i) = facet_data%x(1)
503  call fmp%set(face, facet_data)
504  end if
505  else
506  facet_data%x(1) = el_glb_idx
507  m%facet_neigh(j, i) = facet_data%x(2)
508  call fmp%set(face, facet_data)
509  end if
510  end do
511  end do
512  end do
513  class default
514  call neko_error('Invalid facet map')
515  end select
516 
517 
518  !
519  ! Find all external (between PEs) boundaries
520  !
521  if (pe_size .gt. 1) then
522  call mesh_generate_external_facet_conn(m)
523 
524  call mesh_generate_external_point_conn(m)
525  end if
526 
527  !
528  ! Find all internal/extenral edge connections
529  ! (Note it needs to be called after external point connections has
530  ! been established)
531  !
532  if (m%gdim .eq. 3) then
533  call mesh_generate_edge_conn(m)
534  end if
535 
536 
537  call mesh_generate_facet_numbering(m)
538 
539  m%lconn = .true.
540 
541  end subroutine mesh_generate_conn

544  subroutine mesh_generate_external_facet_conn(m)
545  type(mesh_t), intent(inout) :: m
546  type(tuple_i4_t) :: edge, edge2
547  type(tuple4_i4_t) :: face, face2
548  type(tuple_i4_t) :: facet_data
549  type(stack_i4_t) :: buffer
550  type(mpi_status) :: status
551  integer, allocatable :: recv_buffer(:)
552  integer :: i, j, k, el_glb_idx, n_sides, n_nodes, facet, element, l
553  integer :: max_recv, ierr, src, dst, n_recv, recv_side, neigh_el
554 
555 
556  if (m%gdim .eq. 2) then
557  n_sides = 4
558  n_nodes = 2
559  else
560  n_sides = 6
561  n_nodes = 4
562  end if
563 
564  call buffer%init()
565 
566  ! Build send buffers containing
567  ! [el_glb_idx, side number, facet_id (global ids of points)]
568  do i = 1, m%nelv
569  el_glb_idx = i + m%offset_el
570  do j = 1, n_sides
571  facet = j ! Adhere to standards...
572  if (m%facet_neigh(j, i) .eq. 0) then
573  if (n_nodes .eq. 2) then
574  call m%elements(i)%e%facet_id(edge, j)
575  call buffer%push(el_glb_idx)
576  call buffer%push(facet)
577  do k = 1, n_nodes
578  call buffer%push(edge%x(k))
579  end do
580  else
581  call m%elements(i)%e%facet_id(face, j)
582  call buffer%push(el_glb_idx)
583  call buffer%push(facet)
584  do k = 1, n_nodes
585  call buffer%push(face%x(k))
586  end do
587  end if
588  end if
589  end do
590  end do
591 
592 
593  call mpi_allreduce(buffer%size(), max_recv, 1, &
594  mpi_integer, mpi_max, neko_comm, ierr)
595 
596  allocate(recv_buffer(max_recv))
597 
598  do i = 1, pe_size - 1
599  src = modulo(pe_rank - i + pe_size, pe_size)
600  dst = modulo(pe_rank + i, pe_size)
601 
602  call mpi_sendrecv(buffer%array(), buffer%size(), mpi_integer, dst, 0, &
603  recv_buffer, max_recv, mpi_integer, src, 0, neko_comm, status, ierr)
604 
605  call mpi_get_count(status, mpi_integer, n_recv, ierr)
606 
607  select type (fmp => m%facet_map)
608  type is(htable_i4t2_t)
609  do j = 1, n_recv, n_nodes + 2
610  neigh_el = recv_buffer(j)
611  recv_side = recv_buffer(j+1)
612 
613  edge = (/ recv_buffer(j+2), recv_buffer(j+3) /)
614 
615  facet_data = (/ 0, 0 /)
616  !Check if the face is present on this PE
617  if (fmp%get(edge, facet_data) .eq. 0) then
618  element = facet_data%x(1) - m%offset_el
619  !Check which side is connected
620  do l = 1, n_sides
621  call m%elements(element)%e%facet_id(edge2, l)
622  if(edge2 .eq. edge) then
623  facet = l
624  exit
625  end if
626  end do
627  m%facet_neigh(facet, element) = -neigh_el
628  facet_data%x(2) = -neigh_el
629 
630  ! Update facet map
631  call fmp%set(edge, facet_data)
632 
633  call distdata_set_shared_el_facet(m%ddata, element, facet)
634 
635  if (m%hte%get(edge, facet) .eq. 0) then
636  call distdata_set_shared_facet(m%ddata, facet)
637  else
638  call neko_error("Invalid shared edge")
639  end if
640 
641  end if
642 
643  end do
644  type is(htable_i4t4_t)
645  do j = 1, n_recv, n_nodes + 2
646  neigh_el = recv_buffer(j)
647  recv_side = recv_buffer(j+1)
648 
649  face%x = (/ recv_buffer(j+2), recv_buffer(j+3), &
650  recv_buffer(j+4), recv_buffer(j+5) /)

858  !

864

913

915  else

917  end if
918  end do

920

925  mpi_integer, mpi_max, neko_comm, ierr)
926 
927  !
928  ! Update ghosted edges with new global id
929  !
930

932  mpi_integer, mpi_max, neko_comm, ierr)
933 
934  deallocate(recv_buff)
935  allocate(recv_buff(max_recv))
936 
937

940  dst = modulo(pe_rank + i, pe_size)

963  end do
