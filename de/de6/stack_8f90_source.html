147  subroutine stack_init(this, size)
148  class(stack_t), intent(inout) :: this
149  integer, optional :: size
150  integer :: size_t
151 
152  if (present(size)) then
153  if (size .gt. 0) then
154  size_t = size
155  else
156  call neko_warning('Invalid stack size, using default')
157  size_t = neko_stack_size_t
158  end if
159  else
160  size_t = neko_stack_size_t
161  end if
162 
163  this%size_ = ishft(1, ceiling(log(real(size_t, rp)) / neko_m_ln2))
164  this%top_ = 0
165  select type(this)
166  type is(stack_i4_t)
167  allocate(integer::this%data(this%size_))
168  type is(stack_i8_t)
169  allocate(integer(i8)::this%data(this%size_))
170  type is (stack_r8_t)
171  allocate(double precision::this%data(this%size_))
172  type is (stack_i4t2_t)
173  allocate(tuple_i4_t::this%data(this%size_))
174  type is (stack_i4t4_t)
175  allocate(tuple4_i4_t::this%data(this%size_))
176  type is (stack_i4r8t2_t)
177  allocate(tuple_i4r8_t::this%data(this%size_))
178  type is (stack_2i4r8t3_t)
179  allocate(tuple_2i4r8_t::this%data(this%size_))
180  type is (stack_curve_t)
181  allocate(struct_curve_t::this%data(this%size_))
182  type is (stack_nq_t)
183  allocate(nmsh_quad_t::this%data(this%size_))
184  type is (stack_nh_t)
185  allocate(nmsh_hex_t::this%data(this%size_))
186  type is (stack_nz_t)
187  allocate(nmsh_zone_t::this%data(this%size_))
188  type is (stack_nc_t)
189  allocate(nmsh_curve_el_t::this%data(this%size_))
190  class default
191  call neko_error('Invalid data type')
192  end select
193 
194  end subroutine stack_init
195

222  subroutine stack_push(this, data)
223  class(stack_t), target, intent(inout) :: this
224  class(*), intent(inout) :: data
225  class(*), allocatable :: tmp(:)
226  integer :: i
227 
228  if (this%top_ .eq. this%size_) then
229  this%size_ = ishft(this%size_, 1)
230  select type(data)
231  type is(integer)
232  allocate(integer::tmp(this%size_))
233  type is(integer(i8))
234  allocate(integer(i8)::tmp(this%size_))
235  type is(double precision)
236  allocate(double precision::tmp(this%size_))
237  type is(tuple_i4_t)
238  allocate(tuple_i4_t::tmp(this%size_))
239  type is(tuple4_i4_t)
240  allocate(tuple4_i4_t::tmp(this%size_))
241  type is(tuple_i4r8_t)
242  allocate(tuple_i4r8_t::tmp(this%size_))
243  type is(tuple_2i4r8_t)
244  allocate(tuple_2i4r8_t::tmp(this%size_))
245  type is(struct_curve_t)
246  allocate(struct_curve_t::tmp(this%size_))
247  type is (nmsh_quad_t)
248  allocate(nmsh_quad_t::tmp(this%size_))
249  type is (nmsh_hex_t)
250  allocate(nmsh_hex_t::tmp(this%size_))
251  type is (nmsh_zone_t)
252  allocate(nmsh_zone_t::tmp(this%size_))
253  type is (nmsh_curve_el_t)
254  allocate(nmsh_curve_el_t::tmp(this%size_))
255  class default
256  call neko_error('Invalid data type (stack_push)')
257  end select
258 
259  select type(tmp)
260  type is (integer)
261  select type(sdp=>this%data)
262  type is (integer)
263  tmp(1:this%top_) = sdp
264  end select
265  type is (integer(i8))
266  select type(sdp=>this%data)
267  type is (integer(i8))
268  tmp(1:this%top_) = sdp
269  end select
270  type is (double precision)
271  select type(sdp=>this%data)
272  type is (double precision)
273  tmp(1:this%top_) = sdp
274  end select
275  type is (tuple_i4_t)
276  select type(sdp=>this%data)
277  type is (tuple_i4_t)
278  do i = 1, this%top_
279  tmp(i) = sdp(i)
280  end do
281  end select
282  type is (tuple4_i4_t)
283  select type(sdp=>this%data)
284  type is (tuple4_i4_t)
285  do i = 1, this%top_
286  tmp(i) = sdp(i)
287  end do
288  end select
289  type is (tuple_i4r8_t)
290  select type(sdp=>this%data)
291  type is (tuple_i4r8_t)
292  do i = 1, this%top_
293  tmp(i) = sdp(i)
294  end do
295  end select
296  type is (tuple_2i4r8_t)
297  select type(sdp=>this%data)
298  type is (tuple_2i4r8_t)
299  do i = 1, this%top_
300  tmp(i) = sdp(i)
301  end do
302  end select
303  type is (struct_curve_t)
304  select type(sdp=>this%data)
305  type is (struct_curve_t)
306  tmp(1:this%top_) = sdp
307  end select
308  type is (nmsh_quad_t)
309  select type(sdp=>this%data)
310  type is(nmsh_quad_t)
311  tmp(1:this%top_) = sdp
312  end select
313  type is (nmsh_hex_t)
314  select type(sdp=>this%data)
315  type is(nmsh_hex_t)
316  tmp(1:this%top_) = sdp
317  end select
318  type is (nmsh_zone_t)
319  select type(sdp=>this%data)
320  type is(nmsh_zone_t)
321  tmp(1:this%top_) = sdp
322  end select
323  type is (nmsh_curve_el_t)
324  select type(sdp=>this%data)
325  type is(nmsh_curve_el_t)
326  tmp(1:this%top_) = sdp
327  end select
328  class default
329  call neko_error('Invalid data type (stack_push tmp)')
330  end select
331  call move_alloc(tmp, this%data)
332  end if
333 
334  this%top_ = this%top_ + 1
335 
336  select type(sdp=>this%data)
337  type is (integer)
338  select type(data)
339  type is (integer)
340  sdp(this%top_) = data
341  end select
342  type is (integer(i8))
343  select type(data)
344  type is (integer(i8))
345  sdp(this%top_) = data
346  end select
347  type is (double precision)
348  select type(data)
349  type is (double precision)
350  sdp(this%top_) = data
351  end select
352  type is (tuple_i4_t)
353  select type(data)
354  type is (tuple_i4_t)
355  sdp(this%top_) = data
356  end select
357  type is (tuple4_i4_t)
358  select type(data)
359  type is (tuple4_i4_t)
360  sdp(this%top_) = data
361  end select
362  type is (tuple_i4r8_t)
363  select type(data)
364  type is (tuple_i4r8_t)
365  sdp(this%top_) = data
366  end select
367  type is (tuple_2i4r8_t)
368  select type(data)
369  type is (tuple_2i4r8_t)
370  sdp(this%top_) = data
371  end select
372  type is (struct_curve_t)
373  select type(data)
374  type is (struct_curve_t)
375  sdp(this%top_) = data
376  end select
377  type is (nmsh_quad_t)
378  select type(data)
379  type is (nmsh_quad_t)
380  sdp(this%top_) = data
381  end select
382  type is (nmsh_hex_t)
383  select type(data)
384  type is (nmsh_hex_t)
385  sdp(this%top_) = data
386  end select
387  type is (nmsh_zone_t)
388  select type(data)
389  type is (nmsh_zone_t)
390  sdp(this%top_) = data
391  end select
392  type is (nmsh_curve_el_t)
393  select type(data)
394  type is (nmsh_curve_el_t)
395  sdp(this%top_) = data
396  end select
397  class default
398  call neko_error('Invalid data type in stack (stack_push)')
399  end select
400  end subroutine stack_push
