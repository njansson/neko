130  end do
131  end if

137  end do

141  do i = 1, nlogic
142  read(9, *)
143  end do

147  read(9, *)

150  re2_fname = trim(this%fname(1:scan(trim(this%fname), &

152  call re2_file%init(re2_fname)
153  call re2_file%read(msh)

156 1 format('gdim = ', i1, ', nelements =', i7)

190  end do
191  call mesh_add_element(msh, el_idx, p(1), p(2), p(3), p(4))

193  else if (ndim .eq. 3) then
194  read(9, *) (xc(j),j=1,4)
195  read(9, *) (yc(j),j=1,4)
196  read(9, *) (zc(j),j=1,4)
197  read(9, *) (xc(j),j=5,8)
198  read(9, *) (yc(j),j=5,8)
199  read(9, *) (zc(j),j=5,8)
200  if (i .ge. start_el .and. i .le. end_el) then
201  do j = 1, 8
202  p(j) = point_t(real(xc(j),dp), real(yc(j),dp), real(zc(j),dp))
203  call rea_file_add_point(htp, p(j), pt_idx)
204  end do
205  call mesh_add_element(msh, el_idx, &
206  p(1), p(2), p(3), p(4), p(5), p(6), p(7), p(8))
207  end if
208  end if
209  if (i .ge. start_el .and. i .le. end_el) then
210  el_idx = el_idx + 1
211  end if
212  end do
213 
214  call htp%free()
215 
216  read(9, *)
217  read(9, *) ncurve
218  allocate(curve_data(5,8,nelgv))

222  curve_element(i) = .false.

224  curve_type(j,i) = 0
225  do l = 1, 5
226  curve_data(l,j,i) = 0d0
227  end do
228  end do
229  end do
230  do i = 1, ncurve

233  curve_data(j,edge,el_idx) = curve(j)
234  end do
235  curve_element(el_idx) = .true.
236  select case(trim(chtemp))

240  case ('e')
241  curve_type(edge,el_idx) = 2
242  curve_skip = .true.
243  case ('C')

245  end select

247  if (curve_skip) then

249  else

253  curve_data(1,1,el_idx), curve_type(1,el_idx))

255  end do
256  end if

258  deallocate(curve_element)

261  ! Read fluid boundary conditions
262  read(9,*)

270  do i = 1, nelgv
271  if (i .ge. start_el .and. i .le. end_el) then
272  el_idx = i - start_el + 1
273  do j = 1, 2*ndim
274  read(9, *) cbc(j, i), (bc_data(l,j,i),l=1,6)
275  sym_facet = facet_map(j)
276  select case(trim(cbc(j,i)))
277  case ('W')
278  call mesh_mark_wall_facet(msh, sym_facet, el_idx)
279  case ('v', 'V')
280  call mesh_mark_inlet_facet(msh, sym_facet, el_idx)
281  case ('O', 'o')
282  call mesh_mark_outlet_facet(msh, sym_facet, el_idx)
283  case ('SYM')
284  call mesh_mark_sympln_facet(msh, sym_facet, el_idx)
285  case ('P')
286  p_el_idx = int(bc_data(2+off,j,i))
287  p_facet = facet_map(int(bc_data(3+off,j,i)))
288  call mesh_get_facet_ids(msh, sym_facet, el_idx, pids)
289  call mesh_mark_periodic_facet(msh, sym_facet, el_idx, &
290  p_facet, p_el_idx, pids)
291  end select
292  end do
293  end if
294  end do
295  do i = 1, nelgv
296  if (i .ge. start_el .and. i .le. end_el) then
297  el_idx = i - start_el + 1
298  do j = 1, 2*ndim
299  sym_facet = facet_map(j)
300  select case(trim(cbc(j,i)))
301  case ('P')
302  p_el_idx = int(bc_data(2+off,j,i))
303  p_facet = facet_map(int(bc_data(3+off,j,i)))
304  call mesh_create_periodic_ids(msh, sym_facet, el_idx, &
305  p_facet, p_el_idx)
306  end select
307  end do
308  end if
309  end do
310  do i = 1, nelgv
311  if (i .ge. start_el .and. i .le. end_el) then
312  el_idx = i - start_el + 1
313  do j = 1, 2*ndim
314  sym_facet = facet_map(j)
315  select case(trim(cbc(j,i)))
316  case ('P')
317  p_el_idx = int(bc_data(2+off,j,i))
318  p_facet = facet_map(int(bc_data(3+off,j,i)))
319  call mesh_create_periodic_ids(msh, sym_facet, el_idx, &
320  p_facet, p_el_idx)
321  end select
322  end do
323  end if
324  end do
325  do i = 1, nelgv
326  if (i .ge. start_el .and. i .le. end_el) then
327  el_idx = i - start_el + 1
328  do j = 1, 2*ndim
329  sym_facet = facet_map(j)
330  select case(trim(cbc(j,i)))
331  case ('P')
332  p_el_idx = int(bc_data(2+off,j,i))
333  p_facet = facet_map(int(bc_data(3+off,j,i)))
334  call mesh_create_periodic_ids(msh, sym_facet, el_idx, &
335  p_facet, p_el_idx)
336  end select
337  end do
338  end if
339  end do
340  deallocate(cbc)
341  deallocate(bc_data)
342  else ! Store bcs in a NEKTON session structure
343  allocate(cbc(6,nelgv))
344  do i = 1, nelgv
345  do j = 1, 2*ndim
346  read(9,'(a1, a3)') chtemp, cbc(j, i)
347  end do
348  end do
349  end if
350 
351  call mesh_finalize(msh)
352 
353  call neko_log%message('Done')
