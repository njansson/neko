77  subroutine fd_weights_full(xx, x, n, m, c)
78  integer, intent(in) :: n
79  integer, intent(in) :: m
80  real(kind=rp), intent(in) :: x(0:n)
81  real(kind=rp), intent(out) :: c(0:n,0:m)
82  real(kind=rp), intent(in) :: xx
83  real(kind=rp) :: c1, c2, c3, c4, c5
84  integer :: i, j, k, mn
85 
86  c1 = 1d0
87  c4 = x(0) - xx
88 
89  do k = 0, m
90  do j = 0, n
91  c(j,k) = 0d0
92  end do
93  end do
94 
95  c(0,0) = 1d0
96 
97  do i = 1, n
98  mn = min(i,m)
99  c2 = 1d0
100  c5 = c4
101  c4 = x(i) - xx
102  do j = 0, i - 1
103  c3 = x(i) - x(j)
104  c2 = c2 * c3
105  do k = mn, 1, -1
106  c(i,k) = c1 * (k * c(i-1,k-1) - c5 * c(i-1,k)) / c2
107  end do
108  c(i,0) = -c1 * c5 * c(i-1,0) / c2
109  do k = mn, 1, -1
110  c(j,k) = (c4 * c(j,k) - k * c(j,k-1)) / c3
111  end do
112  c(j,0) = c4 * c(j,0) / c3
113  end do
114  c1 = c2
115  end do
116 
117  end subroutine fd_weights_full

140  subroutine semhat(a, b, c, d, z, dgll, jgll, bgl, zgl, dgl, jgl, n, w)
141  integer, intent(in) :: n
142  real(kind=rp), intent(inout) :: a(0:n,0:n)
143  real(kind=rp), intent(inout) :: b(0:n)
144  real(kind=rp), intent(inout) :: c(0:n,0:n)
145  real(kind=rp), intent(inout) :: d(0:n,0:n)
146  real(kind=rp), intent(inout) :: z(0:n)
147  real(kind=rp), intent(inout) :: dgll(0:n,1:n-1),jgll(0:n,1:n-1)
148  real(kind=rp), intent(inout) :: bgl(1:n-1)
149  real(kind=rp), intent(inout) :: zgl(1:n-1)
150  real(kind=rp), intent(inout) :: dgl(1:n-1,0:n)
151  real(kind=rp), intent(inout) :: jgl(1:n-1,0:n)
152  real(kind=rp), intent(inout) :: w(0:2*n+1)
153  integer :: np, nm, n2, i, j, k
154  np = n+1
155  nm = n-1
156  n2 = n-2
157  call zwgll(z, b, np)
158  do i = 0,n
159  call fd_weights_full(z(i), z, n, 1, w)
160  do j = 0,n
161  d(i,j) = w(j+np) ! Derivative matrix
162  end do
163  end do
164 
165  if (n.eq.1) return ! No interpolation for n=1
166 
167  do i = 0,n
168  call fd_weights_full(z(i), z(1), n2, 1, w(1))
169  do j = 1,nm
170  jgll(i,j) = w(j ) ! Interpolation matrix
171  dgll(i,j) = w(j+nm) ! Derivative matrix
172  end do
173  end do
174  call rzero(a, np*np)
175  do j = 0,n
176  do i = 0,n
177  do k = 0,n
178  a(i,j) = a(i,j) + d(k,i)*b(k)*d(k,j)
179  end do
180  c(i,j) = b(i)*d(i,j)
181  end do
182  end do
183  call zwgl(zgl, bgl, nm)
184  do i = 1,n-1
185  call fd_weights_full(zgl(i), z, n, 1, w)
186  do j = 0,n
187  jgl(i,j) = w(j ) ! Interpolation matrix
188  dgl(i,j) = w(j+np) ! Derivative matrix
189  end do
190  end do
191  end subroutine semhat
