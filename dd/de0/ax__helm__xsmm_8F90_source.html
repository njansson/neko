77  subroutine ax_helm_xsmm_compute(w, u, coef, msh, Xh)
78  type(mesh_t), intent(inout) :: msh
79  type(space_t), intent(inout) :: Xh
80  type(coef_t), intent(inout) :: coef
81  real(kind=rp), intent(inout) :: w(xh%lx, xh%ly, xh%lz, msh%nelv)
82  real(kind=rp), intent(inout) :: u(xh%lx, xh%ly, xh%lz, msh%nelv)
83 
84  real(kind=rp) :: dudr(xh%lx,xh%ly,xh%lz)
85  real(kind=rp) :: duds(xh%lx,xh%ly,xh%lz)
86  real(kind=rp) :: dudt(xh%lx,xh%ly,xh%lz)
87  real(kind=rp) :: tmp1(xh%lx,xh%ly,xh%lz)
88  real(kind=rp) :: tmp2(xh%lx,xh%ly,xh%lz)
89  real(kind=rp) :: tmp3(xh%lx,xh%ly,xh%lz)
90  real(kind=rp) :: tm1(xh%lx,xh%ly,xh%lz)
91  real(kind=rp) :: tm2(xh%lx,xh%ly,xh%lz)
92  real(kind=rp) :: tm3(xh%lx,xh%ly,xh%lz)
93  integer :: e, k, lxy, lxz, lyz, lxyz
94 #ifdef HAVE_LIBXSMM
95  type(libxsmm_dmmfunction), save :: ax_helm_xmm1
96  type(libxsmm_dmmfunction), save :: ax_helm_xmm2
97  type(libxsmm_dmmfunction), save :: ax_helm_xmm3
98  integer, save :: ax_helm_xsmm_lx = 0
99  logical, save :: ax_helm_xsmm_init = .false.
100 
101  lxy = xh%lx*xh%ly
102  lxz = xh%lx*xh%lz
103  lyz = xh%ly*xh%lz
104  lxyz = xh%lx*xh%ly*xh%lz
105 
106  if (.not. ax_helm_xsmm_init .or. (ax_helm_xsmm_lx .ne. xh%lx)) then
107  call libxsmm_dispatch(ax_helm_xmm1, xh%lx, xh%ly*xh%lz, xh%lx, &
108  alpha=1d0, beta=0d0, prefetch=libxsmm_prefetch_auto)
109  call libxsmm_dispatch(ax_helm_xmm2, xh%lx, xh%ly, xh%ly, &
110  alpha=1d0, beta=0d0, prefetch=libxsmm_prefetch_auto)
111  call libxsmm_dispatch(ax_helm_xmm3, xh%lx*xh%ly, xh%lz, xh%lz, &
112  alpha=1d0, beta=0d0, prefetch=libxsmm_prefetch_auto)
113  ax_helm_xsmm_init = .true.
114  ax_helm_xsmm_lx = xh%lx
115  end if
116 
117 
118  do e = 1, msh%nelv
119  if(msh%gdim .eq. 2) then
120  call mxm(xh%dx, xh%lx,u(1,1,1,e), xh%lx, dudr, lyz)
121  call mxm(u(1,1,1,e), xh%lx, xh%dyt, xh%ly, duds, xh%ly)
122  call col3(tmp1, dudr, coef%G11(1,1,1,e), lxyz)
123  call col3(tmp2, duds, coef%G22(1,1,1,e), lxyz)
124  if (msh%dfrmd_el(e)) then
125  call addcol3(tmp1, duds, coef%G12(1,1,1,e), lxyz)
126  call addcol3(tmp2, dudr, coef%G12(1,1,1,e), lxyz)
127  end if
128  call col2(tmp1, coef%h1(1,1,1,e), lxyz)
129  call col2(tmp2, coef%h1(1,1,1,e), lxyz)
130  call mxm(xh%dxt, xh%lx, tmp1, xh%lx, tm1, lyz)
131  call mxm(tmp2, xh%lx, xh%dy, xh%ly, tm2, xh%ly)
132  call add3(w(1,1,1,e), tm1, tm2, lxyz)
133 
134  ! 3D evaluation!
135  else
136  call libxsmm_mmcall(ax_helm_xmm1, xh%dx, u(1,1,1,e), dudr)
137  do k = 1,xh%lz
138  call libxsmm_mmcall(ax_helm_xmm2, u(1,1,k,e), xh%dyt, duds(1,1,k))
139  end do
140  call libxsmm_mmcall(ax_helm_xmm3, u(1,1,1,e), xh%dzt, dudt)
141  call col3(tmp1, dudr, coef%G11(1,1,1,e), lxyz)
142  call col3(tmp2, duds, coef%G22(1,1,1,e), lxyz)
143  call col3(tmp3, dudt, coef%G33(1,1,1,e), lxyz)
144  if (msh%dfrmd_el(e)) then
145  call addcol3(tmp1, duds, coef%G12(1,1,1,e), lxyz)
146  call addcol3(tmp1, dudt, coef%G13(1,1,1,e), lxyz)
147  call addcol3(tmp2, dudr, coef%G12(1,1,1,e), lxyz)
148  call addcol3(tmp2, dudt, coef%G23(1,1,1,e), lxyz)
149  call addcol3(tmp3, dudr, coef%G13(1,1,1,e), lxyz)
150  call addcol3(tmp3, duds, coef%G23(1,1,1,e), lxyz)
151  end if
152  call col2(tmp1, coef%h1(1,1,1,e), lxyz)
153  call col2(tmp2, coef%h1(1,1,1,e), lxyz)
154  call col2(tmp3, coef%h1(1,1,1,e), lxyz)
155  call libxsmm_mmcall(ax_helm_xmm1, xh%dxt, tmp1, tm1)
156  do k = 1,xh%lz
157  call libxsmm_mmcall(ax_helm_xmm2, tmp2(1,1,k), xh%dy, tm2(1,1,k))
158  end do
159  call libxsmm_mmcall(ax_helm_xmm3, tmp3, xh%dz, tm3)
160  call add4(w(1,1,1,e), tm1, tm2, tm3, lxyz)
161  end if
162  end do
163 
164  if (coef%ifh2) call addcol4 (w,coef%h2,coef%B,u,coef%dof%n_dofs)
165 #endif
166 
167  end subroutine ax_helm_xsmm_compute
