104 
105  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &

118  (neko_bcknd_opencl .eq. 1)) then

152  subroutine set_flow_ic_blasius(u, v, w, delta, uinf, type)
153  type(field_t), intent(inout) :: u
154  type(field_t), intent(inout) :: v
155  type(field_t), intent(inout) :: w
156  real(kind=rp), intent(in) :: delta
157  real(kind=rp), intent(in) :: uinf(3)
158  character(len=*), intent(in) :: type
159  procedure(blasius_profile), pointer :: bla => null()
160  integer :: i
161 
162  select case(trim(type))
163  case('linear')
164  bla => blasius_linear
165  case('quadratic')
166  bla => blasius_quadratic
167  case('cubic')
168  bla => blasius_cubic
169  case('quartic')
170  bla => blasius_quartic
171  case('sin')
172  bla => blasius_sin
173  case default
174  call neko_error('Invalid Blasius approximation')
175  end select
176 
177  if ((uinf(1) .gt. 0.0_rp) .and. (uinf(2) .eq. 0.0_rp) &
178  .and. (uinf(3) .eq. 0.0_rp)) then
179  do i = 1, u%dof%size()
180  u%x(i,1,1,1) = bla(u%dof%z(i,1,1,1), delta, uinf(1))
181  v%x(i,1,1,1) = 0.0_rp
182  w%x(i,1,1,1) = 0.0_rp
183  end do
184  else if ((uinf(1) .eq. 0.0_rp) .and. (uinf(2) .gt. 0.0_rp) &
185  .and. (uinf(3) .eq. 0.0_rp)) then
186  do i = 1, u%dof%size()
187  u%x(i,1,1,1) = 0.0_rp
188  v%x(i,1,1,1) = bla(u%dof%x(i,1,1,1), delta, uinf(2))
189  w%x(i,1,1,1) = 0.0_rp
190  end do
191  else if ((uinf(1) .eq. 0.0_rp) .and. (uinf(2) .eq. 0.0_rp) &
192  .and. (uinf(3) .gt. 0.0_rp)) then
193  do i = 1, u%dof%size()
194  u%x(i,1,1,1) = 0.0_rp
195  v%x(i,1,1,1) = 0.0_rp
196  w%x(i,1,1,1) = bla(u%dof%y(i,1,1,1), delta, uinf(3))
197  end do
198  end if
199 
200  end subroutine set_flow_ic_blasius
