36 __global__ void makeabf_kernel(T * __restrict__ ta1,
37  T * __restrict__ ta2,
38  T * __restrict__ ta3,
39  T * __restrict__ abx1,
40  T * __restrict__ aby1,
41  T * __restrict__ abz1,
42  T * __restrict__ abx2,
43  T * __restrict__ aby2,
44  T * __restrict__ abz2,
45  T * __restrict__ bfx,
46  T * __restrict__ bfy,
47  T * __restrict__ bfz,
48  const T rho,
49  const T ab1,
50  const T ab2,
51  const T ab3,
52  const int n) {
53 
54  const int idx = blockIdx.x * blockDim.x + threadIdx.x;
55  const int str = blockDim.x * gridDim.x;
56 
57  for (int i = idx; i < n; i += str) {
58  ta1[i] = ab2 * abx1[i] + ab3 * abx2[i];
59  ta2[i] = ab2 * aby1[i] + ab3 * aby2[i];
60  ta3[i] = ab2 * abz1[i] + ab3 * abz2[i];
61  }
62 
63  for (int i = idx; i < n; i += str) {
64  abx2[i] = abx1[i];
65  aby2[i] = aby1[i];
66  abz2[i] = abz1[i];
67  abx1[i] = bfx[i];
68  aby1[i] = bfy[i];
69  abz1[i] = bfz[i];
70  }
71 
72  for (int i = idx; i < n; i += str) {
73  bfx[i] = (ab1 * bfx[i] + ta1[i]) * rho;
74  bfy[i] = (ab1 * bfy[i] + ta2[i]) * rho;
75  bfz[i] = (ab1 * bfz[i] + ta3[i]) * rho;
76  }
77 
78 }
