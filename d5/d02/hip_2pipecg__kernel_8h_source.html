39 __global__ void cg_update_xp_kernel(T * __restrict__ x,
40  T * __restrict__ p,
41  T ** __restrict__ u,
42  const T * alpha,
43  const T * beta,
44  const int p_cur,
45  const int p_space,
46  const int n) {
47 
48  const int idx = blockIdx.x * blockDim.x + threadIdx.x;
49  const int str = blockDim.x * gridDim.x;
50 
51 
52  for (int i = idx; i < n; i+= str) {
53  T tmp = 0.0;
54  int p_prev = p_space;
55  for (int j = 0; j < p_cur; j ++) {
56  p[i] = beta[j]*p[i] + u[p_prev][i];
57  tmp += alpha[j]*p[i];
58  p_prev = j;
59  }
60  x[i] += tmp;
61  u[p_space][i] = u[p_space-1][i];
62  }
63 }

89 
90  __shared__ T buf1[1024];

92  __shared__ T buf3[1024];
93  T tmp1 = 0.0;
94  T tmp2 = 0.0;
95  T tmp3 = 0.0;
96 
97  for (int i = idx; i < n; i+= str) {
98  z[i] = beta * z[i] + ni[i];
99  q[i] = beta * q[i] + mi[i];
100  s[i] = beta * s[i] + w[i];
101  r[i] = r[i] - alpha * s[i];
102  u2[i] = u1[i] - alpha * q[i];
103  w[i] = w[i] - alpha * z[i];
104  tmp1 = tmp1 + r[i] * mult[i] * u2[i];
105  tmp2 = tmp2 + w[i] * mult[i] * u2[i];
106  tmp3 = tmp3 + r[i] * mult[i] * r[i];
107 
108  }
109  buf1[threadIdx.x] = tmp1;
110  buf2[threadIdx.x] = tmp2;
111  buf3[threadIdx.x] = tmp3;
112  __syncthreads();
113 
114  int i = blockDim.x>>1;
115  while (i != 0) {
116  if (threadIdx.x < i) {
117  buf1[threadIdx.x] += buf1[threadIdx.x + i];
118  buf2[threadIdx.x] += buf2[threadIdx.x + i];
119  buf3[threadIdx.x] += buf3[threadIdx.x + i];
120  }
