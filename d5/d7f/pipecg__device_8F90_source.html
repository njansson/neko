291  function pipecg_device_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
292  class(pipecg_device_t), intent(inout) :: this
293  class(ax_t), intent(inout) :: ax
294  type(field_t), intent(inout) :: x
295  integer, intent(inout) :: n
296  real(kind=rp), dimension(n), intent(inout) :: f
297  type(coef_t), intent(inout) :: coef
298  type(bc_list_t), intent(inout) :: blst
299  type(gs_t), intent(inout) :: gs_h
300  type(ksp_monitor_t) :: ksp_results
301  integer, optional, intent(in) :: niter
302  integer :: iter, max_iter, ierr, p_cur, p_prev, u_prev
303  real(kind=rp) :: rnorm, rtr, reduction(3), norm_fac
304  real(kind=rp) :: alpha(this%p_space), beta(this%p_space), gamma1, gamma2, delta
305  real(kind=rp) :: tmp1, tmp2, tmp3
306  type(mpi_request) :: request
307  type(mpi_status) :: status
308  type(c_ptr) :: f_d, alpha_d, beta_d
309  f_d = device_get_ptr(f)
310  alpha_d = c_null_ptr
311  call device_map(alpha, alpha_d, this%p_space)
312  beta_d = c_null_ptr
313  call device_map(beta, beta_d, this%p_space)
314 
315  if (present(niter)) then
316  max_iter = niter
317  else
318  max_iter = ksp_max_iter
319  end if
320  norm_fac = 1.0_rp / sqrt(coef%volume)
321 
322  associate(p => this%p, q => this%q, r => this%r, s => this%s, &
323  u => this%u, w => this%w, z => this%z, mi => this%mi, ni => this%ni, &
324  p_d => this%p_d, q_d => this%q_d, r_d => this%r_d, &
325  s_d => this%s_d, u_d => this%u_d, u_d_d => this%u_d_d, &
326  w_d => this%w_d, z_d => this%z_d, mi_d => this%mi_d, ni_d => this%ni_d)
327 
328  p_prev = this%p_space
329  u_prev = this%p_space+1
330  p_cur = 1
331  call device_rzero(x%x_d, n)
332  call device_rzero(z_d, n)
333  call device_rzero(q_d, n)
334  call device_rzero(p_d, n)
335  call device_rzero(s_d, n)
336  call device_copy(r_d, f_d, n)
337  !apply u=M^-1r
338  !call device_copy(u_d(u_prev), r_d, n)
339  call this%M%solve(u(1,u_prev), r, n)
340  call ax%compute(w, u(1,u_prev), coef, x%msh, x%Xh)
341  call gs_op(gs_h, w, n, gs_op_add)
342  call bc_list_apply(blst, w, n)
343 
344  rtr = device_glsc3(r_d, coef%mult_d, r_d, n)
345  rnorm = sqrt(rtr)*norm_fac
346  ksp_results%res_start = rnorm
347  ksp_results%res_final = rnorm
348  ksp_results%iter = 0
349  if(rnorm .eq. 0.0_rp) return
350 
351  gamma1 = 0.0_rp
352  tmp1 = 0.0_rp
353  tmp2 = 0.0_rp
354  tmp3 = 0.0_rp
355  tmp1 = device_glsc3(r_d,coef%mult_d,u_d(u_prev),n)
356  tmp2 = device_glsc3(w_d,coef%mult_d,u_d(u_prev),n)
357  tmp3 = device_glsc3(r_d,coef%mult_d,r_d,n)
358  reduction(1) = tmp1
359  reduction(2) = tmp2
360  reduction(3) = tmp3
361 
362  do iter = 1, max_iter
363  call mpi_iallreduce(mpi_in_place, reduction, 3, &
364  mpi_real_precision, mpi_sum, neko_comm, request, ierr)
365 
366  call this%M%solve(mi, w, n)
367  call ax%compute(ni, mi, coef, x%msh, x%Xh)
368  call gs_op(gs_h, ni, n, gs_op_add)
369  call bc_list_apply(blst, ni, n)
370 
371  call mpi_wait(request, status, ierr)
372  gamma2 = gamma1
373  gamma1 = reduction(1)
374  delta = reduction(2)
375  rtr = reduction(3)
376 
377  rnorm = sqrt(rtr)*norm_fac
378  if (rnorm .lt. this%abs_tol) exit
379 
380 
381  if (iter .gt. 1) then
382  beta(p_cur) = gamma1 / gamma2
383  alpha(p_cur) = gamma1 / (delta - (beta(p_cur) * gamma1/alpha(p_prev)))
384  else
385  beta(p_cur) = 0.0_rp
386  alpha(p_cur) = gamma1/delta
387  end if
388 
389  !call device_add2s1(z_d, ni_d, beta(p_cur), n)
390  !call device_add2s1(q_d, mi_d, beta(p_cur), n)
391  !call device_add2s1(s_d, w_d, beta(p_cur), n)
392  !call device_add2s1(p_d, u_d(1), beta(p_cur), n)
393  !call device_add2s2(r_d, s_d, -alpha(p_cur), n)
394  !call device_add2s2(u_d(1), q_d, -alpha(p_cur), n)
395  !call device_add2s2(w_d, z_d, -alpha(p_cur), n)
396  !tmp1 = device_glsc3(r_d,coef%mult_d,u_d(1),n)
397  !tmp2 = device_glsc3(w_d,coef%mult_d,u_d(1),n)
398  !tmp3 = device_glsc3(r_d,coef%mult_d,r_d,n)
399  call device_pipecg_vecops(p_d, q_d, r_d,&
400  s_d, u_d(u_prev), u_d(p_cur),&
401  w_d, z_d, ni_d,&
402  mi_d, alpha(p_cur), beta(p_cur),&
403  coef%mult_d, reduction,n)
404  if (p_cur .eq. this%p_space) then
405  call device_memcpy(alpha, alpha_d, p_cur, host_to_device)
406  call device_memcpy(beta, beta_d, p_cur, host_to_device)
407  call device_cg_update_xp(x%x_d, p_d, u_d_d, alpha_d, beta_d, p_cur, this%p_space, n)
408  p_prev = p_cur
409  u_prev = this%p_space+1
410  alpha(1) = alpha(p_cur)
411  beta(1) = beta(p_cur)
412  p_cur = 1
413  else
414  u_prev = p_cur
415  p_prev = p_cur
416  p_cur = p_cur + 1
417  end if
418  end do
419  if ( p_cur .ne. 1) then
420  call device_memcpy(alpha, alpha_d, p_cur, host_to_device)
421  call device_memcpy(beta, beta_d, p_cur, host_to_device)
422  call device_cg_update_xp(x%x_d, p_d, u_d_d, alpha_d, beta_d, p_cur, this%p_space, n)
423  end if
424 
425  ksp_results%res_final = rnorm
426  ksp_results%iter = iter
427 
428  end associate
429 
430  end function pipecg_device_solve
