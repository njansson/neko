121  function cg_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
122  class(cg_t), intent(inout) :: this
123  class(ax_t), intent(inout) :: ax
124  type(field_t), intent(inout) :: x
125  integer, intent(inout) :: n
126  real(kind=rp), dimension(n), intent(inout) :: f
127  type(coef_t), intent(inout) :: coef
128  type(bc_list_t), intent(inout) :: blst
129  type(gs_t), intent(inout) :: gs_h
130  type(ksp_monitor_t) :: ksp_results
131  integer, optional, intent(in) :: niter
132  integer :: iter, max_iter, i, j, k, p_cur, p_prev
133  real(kind=rp) :: rnorm, rtr, rtz2, rtz1, x_plus(neko_blk_size)
134  real(kind=rp) :: beta, pap, norm_fac
135 
136  if (present(niter)) then
137  max_iter = niter
138  else
139  max_iter = ksp_max_iter
140  end if
141  norm_fac = 1.0_rp / sqrt(coef%volume)
142 
143  associate(w => this%w, r => this%r, p => this%p, &
144  z => this%z, alpha => this%alpha)
145 
146  rtz1 = 1.0_rp
147  call rzero(x%x, n)
148  call rzero(p(1,cg_p_space), n)
149  call copy(r, f, n)
150 
151  rtr = glsc3(r, coef%mult, r, n)
152  rnorm = sqrt(rtr) * norm_fac
153  ksp_results%res_start = rnorm
154  ksp_results%res_final = rnorm
155  ksp_results%iter = 0
156  p_prev = cg_p_space
157  p_cur = 1
158  if(rnorm .eq. 0.0_rp) return
159  do iter = 1, max_iter
160  call this%M%solve(z, r, n)
161  rtz2 = rtz1
162  rtz1 = glsc3(r, coef%mult, z, n)
163 
164  beta = rtz1 / rtz2
165  if (iter .eq. 1) beta = 0.0_rp
166  do i = 1, n
167  p(i,p_cur) = z(i) + beta * p(i,p_prev)
168  end do
169 
170  call ax%compute(w, p(1,p_cur), coef, x%msh, x%Xh)
171  call gs_op(gs_h, w, n, gs_op_add)
172  call bc_list_apply(blst, w, n)
173 
174  pap = glsc3(w, coef%mult, p(1,p_cur), n)
175 
176  alpha(p_cur) = rtz1 / pap
177  call second_cg_part(rtr, r, coef%mult, w, alpha(p_cur), n)
178  rnorm = sqrt(rtr) * norm_fac
179 
180  if ((p_cur .eq. cg_p_space) .or. &
181  (rnorm .lt. this%abs_tol) .or. iter .eq. max_iter) then
182  do i = 0, n, neko_blk_size
183  if (i + neko_blk_size .le. n) then
184  do k = 1, neko_blk_size
185  x_plus(k) = 0.0_rp
186  end do
187  do j = 1, p_cur
188  do k = 1, neko_blk_size
189  x_plus(k) = x_plus(k) + alpha(j) * p(i+k,j)
190  end do
191  end do
192  do k = 1, neko_blk_size
193  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + x_plus(k)
194  end do
195  else
196  do k = 1, n-i
197  x_plus(1) = 0.0_rp
198  do j = 1, p_cur
199  x_plus(1) = x_plus(1) + alpha(j) * p(i+k,j)
200  end do
201  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + x_plus(1)
202  end do
203  end if
204  end do
205  p_prev = p_cur
206  p_cur = 1
207  if (rnorm .lt. this%abs_tol) exit
208  else
209  p_prev = p_cur
210  p_cur = p_cur + 1
211  end if
212  end do
213  end associate
214 
215  ksp_results%res_final = rnorm
216  ksp_results%iter = iter
217 
218  end function cg_solve

220  subroutine second_cg_part(rtr, r, mult, w, alpha, n)
221  integer, intent(in) :: n
222  real(kind=rp), intent(inout) :: r(n), rtr
223  real(kind=rp), intent(in) ::mult(n), w(n), alpha
224  integer :: i, ierr
225 
226  rtr = 0.0_rp
227  do i = 1, n
228  r(i) = r(i) - alpha*w(i)
229  rtr = rtr + r(i) * r(i) * mult(i)
230  end do
231  call mpi_allreduce(mpi_in_place, rtr, 1, &
232  mpi_real_precision, mpi_sum, neko_comm, ierr)
233 
234  end subroutine second_cg_part
