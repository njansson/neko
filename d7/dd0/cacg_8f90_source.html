118  function cacg_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
119  class(cacg_t), intent(inout) :: this
120  class(ax_t), intent(inout) :: ax
121  type(field_t), intent(inout) :: x
122  integer, intent(inout) :: n
123  real(kind=rp), dimension(n), intent(inout) :: f
124  type(coef_t), intent(inout) :: coef
125  type(bc_list_t), intent(inout) :: blst
126  type(gs_t), intent(inout) :: gs_h
127  type(ksp_monitor_t) :: ksp_results
128  integer, optional, intent(in) :: niter
129  integer :: i, j, k, l, iter, max_iter, s, ierr, it
130  real(kind=rp) :: rnorm, rtr, rtz1, tmp
131  real(kind=rp) :: beta(this%s+1), alpha(this%s+1), alpha1, alpha2, norm_fac
132  real(kind=rp), dimension(4*this%s+1,4*this%s+1) :: tt, g, gtt, temp, temp2
133  real(kind=rp) :: p_c(4*this%s+1,this%s+1)
134  real(kind=rp) :: r_c(4*this%s+1,this%s+1)
135  real(kind=rp) :: z_c(4*this%s+1,this%s+1)
136  real(kind=rp) :: x_c(4*this%s+1,this%s+1)
137 
138  associate(pr => this%PR, r => this%r, p => this%p)
139  s = this%s
140  if (present(niter)) then
141  max_iter = niter
142  else
143  max_iter = ksp_max_iter
144  end if
145  norm_fac = 1.0_rp / sqrt(coef%volume)
146 
147  rtz1 = 1.0_rp
148  call rzero(x%x, n)
149  call copy(r, f, n)
150  call this%M%solve(p, r, n)
151 
152  rtr = glsc3(r, coef%mult, r, n)
153  rnorm = sqrt(rtr)*norm_fac
154  ksp_results%res_start = rnorm
155  ksp_results%res_final = rnorm
156  ksp_results%iter = 0
157  iter = 0
158  if(rnorm .eq. 0.0_rp) return
159  do while (iter < max_iter)
160 
161  call copy(pr,p, n)
162  call copy(pr(1,2*s+2), r, n)
163 
164  !Here we have hardcoded a monomial basis atm.
165  do i = 2, 2*s + 1
166  if (mod(i,2) .eq. 0) then
167  call ax%compute(pr(1,i), pr(1,i-1), coef, x%msh, x%Xh)
168  call gs_op_vector(gs_h, pr(1,i), n, gs_op_add)
169  call bc_list_apply_scalar(blst, pr(1,i), n)
170  else
171  call this%M%solve(pr(1,i), pr(1,i-1), n)
172  end if
173  end do
174 
175  do i = 2*s+2, 4*s
176  if (mod(i,2) == 0) then
177  call this%M%solve(pr(1,i+1), pr(1,i), n)
178  else
179  call ax%compute(pr(1,i+1), pr(1,i), coef, x%msh, x%Xh)
180  call gs_op_vector(gs_h, pr(1,i+1), n, gs_op_add)
181  call bc_list_apply_scalar(blst, pr(1,1+i), n)
182  end if
183  end do
184 
185  call construct_basis_matrix(tt, s)
186  call rzero(p_c, (4*s+1) * (s+1))
187  p_c(1,1) = 1.0_rp
188  call rzero(r_c, (4*s+1) * (s+1))
189  r_c(2*s+2,1) = 1.0_rp
190  call mxm(tt, 4*s+1, r_c, 4*s+1, z_c,s+1)
191  call rzero(x_c, (4*s+1) * (s+1))
192  call rzero(temp, (4*s+1)**2)
193 
194  do i = 0, n, neko_blk_size
195  it = 0
196  if (i + neko_blk_size .le. n) then
197  do j = 1, 4*s+1
198  do l = 1, j
199  it = it + 1
200  do k = 1, neko_blk_size
201  temp(it,1) = temp(it,1) &
202  + pr(i+k,j) * pr(i+k,l) * coef%mult(i+k,1,1,1)
203  end do
204  end do
205  end do
206  else
207  do j = 1, 4*s+1
208  do l = 1, j
209  it = it + 1
210  do k = 1, n-i
211  temp(it,1) = temp(it,1) &
212  + pr(i+k,j) * pr(i+k,l) * coef%mult(i+k,1,1,1)
213  end do
214  end do
215  end do
216  end if
217  end do
218 
219  call mpi_allreduce(temp, temp2, it, &
220  mpi_real_precision, mpi_sum, neko_comm, ierr)
221  it = 0
222  do j = 1, 4*s+1
223  do k = 1, j
224  it = it + 1
225  g(j,k) = temp2(it,1)
226  g(k,j) = temp2(it,1)
227  end do
228  end do
229 
230  call mxm(g,4*s+1, tt, 4*s+1,gtt,4*s+1)
231 
232  do j = 1, s
233  iter = iter + 1
234 
235  call mxm(g, 4*s+1, r_c(1,j), 4*s+1,temp, 1)
236  call mxm(gtt, 4*s+1, p_c(1,j), 4*s+1,temp2, 1)
237  alpha1 = 0.0_rp
238  alpha2 = 0.0_rp
239  do i = 1,4*s+1
240  alpha1 = alpha1 + temp(i,1) * z_c(i,j)
241  alpha2 = alpha2 + temp2(i,1) * p_c(i,j)
242  end do
243  alpha(j) = alpha1/alpha2
244 
245  do i = 1, 4*s+1
246  x_c(i,j+1) = x_c(i,j) + alpha(j) * p_c(i,j)
247  tmp = 0.0_rp
248  do k = 1, 4*s+1
249  tmp = tmp + tt(i,k) * p_c(k,j)
250  end do
251  r_c(i,j+1) = r_c(i,j) - alpha(j)*tmp
252  tmp = 0.0_rp
253  do k = 1, 4*s+1
254  tmp = tmp + tt(i,k)*r_c(k,j+1)
255  end do
256  z_c(i,j+1) = tmp
257  end do
258 
259  call mxm(g,4*s+1,r_c(1,j+1),4*s+1,temp2,1)
260  alpha2 = 0.0_rp
261  do i = 1,4*s+1
262  alpha2 = alpha2 + temp2(i,1)*z_c(i,j+1)
263  end do
264  beta(j) = alpha2 / alpha1
265  do i = 1,4*s+1
266  p_c(i,j+1) = z_c(i,j+1) + beta(j)*p_c(i,j)
267  end do
268  end do
269 
270  call rzero(p, n)
271  call rzero(r, n)
272  rtr = 0.0_rp
273  do i = 0, n, neko_blk_size
274  if (i + neko_blk_size .le. n) then
275  do j = 1, 4*s + 1
276  do k = 1, neko_blk_size
277  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + pr(i+k,j) * x_c(j,s+1)
278  p(i+k) = p(i+k) + pr(i+k,j) * p_c(j,s+1)
279  tmp = pr(i+k,j) * r_c(j,s+1)
280  r(i+k) = r(i+k) + tmp
281  end do
282  end do
283  do k = 1, neko_blk_size
284  rtr = rtr + r(i+k)**2 * coef%mult(i+k,1,1,1)
285  end do
286  else
287  do j = 1,4*s+1
288  do k = 1, n-i
289  x%x(i+k,1,1,1) = x%x(i+k,1,1,1) + pr(i+k,j) * x_c(j,s+1)
290  p(i+k) = p(i+k) + pr(i+k,j) * p_c(j,s+1)
291  tmp = pr(i+k,j) * r_c(j,s+1)
292  r(i+k) = r(i+k) + tmp
293  end do
294  end do
295  do k = 1, n-i
296  rtr = rtr + r(i+k)**2 * coef%mult(i+k,1,1,1)
297  end do
298  end if
299  end do
300 
301  call mpi_allreduce(rtr, tmp, 1, &
302  mpi_real_precision, mpi_sum, neko_comm, ierr)
303  rnorm = norm_fac*sqrt(tmp)
304  if( rnorm <= this%abs_tol) exit
305  end do
306 
307  ksp_results%res_final = rnorm
308  ksp_results%iter = iter
309 
310  end associate
311 
312  end function cacg_solve
