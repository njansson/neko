132  function bicgstab_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
133  class(bicgstab_t), intent(inout) :: this
134  class(ax_t), intent(inout) :: ax
135  type(field_t), intent(inout) :: x
136  integer, intent(inout) :: n
137  real(kind=rp), dimension(n), intent(inout) :: f
138  type(coef_t), intent(inout) :: coef
139  type(bc_list_t), intent(inout) :: blst
140  type(gs_t), intent(inout) :: gs_h
141  type(ksp_monitor_t) :: ksp_results
142  integer, optional, intent(in) :: niter
143  integer :: iter, max_iter
144  real(kind=rp) :: rnorm, rtr, norm_fac, gamma
145  real(kind=rp) :: beta, alpha, omega, rho_1, rho_2
146 
147  if (present(niter)) then
148  max_iter = niter
149  else
150  max_iter = ksp_max_iter
151  end if
152  norm_fac = 1./sqrt(coef%volume)
153 
154  call rzero(x%x, n)
155  call copy(this%r, f, n)
156 
157  rtr = sqrt(glsc3(this%r,coef%mult, this%r, n))
158  rnorm = rtr*norm_fac
159  gamma = rnorm * this%rel_tol
160  ksp_results%res_start = rnorm
161  ksp_results%res_final = rnorm
162  ksp_results%iter = 0
163  if(rnorm .eq. 0d0) return
164  do iter = 1, max_iter
165 
166  rho_1 = glsc3(this%r,coef%mult,f,n)
167 
168  if (abs(rho_1) .lt. 1d-14) call neko_warning('Bi-CGStab rho failure')
169 
170  if (iter .eq. 1) then
171  call copy(this%p, this%r, n)
172  else
173  beta = (rho_1 / rho_2) * (alpha / omega)
174  call p_update(this%p, this%r, this%v, beta, omega, n)
175  end if
176 
177  call this%M%solve(this%p_hat, this%p, n)
178  call ax%compute(this%v, this%p_hat, coef, x%msh, x%Xh)
179  call gs_op(gs_h, this%v, n, gs_op_add)
180  call bc_list_apply(blst, this%v, n)
181  alpha = rho_1 / glsc3(f,coef%mult,this%v,n)
182  call copy(this%s, this%r, n)
183  call add2s2(this%s, this%v, -alpha, n)
184  rtr = glsc3(this%s, coef%mult, this%s, n)
185  rnorm = sqrt(rtr)*norm_fac
186  if (rnorm .lt. this%abs_tol .or. rnorm .lt. gamma) then
187  call add2s2(x%x, this%p_hat, alpha,n)
188  exit
189  end if
190 
191  call this%M%solve(this%s_hat, this%s, n)
192  call ax%compute(this%t, this%s_hat, coef, x%msh, x%Xh)
193  call gs_op(gs_h, this%t, n, gs_op_add)
194  call bc_list_apply(blst, this%t, n)
195  omega = glsc3(this%t,coef%mult,this%s, n) &
196  / glsc3(this%t, coef%mult, this%t, n)
197  call x_update(x%x, this%p_hat, this%s_hat, alpha, omega, n)
198  call copy(this%r, this%s, n)
199  call add2s2(this%r, this%t, -omega, n)
200 
201  rtr = glsc3(this%r, coef%mult, this%r, n)
202  rnorm = sqrt(rtr)*norm_fac
203  if (rnorm .lt. this%abs_tol .or. rnorm .lt. gamma) then
204  exit
205  end if
206 
207  if (omega .eq. 0d0) call neko_warning('Bi-CGstab omega failure')
208  rho_2 = rho_1
209 
210  end do
211  ksp_results%res_final = rnorm
212  ksp_results%iter = iter
213  end function bicgstab_solve
