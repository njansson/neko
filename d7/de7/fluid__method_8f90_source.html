148  subroutine fluid_scheme_init_common(this, msh, lx, params)
149  class(fluid_scheme_t), target, intent(inout) :: this
150  type(mesh_t), target, intent(inout) :: msh
151  integer, intent(inout) :: lx
152  type(param_t), target, intent(inout) :: params
153  type(dirichlet_t) :: bdry_mask
154  character(len=LOG_SIZE) :: log_buf
155 
156  call neko_log%section('Fluid')
157  call neko_log%message('Ksp vel. : ('// trim(params%ksp_vel) // &
158  ', ' // trim(params%pc_vel) // ')')
159  call neko_log%message('Ksp prs. : ('// trim(params%ksp_prs) // &
160  ', ' // trim(params%pc_prs) // ')')
161  write(log_buf, '(A, L1)') 'Dealias : ', params%dealias
162  call neko_log%message(log_buf)
163  write(log_buf, '(A, L1)') 'Save bdry: ', params%output_bdry
164  call neko_log%message(log_buf)
165 
166  if (msh%gdim .eq. 2) then
167  call space_init(this%Xh, gll, lx, lx)
168  else
169  call space_init(this%Xh, gll, lx, lx, lx)
170  end if
171 
172  this%dm_Xh = dofmap_t(msh, this%Xh)
173 
174  this%params => params
175 
176  this%msh => msh
177 
178  call gs_init(this%gs_Xh, this%dm_Xh)
179 
180  call coef_init(this%c_Xh, this%gs_Xh)
181 
182  call source_init(this%f_Xh, this%dm_Xh)
183 
184  !
185  ! Setup velocity boundary conditions
186  !
187  call bc_list_init(this%bclst_vel)
188 
189  call this%bc_sym%init(this%dm_Xh)
190  call this%bc_sym%mark_zone(msh%sympln)
191  call this%bc_sym%mark_zones_from_list(msh%labeled_zones,&
192  'sym', this%params%bc_labels)
193  call this%bc_sym%finalize()
194  call this%bc_sym%init_msk(this%c_Xh)
195  call bc_list_add(this%bclst_vel, this%bc_sym)
196 
197  if (trim(params%fluid_inflow) .eq. "default") then
198  allocate(inflow_t::this%bc_inflow)
199  else if (trim(params%fluid_inflow) .eq. "blasius") then
200  allocate(blasius_t::this%bc_inflow)
201  else if (trim(params%fluid_inflow) .eq. "user") then
202  allocate(usr_inflow_t::this%bc_inflow)
203  else
204  call neko_error('Invalid Inflow condition')
205  end if
206 
207  call this%bc_inflow%init(this%dm_Xh)
208  call this%bc_inflow%mark_zone(msh%inlet)
209  call this%bc_inflow%mark_zones_from_list(msh%labeled_zones,&
210  'v', this%params%bc_labels)
211  call this%bc_inflow%finalize()
212  call this%bc_inflow%set_inflow(params%uinf)
213  call bc_list_add(this%bclst_vel, this%bc_inflow)
214 
215  if (trim(params%fluid_inflow) .eq. "blasius") then
216  select type(bc_if => this%bc_inflow)
217  type is(blasius_t)
218  call bc_if%set_coef(this%C_Xh)
219  call bc_if%set_params(params%delta, params%blasius_approx)
220  end select
221  end if
222 
223  if (trim(params%fluid_inflow) .eq. "user") then
224  select type(bc_if => this%bc_inflow)
225  type is(usr_inflow_t)
226  call bc_if%set_coef(this%C_Xh)
227  end select
228  end if
229 
230  call this%bc_wall%init(this%dm_Xh)
231  call this%bc_wall%mark_zone(msh%wall)
232  call this%bc_wall%mark_zones_from_list(msh%labeled_zones,&
233  'w', this%params%bc_labels)
234  call this%bc_wall%finalize()
235  call bc_list_add(this%bclst_vel, this%bc_wall)
236 
237  if (params%output_bdry) then
238  call field_init(this%bdry, this%dm_Xh, 'bdry')
239  this%bdry = 0.0_rp
240 
241  call bdry_mask%init(this%dm_Xh)
242  call bdry_mask%mark_zone(msh%wall)
243  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
244  'w', this%params%bc_labels)
245  call bdry_mask%finalize()
246  call bdry_mask%set_g(1.0_rp)
247  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
248  call bdry_mask%free()
249 
250  call bdry_mask%init(this%dm_Xh)
251  call bdry_mask%mark_zone(msh%inlet)
252  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
253  'v', this%params%bc_labels)
254 
255  call bdry_mask%finalize()
256  call bdry_mask%set_g(2.0_rp)
257  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
258  call bdry_mask%free()
259 
260  call bdry_mask%init(this%dm_Xh)
261  call bdry_mask%mark_zone(msh%outlet)
262  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
263  'o', this%params%bc_labels)
264  call bdry_mask%finalize()
265  call bdry_mask%set_g(3.0_rp)
266  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
267  call bdry_mask%free()
268 
269  call bdry_mask%init(this%dm_Xh)
270  call bdry_mask%mark_zone(msh%sympln)
271  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
272  'sym', this%params%bc_labels)
273  call bdry_mask%finalize()
274  call bdry_mask%set_g(4.0_rp)
275  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
276  call bdry_mask%free()
277 
278  call bdry_mask%init(this%dm_Xh)
279  call bdry_mask%mark_zone(msh%periodic)
280  call bdry_mask%finalize()
281  call bdry_mask%set_g(5.0_rp)
282  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
283  call bdry_mask%free()
284 
285  call bdry_mask%init(this%dm_Xh)
286  call bdry_mask%mark_zone(msh%outlet_normal)
287  call bdry_mask%mark_zones_from_list(msh%labeled_zones,&
288  'on', this%params%bc_labels)
289  call bdry_mask%finalize()
290  call bdry_mask%set_g(6.0_rp)
291  call bdry_mask%apply_scalar(this%bdry%x, this%dm_Xh%n_dofs)
292  call bdry_mask%free()
293 
294  end if
295 
296  end subroutine fluid_scheme_init_common
297 
299  subroutine fluid_scheme_init_uvw(this, msh, lx, params, kspv_init)
300  class(fluid_scheme_t), target, intent(inout) :: this
301  type(mesh_t), target, intent(inout) :: msh
302  integer, intent(inout) :: lx

315  call fluid_scheme_precon_factory(this%pc_vel, this%ksp_vel, &

323  subroutine fluid_scheme_init_all(this, msh, lx, params, kspv_init, kspp_init)
324  class(fluid_scheme_t), target, intent(inout) :: this
325  type(mesh_t), target, intent(inout) :: msh
326  integer, intent(inout) :: lx
327  type(param_t), target, intent(inout) :: params
328  logical :: kspv_init
329  logical :: kspp_init
330 
331  call fluid_scheme_init_common(this, msh, lx, params)
332 
333  call field_init(this%u, this%dm_Xh, 'u')
334  call field_init(this%v, this%dm_Xh, 'v')
335  call field_init(this%w, this%dm_Xh, 'w')
336  call field_init(this%p, this%dm_Xh, 'p')
337 
338  !
339  ! Setup pressure boundary conditions
340  !
341  call bc_list_init(this%bclst_prs)
342  call this%bc_prs%init(this%dm_Xh)
343  call this%bc_prs%mark_zones_from_list(msh%labeled_zones,&
344  'o', this%params%bc_labels)
345  call this%bc_prs%mark_zones_from_list(msh%labeled_zones,&
346  'on', this%params%bc_labels)
347 
348  if (msh%outlet%size .gt. 0) then
349  call this%bc_prs%mark_zone(msh%outlet)
350  end if
351  if (msh%outlet_normal%size .gt. 0) then
352  call this%bc_prs%mark_zone(msh%outlet_normal)
353  end if
354 
355  call this%bc_prs%finalize()
356  call this%bc_prs%set_g(0.0_rp)
357  call bc_list_add(this%bclst_prs, this%bc_prs)
358  call this%bc_dong%init(this%dm_Xh)
359  call this%bc_dong%mark_zones_from_list(msh%labeled_zones,&
360  'o+dong', this%params%bc_labels)
361  call this%bc_dong%mark_zones_from_list(msh%labeled_zones,&
362  'on+dong', this%params%bc_labels)
363  call this%bc_dong%finalize()
364  call this%bc_dong%set_vars(this%c_Xh, this%u, this%v, this%w,&
365  params%dong_uchar, params%dong_delta)
366 
367  call bc_list_add(this%bclst_prs, this%bc_dong)
368 
369  if (kspv_init) then
370  call fluid_scheme_solver_factory(this%ksp_vel, this%dm_Xh%size(), &
371  params%ksp_vel, params%abstol_vel)
372  call fluid_scheme_precon_factory(this%pc_vel, this%ksp_vel, &
373  this%c_Xh, this%dm_Xh, this%gs_Xh, this%bclst_vel, params%pc_vel)
374  end if
375 
376  if (kspp_init) then
377  call fluid_scheme_solver_factory(this%ksp_prs, this%dm_Xh%size(), &
378  params%ksp_prs, params%abstol_prs)
379  call fluid_scheme_precon_factory(this%pc_prs, this%ksp_prs, &
380  this%c_Xh, this%dm_Xh, this%gs_Xh, this%bclst_prs, params%pc_prs)
381  end if
382 
383 
384  call neko_log%end_section()
385 
386  end subroutine fluid_scheme_init_all

441  subroutine fluid_scheme_validate(this)
442  class(fluid_scheme_t), target, intent(inout) :: this
443 
444  if ( (.not. allocated(this%u%x)) .or. &
445  (.not. allocated(this%v%x)) .or. &
446  (.not. allocated(this%w%x)) .or. &
447  (.not. allocated(this%p%x))) then
448  call neko_error('Fields are not allocated')
449  end if
450 
451  if (.not. allocated(this%ksp_vel)) then
452  call neko_error('No Krylov solver for velocity defined')
453  end if
454 
455  if (.not. allocated(this%ksp_prs)) then
456  call neko_error('No Krylov solver for pressure defined')
457  end if
458 
459  if (.not. associated(this%f_Xh%eval)) then
460  call neko_error('No source term defined')
461  end if
462 
463  if (.not. associated(this%params)) then
464  call neko_error('No parameters defined')
465  end if
466 
467  select type(ip => this%bc_inflow)
468  type is(usr_inflow_t)
469  call ip%validate
470  end select
471 
472  !
473  ! Setup checkpoint structure (if everything is fine)
474  !
475  call this%chkp%init(this%u, this%v, this%w, this%p)
476 
477  !
478  ! Setup mean flow fields if requested
479  !
480  if (this%params%stats_mean_flow) then
481  call this%mean%init(this%u, this%v, this%w, this%p)
482  end if
483 
484  if (this%params%stats_mean_sqr_flow) then
485  call this%mean_sqr%init(this%u, this%v, this%w, this%p)
486  end if
487 
488  end subroutine fluid_scheme_validate

518  subroutine fluid_scheme_precon_factory(pc, ksp, coef, dof, gs, bclst, pctype)
519  class(pc_t), allocatable, target, intent(inout) :: pc
520  class(ksp_t), target, intent(inout) :: ksp
521  type(coef_t), target, intent(inout) :: coef
522  type(dofmap_t), target, intent(inout) :: dof
523  type(gs_t), target, intent(inout) :: gs
524  type(bc_list_t), target, intent(inout) :: bclst
525  character(len=20) :: pctype
526 
527  call precon_factory(pc, pctype)
528 
529  select type(pcp => pc)
530  type is(jacobi_t)
531  call pcp%init(coef, dof, gs)
532  type is (sx_jacobi_t)
533  call pcp%init(coef, dof, gs)
534  type is (device_jacobi_t)
535  call pcp%init(coef, dof, gs)
536  type is(hsmg_t)
537  if (len_trim(pctype) .gt. 4) then
538  if (index(pctype, '+') .eq. 5) then
539  call pcp%init(dof%msh, dof%Xh, coef, dof, gs, &
540  bclst, trim(pctype(6:)))
541  else
542  call neko_error('Unknown coarse grid solver')
543  end if
544  else
545  call pcp%init(dof%msh, dof%Xh, coef, dof, gs, bclst)
546  end if
547  end select
548 
549  call ksp%set_pc(pc)
550 
551  end subroutine fluid_scheme_precon_factory
