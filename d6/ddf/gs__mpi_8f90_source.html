157  subroutine gs_nbwait_mpi(this, u, n, op)
158  class(gs_mpi_t), intent(inout) :: this
159  integer, intent(in) :: n
160  real(kind=rp), dimension(n), intent(inout) :: u
161  integer :: i, j, src, ierr
162  integer :: op
163  integer , pointer :: sp(:)
164  integer :: nreqs
165 
166  nreqs = size(this%recv_pe)
167 
168  do while (nreqs .gt. 0)
169  do i = 1, size(this%recv_pe)
170  if (.not. this%recv_buf(i)%flag) then
171  call mpi_test(this%recv_buf(i)%request, this%recv_buf(i)%flag, &
172  this%recv_buf(i)%status, ierr)
173  if (this%recv_buf(i)%flag) then
174  nreqs = nreqs - 1
176  src = this%recv_pe(i)
177  sp => this%recv_dof(src)%array()
178  select case(op)
179  case (gs_op_add)
180  !NEC$ IVDEP
181  do j = 1, this%send_dof(src)%size()
182  u(sp(j)) = u(sp(j)) + this%recv_buf(i)%data(j)
183  end do
184  case (gs_op_mul)
185  !NEC$ IVDEP
186  do j = 1, this%send_dof(src)%size()
187  u(sp(j)) = u(sp(j)) * this%recv_buf(i)%data(j)
188  end do
189  case (gs_op_min)
190  !NEC$ IVDEP
191  do j = 1, this%send_dof(src)%size()
192  u(sp(j)) = min(u(sp(j)), this%recv_buf(i)%data(j))
193  end do
194  case (gs_op_max)
195  !NEC$ IVDEP
196  do j = 1, this%send_dof(src)%size()
197  u(sp(j)) = max(u(sp(j)), this%recv_buf(i)%data(j))
198  end do
199  end select
200  end if
201  end if
202  end do
203  end do
204 
205  nreqs = size(this%send_pe)
206  do while (nreqs .gt. 0)
207  do i = 1, size(this%send_pe)
208  if (.not. this%send_buf(i)%flag) then
209  call mpi_test(this%send_buf(i)%request, this%send_buf(i)%flag, &
210  mpi_status_ignore, ierr)
211  if (this%send_buf(i)%flag) nreqs = nreqs - 1
212  end if
213  end do
214  end do
215 
216  end subroutine gs_nbwait_mpi
