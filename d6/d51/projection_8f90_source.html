250  subroutine cpu_project_on(this, b, coef, n)
251  class(projection_t), intent(inout) :: this
252  integer, intent(inout) :: n
253  class(coef_t), intent(inout) :: coef
254  real(kind=rp), intent(inout), dimension(n) :: b
255  integer :: i, j, k, ierr
256  real(kind=rp) :: work(this%L), alpha(this%L)
257 
258  associate(xbar => this%xbar, xx => this%xx, &
259  bb => this%bb)
260 
261  if (this%m .le. 0) return
262 
263  !First round of CGS
264  call rzero(alpha, this%m)
265  this%proj_res = glsc3(b,b,coef%mult,n)
266  this%proj_m = this%m
267  do i = 1, n, neko_blk_size
268  j = min(neko_blk_size, n-i+1)
269  do k = 1, this%m
270  alpha(k) = alpha(k) + vlsc3(xx(i,k), coef%mult(i,1,1,1), b(i), j)
271  end do
272  end do
273 
274  !First one outside loop to avoid zeroing xbar and bbar
275  call mpi_allreduce(mpi_in_place, alpha, this%m, &
276  mpi_real_precision, mpi_sum, neko_comm, ierr)
277 
278  call rzero(work, this%m)
279 
280  do i = 1, n, neko_blk_size
281  j = min(neko_blk_size, n-i+1)
282  call cmult2(xbar(i), xx(i,1), alpha(1), j)
283  call add2s2(b(i), bb(i,1), -alpha(1), j)
284  do k = 2,this%m
285  call add2s2(xbar(i), xx(i,k), alpha(k), j)
286  call add2s2(b(i), bb(i,k), -alpha(k), j)
287  end do
288  !Second round of CGS
289  do k = 1, this%m
290  work(k) = work(k) + vlsc3(xx(i,k), coef%mult(i,1,1,1), b(i), j)
291  end do
292  end do
293 
294  call mpi_allreduce(work, alpha, this%m, &
295  mpi_real_precision, mpi_sum, neko_comm, ierr)
296 
297  do i = 1, n, neko_blk_size
298  j = min(neko_blk_size, n-i+1)
299  do k = 1,this%m
300  call add2s2(xbar(i), xx(i,k), alpha(k), j)
301  call add2s2(b(i), bb(i,k), -alpha(k), j)
302  end do
303  end do
304  end associate
305  end subroutine cpu_project_on

321

332  call device_memcpy(alpha, alpha_d, this%m, host_to_device)
333  call device_add2s2_many(xbar_d, xx_d_d, alpha_d, this%m, n)

340  !This is a lot more primitive than on the CPU
341  subroutine device_proj_ortho(this, xx_d, bb_d, w_d, n)
342  type(projection_t) :: this
343  integer, intent(inout) :: n
344  type(c_ptr), dimension(this%L) :: xx_d, bb_d
345  type(c_ptr) :: w_d
346  real(kind=rp) :: nrm, scl
347  real(kind=rp) :: alpha(this%L)
348 
349  associate(m => this%m, xx_d_d => this%xx_d_d, &
350  bb_d_d => this%bb_d_d, alpha_d => this%alpha_d)
351 
352  if(m .le. 0) return
353 
354  call device_glsc3_many(alpha,bb_d(m),xx_d_d,w_d,m,n)
355  nrm = sqrt(alpha(m))
356  call cmult(alpha, -1.0_rp,m)
357  call device_memcpy(alpha, alpha_d, this%m, host_to_device)
358  call device_add2s2_many(xx_d(m),xx_d_d,alpha_d,m-1,n)
359  call device_add2s2_many(bb_d(m),bb_d_d,alpha_d,m-1,n)
360 
361  call device_glsc3_many(alpha,bb_d(m),xx_d_d,w_d,m,n)
362  call cmult(alpha, -1.0_rp,m)
363  call device_memcpy(alpha, alpha_d, m, host_to_device)
364  call device_add2s2_many(xx_d(m),xx_d_d,alpha_d,m-1,n)
365  call device_add2s2_many(bb_d(m),bb_d_d,alpha_d,m-1,n)
366  call device_glsc3_many(alpha,bb_d(m),xx_d_d,w_d,m,n)
367  alpha(m) = device_glsc3(xx_d(m), w_d, bb_d(m), n)
368  alpha(m) = sqrt(alpha(m))
369 
370  if(alpha(m) .gt. this%tol*nrm) then !New vector is linearly independent
371  scl = 1.0_rp / alpha(m)
372  call device_cmult(xx_d(m), scl, n)
373  call device_cmult(bb_d(m), scl, n)
374 
375 
376  else !New vector is not linearly independent, forget about it
377  if(pe_rank .eq. 0) then
378  call neko_warning('New vector not linearly indepependent!')
379  end if
380  m = m - 1 !Remove column
381  endif
382 
383  end associate
384 
385  end subroutine device_proj_ortho

388  subroutine cpu_proj_ortho(this, xx, bb, w, n)
389  type(projection_t) :: this
390  integer, intent(inout) :: n
391  real(kind=rp), dimension(n, this%L), intent(inout) :: xx, bb
392  real(kind=rp), dimension(n), intent(inout) :: w
393  real(kind=rp) :: nrm, scl1, scl2, c, s
394  real(kind=rp) :: alpha(this%L), beta(this%L)
395  integer :: i, j, k, h, ierr
396 
397  associate(m => this%m)
398 
399  if(m .le. 0) return !No vectors to ortho-normalize
400 
401  ! AX = B
402  ! Calculate dx, db: dx = x-XX^Tb, db=b-BX^Tb
403  call rzero(alpha, m)
404  do i = 1, n, neko_blk_size
405  j = min(neko_blk_size, n-i+1)
406  do k = 1, m !First round CGS
407  alpha(k) = alpha(k) + 0.5_rp * (vlsc3(xx(i,k), w(i), bb(i,m), j) &
408  + vlsc3(bb(i,k), w(i), xx(i,m), j))
409  end do
410  end do
411 
412  call mpi_allreduce(mpi_in_place, alpha, this%m, &
413  mpi_real_precision, mpi_sum, neko_comm, ierr)
414 
415  nrm = sqrt(alpha(m)) !Calculate A-norm of new vector
416 
417  call rzero(beta,m)
418 
419  do i = 1, n, neko_blk_size
420  j = min(neko_blk_size, n-i+1)
421  do k = 1,m-1
422  call add2s2(xx(i,m), xx(i,k), -alpha(k), j)
423  call add2s2(bb(i,m), bb(i,k), -alpha(k), j)
424  beta(k) = beta(k) + 0.5_rp * (vlsc3(xx(i,k), w(i), bb(i,m), j) &
425  + vlsc3(bb(i,k), w(i), xx(i,m), j))
426  end do
427  end do
428 
429  call mpi_allreduce(mpi_in_place, beta, this%m-1, &
430  mpi_real_precision, mpi_sum, neko_comm, ierr)
431 
432  alpha(m) = 0.0_rp
433 
434  do i = 1, n, neko_blk_size
435  j = min(neko_blk_size,n-i+1)
436  do k = 1, m-1
437  call add2s2(xx(i,m), xx(i,k), -beta(k), j)
438  call add2s2(bb(i,m), bb(i,k), -beta(k), j)
439  end do
440  alpha(m) = alpha(m) + vlsc3(xx(i,m), w(i), bb(i,m), j)
441  end do
442  do k = 1, m-1
443  alpha(k) = alpha(k) + beta(k)
444  end do
445 
446  !alpha(m) = glsc3(xx(1,m), w, bb(1,m), n)
447  call mpi_allreduce(mpi_in_place, alpha(m), 1, &
448  mpi_real_precision, mpi_sum, neko_comm, ierr)
449  alpha(m) = sqrt(alpha(m))
450  !dx and db now stored in last column of xx and bb
451 
452  if(alpha(m) .gt. this%tol*nrm) then !New vector is linearly independent
453  !Normalize dx and db
454  scl1 = 1.0_rp / alpha(m)
455  call cmult(xx(1,m), scl1, n)
456  call cmult(bb(1,m), scl1, n)
457 
458  !We want to throw away the oldest information
459  !The below propagates newest information to first vector.
460  !This will make the first vector a scalar
461  !multiple of x.
462  do k = m, 2, -1
463  h = k - 1
464  call givens_rotation(alpha(h), alpha(k), c, s, nrm)
465  alpha(h) = nrm
466  do i = 1, n !Apply rotation to xx and bb
467  scl1 = c*xx(i,h) + s*xx(i,k)
468  xx(i,k) = -s*xx(i,h) + c*xx(i,k)
469  xx(i,h) = scl1
470  scl2 = c*bb(i,h) + s*bb(i,k)
471  bb(i,k) = -s*bb(i,h) + c*bb(i,k)
472  bb(i,h) = scl2
473  end do
474  end do
475 
476  else !New vector is not linearly independent, forget about it
477  k = m !location of rank deficient column
478  if(pe_rank .eq. 0) then
479  call neko_warning('New vector not linearly indepependent!')
480  end if
481  m = m - 1 !Remove column
482  endif
483 
484  end associate
485 
486  end subroutine cpu_proj_ortho
487 
488  subroutine givens_rotation(a, b, c, s, r)
489  real(kind=rp), intent(inout) :: a, b, c, s, r
490  real(kind=rp) :: h, d
491 
492  if(b .ne. 0.0_rp) then
493  h = hypot(a, b)
494  d = 1.0_rp / h
495  c = abs(a) * d
496  s = sign(d, a) * b
497  r = sign(1.0_rp, a) * h
498  else
499  c = 1.0_rp
500  s = 0.0_rp
501  r = a
502  endif
503 
504  return
505  end subroutine givens_rotation
