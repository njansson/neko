96  subroutine schwarz_init(this, Xh, dm, gs_h, bclst, msh)
97  class(schwarz_t), target, intent(inout) :: this
98  type(space_t), target, intent(inout) :: Xh
99  type(dofmap_t), target, intent(inout) :: dm
100  type(gs_t), target, intent(inout) :: gs_h
101  type(mesh_t), target, intent(inout) :: msh
102  type(bc_list_t), target, intent(inout):: bclst
103 
104  call this%free()
105 
106  call space_init(this%Xh_schwarz, gll, xh%lx+2, xh%lx+2, xh%lx+2)
107  this%dm_schwarz = dofmap_t(msh, this%Xh_schwarz)
108  call gs_init(this%gs_schwarz, this%dm_schwarz)
109 
110  allocate(this%work1(this%dm_schwarz%n_dofs))
111  allocate(this%work2(this%dm_schwarz%n_dofs))

114  call fdm_init(this%fdm,xh, dm, gs_h)
115 
116 
117  this%msh => msh
118  this%Xh => xh
119  this%bclst => bclst
120  this%dm => dm
121  this%gs_h => gs_h
122  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
123  .or. (neko_bcknd_opencl .eq. 1)) then
124  call device_map(this%work1, this%work1_d,this%dm_schwarz%n_dofs)
125  call device_map(this%work2, this%work2_d,this%dm_schwarz%n_dofs)

128

131  .or. (neko_bcknd_opencl .eq. 1)) then
132  call device_alloc(this%wt_d,int(this%dm%n_dofs*rp, i8))
133  call rone(this%work1, this%dm%n_dofs)

159  subroutine schwarz_setup_wt(this)
160  class(schwarz_t), intent(inout) :: this
161  integer :: enx,eny,enz, n, ie, k, ns
162  real(kind=rp), parameter :: zero = 0.0
163  real(kind=rp), parameter :: one = 1.0
164  associate(work1 => this%work1, work2 => this%work2, msh => this%msh, &
165  xh => this%Xh, xh_schwarz => this%Xh_schwarz)
166 
167  n = this%dm%n_dofs
168 
169  enx = xh_schwarz%lx
170  eny = xh_schwarz%ly
171  enz = xh_schwarz%lz
172  if(.not. msh%gdim .eq. 3) enz=1
173  ns = enx*eny*enz*msh%nelv
174 
175  call rone(work2, ns)
176  call rzero(work1, ns)
177 
178  ! Sum overlap region (border excluded)
179  ! Cred to PFF for this, very clever
180  call schwarz_extrude(work1, 0, zero, work2, 0, one , enx, eny, enz, msh%nelv)
181  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
182  .or. (neko_bcknd_opencl .eq. 1)) then
183  call device_memcpy(work2, this%work2_d, ns, host_to_device)
184  call gs_op(this%gs_schwarz, work2, ns, gs_op_add)
185  call device_memcpy(work2, this%work2_d, ns, device_to_host)
186  else
187  call gs_op(this%gs_schwarz, work2, ns, gs_op_add)
188  end if
189  call schwarz_extrude(work2, 0, one, work1, 0, -one, enx, eny, enz, msh%nelv)
190  call schwarz_extrude(work2, 2, one, work2, 0, one, enx, eny, enz, msh%nelv)
191 
192  ! if(.not.if3d) then ! Go back to regular size array
193  ! call hsmg_schwarz_toreg2d(mg_work,mg_work(i),mg_nh(l))
194  ! else
195  call schwarz_toreg3d(work1, work2, xh%lx, msh%nelv)
196  ! endif
197 
198  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
199  .or. (neko_bcknd_opencl .eq. 1)) then
200  call device_memcpy(work1, this%work1_d, n, host_to_device)
201  call gs_op(this%gs_h, work1, n, gs_op_add)
202  call device_memcpy(work1, this%work1_d, n, device_to_host)
203  else
204  call gs_op(this%gs_h, work1, n, gs_op_add)
205  end if
206 
207  k = 1
208  do ie = 1,msh%nelv
209  if (msh%gdim .eq. 2) then
210  call schwarz_setup_schwarz_wt2d_2(this%wt,ie,xh%lx, work1(k), msh%nelv)
211  end if
212  if (this%msh%gdim.eq. 3) then
213  call schwarz_setup_schwarz_wt3d_2(this%wt,ie,xh%lx, work1(k), msh%nelv)
214  k = k + xh%lxyz
215  end if
216  end do
217  end associate
218  end subroutine schwarz_setup_wt

221  subroutine schwarz_setup_schwarz_wt2d_2(wt,ie,n,work, nelv)
222  integer, intent(in) :: n, nelv
223  real(kind=rp), intent(inout) :: wt(n,4,2,nelv)
224  real(kind=rp), intent(inout) :: work(n,n)
225  integer :: ie,i,j
226  do j = 1, n
227  wt(j,1,1,ie) = 1.0_rp / work(1,j)
228  wt(j,2,1,ie) = 1.0_rp / work(2,j)
229  wt(j,3,1,ie) = 1.0_rp / work(n-1,j)
230  wt(j,4,1,ie) = 1.0_rp / work(n,j)
231  end do
232  do i = 1, n
233  wt(i,1,2,ie) = 1.0_rp / work(i,1)
234  wt(i,2,2,ie) = 1.0_rp / work(i,2)
235  wt(i,3,2,ie) = 1.0_rp / work(i,n-1)
236  wt(i,4,2,ie) = 1.0_rp / work(i,n)
237  end do
238 
239  return
240  end subroutine schwarz_setup_schwarz_wt2d_2
241 
243  subroutine schwarz_setup_schwarz_wt3d_2(wt, ie, n, work, nelv)
244  integer, intent(in) ::n, nelv, ie
245  real(kind=rp), intent(inout) :: wt(n,n,4,3,nelv)
246  real(kind=rp), intent(inout) :: work(n,n,n)
247  integer :: i,j,k
248 
249  do k = 1, n
250  do j = 1, n
251  wt(j,k,1,1,ie) = 1.0_rp / work(1,j,k)
252  wt(j,k,2,1,ie) = 1.0_rp / work(2,j,k)
253  wt(j,k,3,1,ie) = 1.0_rp / work(n-1,j,k)
254  wt(j,k,4,1,ie) = 1.0_rp / work(n,j,k)
255  end do
256  end do
257 
258  do k = 1, n
259  do i = 1, n
260  wt(i,k,1,2,ie) = 1.0_rp / work(i,1,k)
261  wt(i,k,2,2,ie) = 1.0_rp / work(i,2,k)
262  wt(i,k,3,2,ie) = 1.0_rp / work(i,n-1,k)
263  wt(i,k,4,2,ie) = 1.0_rp / work(i,n,k)
264  end do
265  end do
266 
267  do j = 1, n
268  do i = 1, n
269  wt(i,j,1,3,ie) = 1.0_rp / work(i,j,1)
270  wt(i,j,2,3,ie) = 1.0_rp / work(i,j,2)
271  wt(i,j,3,3,ie) = 1.0_rp / work(i,j,n-1)
272  wt(i,j,4,3,ie) = 1.0_rp / work(i,j,n)
273  end do
274  end do
275 
276  end subroutine schwarz_setup_schwarz_wt3d_2

316  subroutine schwarz_extrude(arr1, l1, f1, arr2, l2, f2, nx, ny, nz, nelv)
317  integer, intent(in) :: l1, l2, nx, ny, nz, nelv
318  real(kind=rp), intent(inout) :: arr1(nx,ny,nz,nelv), arr2(nx,ny,nz,nelv)
319  real(kind=rp), intent(in) :: f1, f2
320  integer :: i, j, k, ie, i0, i1
321  i0=2
322  i1=nx-1
323 
324  if(nz .eq. 1) then
325  do ie = 1, nelv
326  do j = i0, i1
327  arr1(l1+1 ,j,1,ie) = f1*arr1(l1+1 ,j,1,ie) &
328  +f2*arr2(l2+1 ,j,1,ie)
329  arr1(nx-l1,j,1,ie) = f1*arr1(nx-l1,j,1,ie) &
330  +f2*arr2(nx-l2,j,1,ie)
331  end do
332  do i = i0, i1
333  arr1(i,l1+1 ,1,ie) = f1*arr1(i,l1+1 ,1,ie) &
334  +f2*arr2(i,l2+1 ,1,ie)
335  arr1(i,ny-l1,1,ie) = f1*arr1(i,ny-l1,1,ie) &
336  +f2*arr2(i,nx-l2,1,ie)
337  end do
338  end do
339  else
340  do ie = 1, nelv
341  do k = i0, i1
342  do j = i0, i1
343  arr1(l1+1 ,j,k,ie) = f1*arr1(l1+1 ,j,k,ie) &
344  +f2*arr2(l2+1 ,j,k,ie)
345  arr1(nx-l1,j,k,ie) = f1*arr1(nx-l1,j,k,ie) &
346  +f2*arr2(nx-l2,j,k,ie)
347  end do
348  end do
349  do k = i0, i1
350  do i = i0, i1
351  arr1(i,l1+1 ,k,ie) = f1*arr1(i,l1+1 ,k,ie) &
352  +f2*arr2(i,l2+1 ,k,ie)
353  arr1(i,nx-l1,k,ie) = f1*arr1(i,nx-l1,k,ie) &
354  +f2*arr2(i,nx-l2,k,ie)
355  end do
356  end do
357  do j = i0, i1
358  do i = i0, i1
359  arr1(i,j,l1+1 ,ie) = f1*arr1(i,j,l1+1 ,ie) &
360  +f2*arr2(i,j,l2+1 ,ie)
361  arr1(i,j,nx-l1,ie) = f1*arr1(i,j,nx-l1,ie) &
362  +f2*arr2(i,j,nx-l2,ie)
363  end do
364  end do
365  end do
366  endif
367  end subroutine schwarz_extrude

369  subroutine schwarz_compute(this, e, r)
370  class(schwarz_t), intent(inout) :: this
371  real(kind=rp), dimension(this%dm%n_dofs), intent(inout) :: e, r
372  integer :: n, enx, eny, enz, ns
373  real(kind=rp), parameter :: zero = 0.0
374  real(kind=rp), parameter :: one = 1.0
375  type(c_ptr) :: e_d, r_d
376  associate(work1 => this%work1, work1_d => this%work1_d,&
377  work2 => this%work2, work2_d => this%work2_d)
378 
379  n = this%dm%n_dofs
380  enx=this%Xh_schwarz%lx
381  eny=this%Xh_schwarz%ly
382  enz=this%Xh_schwarz%lz
383  if(.not. this%msh%gdim .eq. 3) enz=1
384  ns = enx*eny*enz*this%msh%nelv
385  if ((neko_bcknd_cuda .eq. 1) .or. (neko_bcknd_hip .eq. 1) &
386  .or. (neko_bcknd_opencl .eq. 1)) then
387  r_d = device_get_ptr(r)
388  e_d = device_get_ptr(e)
389  call bc_list_apply_scalar(this%bclst, r, n)
390  call device_schwarz_toext3d(work1_d, r_d, this%Xh%lx, this%msh%nelv)
391  call device_schwarz_extrude(work1_d, 0, zero, work1_d, 2, one ,enx,eny,enz, this%msh%nelv)
392  call gs_op(this%gs_schwarz, work1, ns, gs_op_add)
393  call device_schwarz_extrude(work1_d, 0, one, work1_d, 2, -one, enx, eny, enz, this%msh%nelv)
394 
395  call this%fdm%compute(work2, work1) ! do local solves
396 
397  call device_schwarz_extrude(work1_d, 0, zero, work2_d, 0, one, enx, eny, enz, this%msh%nelv)
398  call gs_op(this%gs_schwarz, work2, ns, gs_op_add)
399  call device_schwarz_extrude(work2_d, 0, one, work1_d, 0, -one, enx, eny, enz, this%msh%nelv)
400  call device_schwarz_extrude(work2_d, 2, one, work2_d, 0, one, enx, eny, enz, this%msh%nelv)
401  call device_schwarz_toreg3d(e_d, work2_d, this%Xh%lx, this%msh%nelv)
402 
403  call gs_op(this%gs_h, e, n, gs_op_add)
404  call bc_list_apply_scalar(this%bclst, e, n)
405  call device_col2(e_d,this%wt_d, n)
406  else
407  call bc_list_apply_scalar(this%bclst, r, n)
408  !if (if3d) then ! extended array
409  call schwarz_toext3d(work1, r, this%Xh%lx, this%msh%nelv)
410  ! else
411  ! call hsmg_schwarz_toext2d(mg_work,r,mg_nh(l))
412  ! endif
413 
414 
415 
416  ! exchange interior nodes
417  call schwarz_extrude(work1, 0, zero, work1, 2, one, enx, eny, enz, this%msh%nelv)
418  call gs_op(this%gs_schwarz, work1, ns, gs_op_add)
419  call schwarz_extrude(work1, 0, one, work1, 2, -one, enx, eny, enz, this%msh%nelv)
420 
421  call this%fdm%compute(work2, work1) ! do local solves
422 
423  ! Sum overlap region (border excluded)
424  call schwarz_extrude(work1, 0, zero, work2, 0, one, enx, eny, enz, this%msh%nelv)
425  call gs_op(this%gs_schwarz, work2, ns, gs_op_add)
426  call schwarz_extrude(work2, 0, one, work1, 0, -one, enx, eny, enz, this%msh%nelv)
427  call schwarz_extrude(work2, 2, one, work2, 0, one, enx, eny, enz, this%msh%nelv)
428 
429  ! if(.not.if3d) then ! Go back to regular size array
430  ! call hsmg_schwarz_toreg2d(mg_work,mg_work(i),mg_nh(l))
431  ! else
432  call schwarz_toreg3d(e, work2, this%Xh%lx, this%msh%nelv)
433  ! endif
434 
435 
436  ! sum border nodes
437  call gs_op(this%gs_h, e, n, gs_op_add)
438  call bc_list_apply_scalar(this%bclst, e, n)
439 
440  ! if(.not.if3d) call hsmg_schwarz_wt2d(e,mg_schwarz_wt(mg_schwarz_wt_index(l,mg_fld)),mg_nh(l))
441  !if(if3d)
442  call schwarz_wt3d(e, this%wt, this%Xh%lx, this%msh%nelv)
443  end if
444  end associate
445  end subroutine schwarz_compute

447  !Apply schwarz weights along the boundary of each element.
448  subroutine schwarz_wt3d(e,wt,n, nelv)
449  integer, intent(in) :: n, nelv
450  real(kind=rp), intent(inout) :: e(n,n,n,nelv)
451  real(kind=rp), intent(inout) :: wt(n,n,4,3,nelv)
452  integer :: ie, i, j, k
453 
454  do ie = 1, nelv
455  do k = 1, n
456  do j = 1, n
457  e(1 ,j,k,ie) = e(1 ,j,k,ie) * wt(j,k,1,1,ie)
458  e(2 ,j,k,ie) = e(2 ,j,k,ie) * wt(j,k,2,1,ie)
459  e(n-1,j,k,ie) = e(n-1,j,k,ie) * wt(j,k,3,1,ie)
460  e(n ,j,k,ie) = e(n ,j,k,ie) * wt(j,k,4,1,ie)
461  end do
462  end do
463  do k = 1, n
464  do i = 3, n-2
465  e(i,1 ,k,ie) = e(i,1 ,k,ie) * wt(i,k,1,2,ie)
466  e(i,2 ,k,ie) = e(i,2 ,k,ie) * wt(i,k,2,2,ie)
467  e(i,n-1,k,ie) = e(i,n-1,k,ie) * wt(i,k,3,2,ie)
468  e(i,n ,k,ie) = e(i,n ,k,ie) * wt(i,k,4,2,ie)
469  end do
470  end do
471  do j = 3, n-2
472  do i = 3, n-2
473  e(i,j,1 ,ie) = e(i,j,1 ,ie) * wt(i,j,1,3,ie)
