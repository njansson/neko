218  subroutine bc_mark_zones_from_list(this, bc_zones, bc_key, bc_labels)

223  integer :: i, j, k, msh_bc_type
224 
225  msh_bc_type = 0
226  if(trim(bc_key) .eq. 'o' .or. trim(bc_key) .eq. 'on' &

228  msh_bc_type = 1

238  if (trim(bc_key) .eq. trim(bc_labels(i))) then
239  call bc_mark_zone(this, bc_zones(i))
240  do j = 1,this%msh%nelv
241  do k = 1, 2 * this%msh%gdim
242  if (this%msh%facet_type(k,j) .eq. -i) then
243  this%msh%facet_type(k,j) = msh_bc_type
244  end if
245  end do
246  end do
247  end if
248  end do

254  subroutine bc_finalize(this)
255  class(bc_t), target, intent(inout) :: this
256  type(tuple_i4_t), pointer :: bfp(:)
257  type(tuple_i4_t) :: bc_facet
258  integer :: facet_size, facet, el
259  integer :: i, j, k, l, msk_c
260  integer :: lx, ly, lz, n
261 
262  lx = this%Xh%lx
263  ly = this%Xh%ly
264  lz = this%Xh%lz
265 
267 
268  ! Note we assume that lx = ly = lz
269  facet_size = lx**2
270  allocate(this%msk(0:facet_size * this%marked_facet%size()))
271  allocate(this%facet(0:facet_size * this%marked_facet%size()))
272 
273  msk_c = 0
274  bfp => this%marked_facet%array()
275  do i = 1, this%marked_facet%size()
276  bc_facet = bfp(i)
277  facet = bc_facet%x(1)
278  el = bc_facet%x(2)
279  select case (facet)
280  case (1)
281  do l = 1, lz
282  do k = 1, ly
283  msk_c = msk_c + 1
284  this%msk(msk_c) = linear_index(1,k,l,el,lx,ly,lz)
285  this%facet(msk_c) = 1
286  end do
287  end do
288  case (2)
289  do l = 1, lz
290  do k = 1, ly
291  msk_c = msk_c + 1
292  this%msk(msk_c) = linear_index(lx,k,l,el,lx,ly,lz)
293  this%facet(msk_c) = 2
294  end do
295  end do
296  case(3)
297  do l = 1, lz
298  do j = 1, lx
299  msk_c = msk_c + 1
300  this%msk(msk_c) = linear_index(j,1,l,el,lx,ly,lz)
301  this%facet(msk_c) = 3
302  end do
303  end do
304  case(4)
305  do l = 1, lz
306  do j = 1, lx
307  msk_c = msk_c + 1
308  this%msk(msk_c) = linear_index(j,ly,l,el,lx,ly,lz)
309  this%facet(msk_c) = 4
310  end do
311  end do
312  case(5)
313  do k = 1, ly
314  do j = 1, lx
315  msk_c = msk_c + 1
316  this%msk(msk_c) = linear_index(j,k,1,el,lx,ly,lz)
317  this%facet(msk_c) = 5
318  end do
319  end do
320  case(6)
321  do k = 1, ly
322  do j = 1, lx
323  msk_c = msk_c + 1
324  this%msk(msk_c) = linear_index(j,k,lz,el,lx,ly,lz)
325  this%facet(msk_c) = 6
326  end do
327  end do
328  end select
329  end do
330 
331  this%msk(0) = msk_c
332  this%facet(0) = msk_c
333 
334  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
335  (neko_bcknd_opencl .eq. 1)) then
336  n = facet_size * this%marked_facet%size() + 1
337  call device_map(this%msk, this%msk_d, n)
338  call device_map(this%facet, this%facet_d, n)
339 
340  call device_memcpy(this%msk, this%msk_d, n, host_to_device)
341  call device_memcpy(this%facet, this%facet_d, n, host_to_device)
342  end if
343 
344  end subroutine bc_finalize
