114  pure function nonlinear_index(linear_index,lx,ly,lz) result(index)
115  integer, intent(in) :: linear_index, lx, ly, lz
116  integer :: index(4)
117  integer :: lin_idx
118  lin_idx = linear_index -1
119  index(4) = lin_idx/(lx*ly*lz)
120  index(3) = (lin_idx-(lx*ly*lz)*index(4))/(lx*ly)
121  index(2) = (lin_idx-(lx*ly*lz)*index(4)-(lx*ly)*index(3))/lx
122  index(1) = (lin_idx-(lx*ly*lz)*index(4)-(lx*ly)*index(3)-lx*index(2))
123  index(1) = index(1) + 1
124  index(2) = index(2) + 1
125  index(3) = index(3) + 1
126  index(4) = index(4) + 1
