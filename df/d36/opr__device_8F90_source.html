266  subroutine opr_device_dudxyz(du, u, dr, ds, dt, coef)
267  type(coef_t), intent(in), target :: coef
268  real(kind=rp), dimension(coef%Xh%lx,coef%Xh%ly, & coef%Xh%lz,coef%msh%nelv), intent(inout) :: du
269  real(kind=rp), dimension(coef%Xh%lx,coef%Xh%ly, & coef%Xh%lz,coef%msh%nelv), intent(in) :: u, dr, ds, dt
270  type(c_ptr) :: du_d, u_d, dr_d, ds_d, dt_d
271 
272  du_d = device_get_ptr(du)

274 
275  dr_d = device_get_ptr(dr)
276  ds_d = device_get_ptr(ds)

287  msh%nelv, xh%lx)
288 #elif HAVE_OPENCL
289  call opencl_dudxyz(du_d, u_d, dr_d, ds_d, dt_d, &
290  xh%dx_d, xh%dy_d, xh%dz_d, coef%jacinv_d, &
291  msh%nelv, xh%lx)
292 #else
293  call neko_error('No device backend configured')
294 #endif
295  end associate
