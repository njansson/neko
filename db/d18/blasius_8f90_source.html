99  subroutine blasius_apply_vector(this, x, y, z, n)
100  class(blasius_t), intent(inout) :: this
101  integer, intent(in) :: n
102  real(kind=rp), intent(inout), dimension(n) :: x
103  real(kind=rp), intent(inout), dimension(n) :: y
104  real(kind=rp), intent(inout), dimension(n) :: z
105  integer :: i, m, k, idx(4), facet
106 
107  associate(xc => this%c%dof%x, yc => this%c%dof%y, zc => this%c%dof%z, &
108  nx => this%c%nx, ny => this%c%ny, nz => this%c%nz, &
109  lx => this%c%Xh%lx)
110  m = this%msk(0)
111  do i = 1, m
112  k = this%msk(i)
113  facet = this%facet(i)
114  idx = nonlinear_index(k, lx, lx, lx)
115  select case(facet)
116  case(1,2)
117  x(k) = this%bla(zc(idx(1), idx(2), idx(3), idx(4)), &
118  this%delta, this%x(1))
119  y(k) = 0.0_rp
120  z(k) = 0.0_rp
121  case(3,4)
122  x(k) = 0.0_rp
123  y(k) = this%bla(xc(idx(1), idx(2), idx(3), idx(4)), &
124  this%delta, this%x(2))
125  z(k) = 0.0_rp
126  case(5,6)
127  x(k) = 0.0_rp
128  y(k) = 0.0_rp
129  z(k) = this%bla(yc(idx(1), idx(2), idx(3), idx(4)), &
130  this%delta, this%x(3))
131  end select
132  end do
133  end associate
134  end subroutine blasius_apply_vector

137  subroutine blasius_apply_vector_dev(this, x_d, y_d, z_d)
138  class(blasius_t), intent(inout), target :: this
139  type(c_ptr) :: x_d
140  type(c_ptr) :: y_d
141  type(c_ptr) :: z_d
142  integer :: i, m, k, idx(4), facet
143  integer(c_size_t) :: s
144  real(kind=rp), allocatable :: bla_x(:), bla_y(:), bla_z(:)
145 
146  associate(xc => this%c%dof%x, yc => this%c%dof%y, zc => this%c%dof%z, &
147  nx => this%c%nx, ny => this%c%ny, nz => this%c%nz, &
148  lx => this%c%Xh%lx, blax_d => this%blax_d, blay_d => this%blay_d, &
149  blaz_d => this%blaz_d)
150 
151  m = this%msk(0)
152 
153 
154  ! Pretabulate values during first call to apply
155  if (.not. c_associated(blax_d)) then
156  allocate(bla_x(m), bla_y(m), bla_z(m)) ! Temp arrays
157 
158  if (rp .eq. real32) then
159  s = m * 4
160  else if (rp .eq. real64) then
161  s = m * 8
162  end if
163 
164  call device_alloc(blax_d, s)
165  call device_alloc(blay_d, s)
166  call device_alloc(blaz_d, s)
167 
168  do i = 1, m
169  k = this%msk(i)
170  facet = this%facet(i)
171  idx = nonlinear_index(k, lx, lx, lx)
172  select case(facet)
173  case(1,2)
174  bla_x(i) = this%bla(zc(idx(1), idx(2), idx(3), idx(4)), &
175  this%delta, this%x(1))
176  bla_y(i) = 0.0_rp
177  bla_z(i) = 0.0_rp
178  case(3,4)
179  bla_x(i) = 0.0_rp
180  bla_y(i) = this%bla(xc(idx(1), idx(2), idx(3), idx(4)), &
181  this%delta, this%x(2))
182  bla_z(i) = 0.0_rp
183  case(5,6)
184  bla_x(i) = 0.0_rp
185  bla_y(i) = 0.0_rp
186  bla_z(i) = this%bla(yc(idx(1), idx(2), idx(3), idx(4)), &
187  this%delta, this%x(3))
188  end select
189  end do
190 
191  call device_memcpy(bla_x, blax_d, m, host_to_device)
192  call device_memcpy(bla_y, blay_d, m, host_to_device)
193  call device_memcpy(bla_z, blaz_d, m, host_to_device)
194 
195  deallocate(bla_x, bla_y, bla_z)
196  end if
197 
198  call device_inhom_dirichlet_apply_vector(this%msk_d, x_d, y_d, z_d, &
199  blax_d, blay_d, blaz_d, m)
200 
201  end associate
202 
203  end subroutine blasius_apply_vector_dev
