48  subroutine neko_solve(C)
49  type(case_t), intent(inout) :: c
50  real(kind=rp) :: t, cfl
51  real(kind=dp) :: start_time_org, start_time, end_time
52  character(len=LOG_SIZE) :: log_buf
53  integer :: tstep
54 
55  t = 0d0
56  tstep = 0
57  call neko_log%section('Simulation')
58  write(log_buf,'(A, E15.7,A,E15.7,A)') 'T : [', 0d0,',',c%params%T_end,')'
59  call neko_log%message(log_buf)
60  write(log_buf,'(A, E15.7)') 'dt : ', c%params%dt
61  call neko_log%message(log_buf)
62 
63  if (len_trim(c%params%restart_file) .gt. 0) then
64  call simulation_restart(c, t)
65  end if
66 
67  call c%usr%user_init_modules(t, c%fluid%u, c%fluid%v, c%fluid%w,&
68  c%fluid%p, c%fluid%c_Xh, c%params)
69 
70 
71  call neko_log%newline()
72 
73  start_time_org = mpi_wtime()
74  do while (t .lt. c%params%T_end .and. (.not. jobctrl_time_limit()))
75  tstep = tstep + 1
76  start_time = mpi_wtime()
77  cfl = c%fluid%compute_cfl(c%params%dt)
78  call neko_log%status(t, c%params%T_end)
79  write(log_buf, '(A,I6)') 'Time-step: ', tstep
80  call neko_log%message(log_buf)
81  call neko_log%begin()
82 
83  write(log_buf, '(A,E15.7,1x,A,E15.7)') 'CFL:', cfl, 'dt:', c%params%dt
84  call neko_log%message(log_buf)
85 
86 
87  call simulation_settime(t, c%params%dt, c%ab_bdf, c%tlag, c%dtlag, tstep)
88  call neko_log%section('Fluid')
89  call c%fluid%step(t, tstep, c%ab_bdf)
90  end_time = mpi_wtime()
91  write(log_buf, '(A,E15.7,A,E15.7)') &
92  'Elapsed time (s):', end_time-start_time_org, ' Step time:', &
93  end_time-start_time
94  call neko_log%end_section(log_buf)
95  call c%usr%usr_chk(t, c%params%dt, tstep,&
96  c%fluid%u, c%fluid%v, c%fluid%w, c%fluid%p, c%fluid%c_Xh)
97  call neko_log%end()
98  call c%q%eval(t, c%params%dt)
99  call c%s%sample(t)
100  end do
101 
102  if (t .lt. c%params%T_end) then
103  call simulation_joblimit_chkp(c, t)
104  end if
105 
106  call neko_log%end_section('normal end.')
107 
108  end subroutine neko_solve
