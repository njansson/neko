139  function cg_device_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
140  class(cg_device_t), intent(inout) :: this
141  class(ax_t), intent(inout) :: ax
142  type(field_t), intent(inout) :: x
143  integer, intent(inout) :: n
144  real(kind=rp), dimension(n), intent(inout) :: f
145  type(coef_t), intent(inout) :: coef
146  type(bc_list_t), intent(inout) :: blst
147  type(gs_t), intent(inout) :: gs_h
148  type(ksp_monitor_t) :: ksp_results
149  integer, optional, intent(in) :: niter
150  real(kind=rp), parameter :: one = 1.0
151  real(kind=rp), parameter :: zero = 0.0
152  integer :: iter, max_iter
153  real(kind=rp) :: rnorm, rtr, rtr0, rtz2, rtz1
154  real(kind=rp) :: beta, pap, alpha, alphm, norm_fac
155  type(c_ptr) :: f_d
156 
157  f_d = device_get_ptr(f)
158 
159  if (present(niter)) then
160  max_iter = niter
161  else
162  max_iter = ksp_max_iter
163  end if
164  norm_fac = one/sqrt(coef%volume)
165 
166  rtz1 = one
167  call device_rzero(x%x_d, n)
168  call device_rzero(this%p_d, n)
169  call device_copy(this%r_d, f_d, n)
170 
171  rtr = device_glsc3(this%r_d, coef%mult_d, this%r_d, n)
172  rnorm = sqrt(rtr)*norm_fac
173  ksp_results%res_start = rnorm
174  ksp_results%res_final = rnorm
175  ksp_results%iter = 0
176  if(rnorm .eq. zero) return
177  do iter = 1, max_iter
178  call this%M%solve(this%z, this%r, n)
179  rtz2 = rtz1
180  rtz1 = device_glsc3(this%r_d, coef%mult_d, this%z_d, n)
181  beta = rtz1 / rtz2
182  if (iter .eq. 1) beta = zero
183  call device_add2s1(this%p_d, this%z_d, beta, n)
184 
185  call ax%compute(this%w, this%p, coef, x%msh, x%Xh)
186  call gs_op(gs_h, this%w, n, gs_op_add)
187  call bc_list_apply(blst, this%w, n)
188 
189  pap = device_glsc3(this%w_d, coef%mult_d, this%p_d, n)
190 
191  alpha = rtz1 / pap
192  alphm = -alpha
193  call device_add2s2(x%x_d, this%p_d, alpha, n)
194  call device_add2s2(this%r_d, this%w_d, alphm, n)
195 
196  rtr = device_glsc3(this%r_d, coef%mult_d, this%r_d, n)
197  if (iter .eq. 1) rtr0 = rtr
198  rnorm = sqrt(rtr)*norm_fac
199  if (rnorm .lt. this%abs_tol) then
200  exit
201  end if
202  end do
203  ksp_results%res_final = rnorm
204  ksp_results%iter = iter
205 
206  end function cg_device_solve
