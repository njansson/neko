48  subroutine precon_factory(pc, pctype)
49  class(pc_t), target, allocatable, intent(inout) :: pc
50  character(len=*) :: pctype
51 
52  if (allocated(pc)) then
53  call precon_destroy(pc)
54  deallocate(pc)
55  end if
56 
57  if (trim(pctype) .eq. 'jacobi') then
58  if (neko_bcknd_sx .eq. 1) then
59  allocate(sx_jacobi_t::pc)
60  else if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
61  (neko_bcknd_opencl .eq. 1)) then
62  allocate(device_jacobi_t::pc)
63  else
64  allocate(jacobi_t::pc)
65  end if
66  else if (pctype(1:4) .eq. 'hsmg') then
67  allocate(hsmg_t::pc)
68  else if(trim(pctype) .eq. 'ident') then
69  if ((neko_bcknd_hip .eq. 1) .or. (neko_bcknd_cuda .eq. 1) .or. &
70  (neko_bcknd_opencl .eq. 1)) then
71  allocate(device_ident_t::pc)
72  else
73  allocate(ident_t::pc)
74  end if
75  else
76  call neko_error('Unknown preconditioner')
77  end if
78 
79  end subroutine precon_factory

82  subroutine precon_destroy(pc)
83  class(pc_t), allocatable, intent(inout) :: pc
84 
85  if (allocated(pc)) then
86  select type(pcp => pc)
87  type is(jacobi_t)
88  call pcp%free()
89  type is(sx_jacobi_t)
90  call pcp%free()
91  type is(device_jacobi_t)
92  call pcp%free()
93  type is (hsmg_t)
94  call pcp%free()
95  end select
96  end if
97 
98  end subroutine precon_destroy
