111  subroutine sampler_add(this, out)

115

129  subroutine sampler_sample(this, t)
130  class(sampler_t), intent(inout) :: this
131  real(kind=rp), intent(in) :: t
132  real(kind=dp) :: sample_start_time, sample_end_time
133  real(kind=dp) :: sample_time
134  character(len=LOG_SIZE) :: log_buf
135  integer :: i, ierr
136 
137  if (t .ge. (this%nsample * this%T)) then
138 
139  call mpi_barrier(neko_comm, ierr)
140  sample_start_time = mpi_wtime()
141 
142  ! We should not need this extra select block, and it works great
143  ! without it for GNU, Intel and NEC, but breaks horribly on Cray
144  ! (>11.0.x) when using high opt. levels.
145  select type (samp => this)
146  type is (sampler_t)
147  do i = 1, this%n
148  call samp%output_list(i)%outp%sample(t)
149  end do
150  class default
151  call neko_error('Invalid sampler output list')
152  end select
153 
154  call mpi_barrier(neko_comm, ierr)
155  sample_end_time = mpi_wtime()
156  this%nsample = this%nsample + 1
157 
158  sample_time = sample_end_time - sample_start_time
159  write(log_buf,'(A24,1x,F10.6,A,F9.6)') 'Sampling fields at time:', t, &
160  ' Sample time (s): ', sample_time
161  call neko_log%message(log_buf)
162 
163  end if
164 
165  end subroutine sampler_sample
