75  subroutine facet_normal_apply_surfvec(this, x, y, z, u, v, w, n)
76  class(facet_normal_t), intent(inout) :: this
77  integer, intent(in) :: n
78  real(kind=rp), intent(inout), dimension(n) :: x
79  real(kind=rp), intent(inout), dimension(n) :: y
80  real(kind=rp), intent(inout), dimension(n) :: z
81  real(kind=rp), intent(inout), dimension(n) :: u
82  real(kind=rp), intent(inout), dimension(n) :: v
83  real(kind=rp), intent(inout), dimension(n) :: w
84  integer :: i, m, k, idx(4), facet
85 
86  if (.not. associated(this%c)) then
87  call neko_error('No coefficients assigned')
88  end if
89  associate(c => this%c)
90  m = this%msk(0)
91  do i = 1, m
92  k = this%msk(i)
93  facet = this%facet(i)
94  idx = nonlinear_index(k, c%Xh%lx, c%Xh%lx, c%Xh%lx)
95  select case(facet)
96  case(1,2)
97  x(k) = u(k) * c%nx(idx(2), idx(3), facet, idx(4)) &
98  * c%area(idx(2), idx(3), facet, idx(4))
99  y(k) = v(k) * c%ny(idx(2), idx(3), facet, idx(4)) &
100  * c%area(idx(2), idx(3), facet, idx(4))
101  z(k) = w(k) * c%nz(idx(2), idx(3), facet, idx(4)) &
102  * c%area(idx(2), idx(3), facet, idx(4))
103  case(3,4)
104  x(k) = u(k) * c%nx(idx(1), idx(3), facet, idx(4)) &
105  * c%area(idx(1), idx(3), facet, idx(4))
106  y(k) = v(k) * c%ny(idx(1), idx(3), facet, idx(4)) &
107  * c%area(idx(1), idx(3), facet, idx(4))
108  z(k) = w(k) * c%nz(idx(1), idx(3), facet, idx(4)) &
109  * c%area(idx(1), idx(3), facet, idx(4))
110  case(5,6)
111  x(k) = u(k) * c%nx(idx(1), idx(2), facet, idx(4)) &
112  * c%area(idx(1), idx(2), facet, idx(4))
113  y(k) = v(k) * c%ny(idx(1), idx(2), facet, idx(4)) &
114  * c%area(idx(1), idx(2), facet, idx(4))
115  z(k) = w(k) * c%nz(idx(1), idx(2), facet, idx(4)) &
116  * c%area(idx(1), idx(2), facet, idx(4))
117  end select
118  end do
119  end associate
120 
121  end subroutine facet_normal_apply_surfvec
