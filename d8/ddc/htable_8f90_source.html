229  subroutine htable_init(this, size, key, data)
230  class(htable_t), intent(inout) :: this
231  integer, value :: size
232  class(*), target, intent(in) :: key
233  class(*), target, intent(in), optional :: data
234  class(*), pointer :: dp
235  integer :: i
236 
237 
238  call htable_free(this)
239 
240  if (size .lt. 4) then
241  size = 4
242  end if
243 
244  size = ishft(1, ceiling(log(dble(size)) / neko_m_ln2))
245 
246  allocate(this%t(0:size))
247  this%t(:)%valid = .false.
248  this%size = size
249  this%entries = 0
250 
251  dp => key
252  if (present(data)) then
253  dp => data
254  end if
255 
256  do i = 0, size
257  allocate(this%t(i)%key, source=key)
258  allocate(this%t(i)%data, source=dp)
259  end do
260  end subroutine htable_init

301  recursive subroutine htable_set(this, key, data)
302  class(htable_t), intent(inout) :: this
303  class(*), intent(inout) :: key
304  class(*), intent(inout) :: data
305  class(htable_t), allocatable :: tmp
306  integer index, i
307 
308  index = this%hash(key)
309  if (index .lt. 0) then
310  call neko_error("Invalid hash generated")
311  end if
312 
313  i = this%size - 1
314 
315  do while (i .ge. 0)
316  if ((.not. this%t(index)%valid) .or. &
317  htable_eq_key(this, index, key)) then
318  call htable_set_key(this, index, key)
319  call htable_set_data(this, index, data)
320  if (.not. this%t(index)%valid) then
321  this%entries = this%entries + 1
322  end if
323  this%t(index)%valid = .true.
324  this%t(index)%skip = .false.
325  return
326  end if
327  index = modulo((index + 1), this%size)
328  i = i - 1
329  end do
330 
331  select type(key)
332  type is (integer)
333  allocate(htable_i4_t::tmp)
334  type is (integer(i8))
335  allocate(htable_i8_t::tmp)
336  type is (double precision)
337  allocate(htable_r8_t::tmp)
338  type is (point_t)
339  allocate(htable_pt_t::tmp)
340  type is (tuple_i4_t)
341  allocate(htable_i4t2_t::tmp)
342  type is (tuple4_i4_t)
343  allocate(htable_i4t4_t::tmp)
344  type is (h_cptr_t)
345  allocate(htable_cptr_t::tmp)
346  class default
347  call neko_error('Invalid htable key')
348  end select
349 
350  call htable_init(tmp, ishft(this%size, 1), key, data)
351 
352  do i = 0, this%size - 1
353  if (this%t(i)%valid) then
354  call htable_set(tmp, this%t(i)%key, this%t(i)%data)
355  end if
356  end do
357  this%size = tmp%size
358  call move_alloc(tmp%t, this%t)
359 
360  call htable_set(this, key, data)
361 
362  end subroutine htable_set

365  function htable_get(this, key, data) result(rcode)
366  class(htable_t), intent(inout) :: this
367  class(*), intent(inout) :: key
368  class(*), intent(inout) :: data
369  integer :: rcode
370  integer :: index, i
371 
372  index = this%hash(key)
373  if (index .lt. 0) then
374  call neko_error("Invalid hash generated")
375  end if
376 
377  i = this%size - 1
378 
379  do while (i .ge. 0)
380  if (.not. this%t(index)%valid .and. &
381  .not. this%t(index)%skip) then
382  rcode = 1
383  return
384  else if ((this%t(index)%valid) .and. &
385  htable_eq_key(this, index, key)) then
386  call htable_get_data(this, index, data)
387  rcode = 0
388  return
389  end if
390  index = modulo((index + 1), this%size)
391  i = i - 1
392  end do
393  rcode = 1
394  end function htable_get
